38void CN3GESnow::Tick()
39{
40 if (m_bActive == FALSE || m_iVC <= 0 || m_pVB == nullptr) return;
41 CN3GlobalEffect::Tick();
42
43 int iCount = m_iVC/3;
44 int iActiveCount = iCount;
45 if(m_iFadeMode > 0) // 차차 많아지게 한다..
46 {
47 if(m_fFadeTime > 0 && m_fFadeTimeCur < m_fFadeTime) // Fade시간땜에 건너뛰고 찍을 양 결정..
48 {
49 iActiveCount = (int)(iCount * m_fFadeTimeCur / m_fFadeTime);
50 if(iActiveCount > iCount) iActiveCount = iCount;
51 else if(iActiveCount < 0) iActiveCount = 0;
52 }
53 }
54 else if(m_iFadeMode < 0)
55 {
56 if(m_fFadeTime > 0 && m_fFadeTimeCur < m_fFadeTime) // Fade시간땜에 건너뛰고 찍을 양 결정..
57 {
58 iActiveCount = (int)(iCount * (1.0f - m_fFadeTimeCur / m_fFadeTime));
59 if(iActiveCount > iCount) iActiveCount = iCount;
60 else if(iActiveCount < 0) iActiveCount = 0;
61 }
62 }
63 if(iActiveCount <= 0) return;
64
65 int i;
66 __VertexXyzT1* pVertices;
67 HRESULT hr = m_pVB->Lock(0, 0, (void**)&pVertices, D3DLOCK_NOSYSLOCK);
68 if (FAILED(hr))
69 return;
70
71 __Vector3 vN = m_vVelocity; vN.Normalize();
72 __Vector3 vAdd = m_vVelocity*s_fSecPerFrm;
73 float fAddRadian = __PI*s_fSecPerFrm*0.1f;
74
75 const float fHalfWidth = m_fWidth/2.0f;
76 const float fHalfHeight = m_fHeight/2.0f;
77 const float fCurY = s_CameraData.vEye.y;
78
79 static const float sqrt3 = sqrtf(3.0f);
80
81 for (i=0; i<iActiveCount; ++i)
82 {
83 // 위치 결정하기
84 __VertexXyzT1* pV1 = pVertices+i*3+0;
85 __VertexXyzT1* pV2 = pVertices+i*3+1;
86 __VertexXyzT1* pV3 = pVertices+i*3+2;
87// __Vector3 vOffset2 = __Vector3(pV2->v) - __Vector3(pV1->v);
88// __Vector3 vOffset3 = __Vector3(pV3->v) - __Vector3(pV1->v);
89
90 __SnowParticle* pParticle = m_pSnowParticle + i;
91 pParticle->vPos += vAdd;
92// pV1->x += vAdd.x; pV1->y += vAdd.y; pV1->z += vAdd.z;
93
94 float fDiff = pParticle->vPos.y - (fCurY-fHalfHeight);
95 if (fDiff < 0) // 높이 범위를 벗어났을 경우
96 {
97 pParticle->vPos.y -= (((int)(fDiff/m_fHeight)-1)*m_fHeight);
98 pParticle->vPos.x = m_fWidth*(rand()%10000-5000)/10000.f;
99 pParticle->vPos.z = m_fWidth*(rand()%10000-5000)/10000.f;
100
101 pParticle->fRadius = (rand()%10000)/10000.f;
102 pParticle->fRadian = 2*__PI*((rand()%10000)/10000.f);
103
104 float fRadian = __PI*((rand()%10000)/10000.f);
105 pParticle->vOffset1.Set(0, sqrt3*m_fSnowSize/3.f, 0);
106 pParticle->vOffset2.Set(cosf(fRadian)*m_fSnowSize/2, -sqrt3*m_fSnowSize/6.f, sinf(fRadian)*m_fSnowSize/2);
107 pParticle->vOffset3.Set(-cosf(fRadian)*m_fSnowSize/2, -sqrt3*m_fSnowSize/6.f, -sinf(fRadian)*m_fSnowSize/2);
108 }
109 else
110 {
111 fDiff = pParticle->vPos.y - (fCurY+fHalfHeight);
112 if (fDiff > 0) // 높이 범위를 반대로 벗어났을경우
113 pParticle->vPos.y -= ((int)(fDiff/m_fHeight)+1)*m_fHeight;
114 // x 너비 범위를 벗어났을 경우
115 fDiff = pParticle->vPos.x - fHalfWidth;
116 if (fDiff > 0) pParticle->vPos.x -= ((int)(fDiff/m_fWidth)+1)*m_fWidth;
117 fDiff = pParticle->vPos.x + fHalfWidth;
118 if (fDiff < 0) pParticle->vPos.x -= ((int)(fDiff/m_fWidth)-1)*m_fWidth;
119 // z 너비 범위를 벗어났을 경우
120 fDiff = pParticle->vPos.z - fHalfWidth;
121 if (fDiff > 0) pParticle->vPos.z -= ((int)(fDiff/m_fWidth)+1)*m_fWidth;
122 fDiff = pParticle->vPos.z + fHalfWidth;
123 if (fDiff < 0) pParticle->vPos.z -= ((int)(fDiff/m_fWidth)-1)*m_fWidth;
124 }
125 // 중심축을 주위로 회전한 위치 계산
126 pParticle->fRadian += fAddRadian;
127 __Vector3 vPos; vPos.Set(cosf(pParticle->fRadian), 0, sinf(pParticle->fRadian));
128 vPos += pParticle->vPos;
129
130 // 버텍스 버퍼의 점 다시 세팅하기
131 pV1->x = vPos.x + pParticle->vOffset1.x; pV1->y = vPos.y + pParticle->vOffset1.y; pV1->z = vPos.z + pParticle->vOffset1.z;
132 pV2->x = vPos.x + pParticle->vOffset2.x; pV2->y = vPos.y + pParticle->vOffset2.y; pV2->z = vPos.z + pParticle->vOffset2.z;
133 pV3->x = vPos.x + pParticle->vOffset3.x; pV3->y = vPos.y + pParticle->vOffset3.y; pV3->z = vPos.z + pParticle->vOffset3.z;
134 }
135 m_pVB->Unlock();

211{

228 if (FAILED(hr)) return;

230 hr = m_pVB->Lock(0, iSnowCount*3*sizeof(__VertexXyzT1), (void**)&pVertices, D3DLOCK_NOSYSLOCK);

233 // __SnowParticle 정보 채워 넣기
234 m_pSnowParticle = new __SnowParticle[iSnowCount];

236 const float sqrt3 = sqrtf(3.0f);
