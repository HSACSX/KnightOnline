153 __CellSub* SubCell(float fX, float fZ)
154 {
155 int x = (int) (fX / CELL_MAIN_SIZE);
156 int z = (int) (fZ / CELL_MAIN_SIZE);
157
158 __ASSERT(x >= 0 && x < MAX_CELL_MAIN && z >= 0 && z < MAX_CELL_MAIN, "Invalid cell number");
159 if (x < 0 || x >= MAX_CELL_MAIN || z < 0 || z >= MAX_CELL_MAIN)
160 return nullptr;
161
162 if (nullptr == m_pCells[x][z])
163 return nullptr;
164
165 int xx = (((int) fX) % CELL_MAIN_SIZE) / CELL_SUB_SIZE;
166 int zz = (((int) fZ) % CELL_MAIN_SIZE) / CELL_SUB_SIZE;
167
168 return &m_pCells[x][z]->SubCells[xx][zz];
169 }
