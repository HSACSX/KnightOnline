64 float fAspect = 0.0f; // 종횡비

72 float fFrustum[6][4] = {};
73
74 // fRadius - 물체의 반지름보다 약간 더 여유 있게 잡고 넣으면 그만큼 클리핑을 여유있게 한다..
75 // 미리 계산된 카메라 평면의 도트 프로덕트 값을 기준으로 카메라 사면체 밖에 있으면 참을 돌려준다.
76 bool IsOutOfFrustum(const __Vector3& vPosition, float fRadius) const
77 {
78 if ((vEye - vPosition).Magnitude() > fFP + fRadius)
79 return true; // Far Plane 거리체크
80
81 for (int p = 0; p < 6; p++)
82 {
83 if (fFrustum[p][0] * vPosition.x + fFrustum[p][1] * vPosition.y
84 + fFrustum[p][2] * vPosition.z + fFrustum[p][3]
85 <= -fRadius)
86 return true;
87 }
88
89 return false;
90 }
