57{
58 if (this == &fxPMesh)

61 Release();
62
63 FileNameSet(fxPMesh.FileName());
64
65 m_iMaxNumVertices = fxPMesh.m_iMaxNumVertices;

92 m_pCollapses = new __EdgeCollapse[m_iNumCollapses + 1];

112bool CN3FXPMesh::Load(File& file)
113{
114 // NOLINTNEXTLINE(bugprone-parent-virtual-call)
115 CN3BaseFileAccess::Load(file);
116
117 file.Read(&m_iNumCollapses, sizeof(m_iNumCollapses));
118 file.Read(&m_iTotalIndexChanges, sizeof(m_iTotalIndexChanges));
119
120 file.Read(&m_iMaxNumVertices, sizeof(int));
121 file.Read(&m_iMaxNumIndices, sizeof(int));
122 file.Read(&m_iMinNumVertices, sizeof(int));
123 file.Read(&m_iMinNumIndices, sizeof(int));
124
125 HRESULT hr = Create(m_iMaxNumVertices, m_iMaxNumIndices);
126 __ASSERT(SUCCEEDED(hr), "Failed to create progressive mesh");
127
128 if (FAILED(hr))
129 {
130 // Just ensure that the file is consistently read, in case this isn't checked,
131 // which is unfortunately common with tooling.
132
133 if (m_iMaxNumVertices > 0)
134 file.Seek(m_iMaxNumVertices * sizeof(__VertexT1), SEEK_CUR);
135
136 if (m_iMaxNumIndices > 0)
137 file.Seek(m_iMaxNumIndices * sizeof(uint16_t), SEEK_CUR);
138
139 if (m_iNumCollapses > 0)
140 file.Seek(m_iNumCollapses * sizeof(__EdgeCollapse), SEEK_CUR);
141
142 if (m_iTotalIndexChanges > 0)
143 file.Seek(m_iTotalIndexChanges * sizeof(int), SEEK_CUR);
144
145 file.Read(&m_iLODCtrlValueCount, sizeof(m_iLODCtrlValueCount));
146
147 if (m_iLODCtrlValueCount > 0)
148 file.Seek(m_iLODCtrlValueCount * sizeof(__LODCtrlValue), SEEK_CUR);
149
150 return false;
151 }
152
153 if (m_iMaxNumVertices > 0)
154 {
155 file.Read(m_pVertices, m_iMaxNumVertices * sizeof(__VertexT1));
156 for (int i = 0; i < m_iMaxNumVertices; i++)
157 {
158 m_pColorVertices[i].x = m_pVertices[i].x;
159 m_pColorVertices[i].y = m_pVertices[i].y;
160 m_pColorVertices[i].z = m_pVertices[i].z;
161 m_pColorVertices[i].color = 0xffffffff;
162 m_pColorVertices[i].tu = m_pVertices[i].tu;
163 m_pColorVertices[i].tv = m_pVertices[i].tv;
164 }
165
166 delete[] m_pVertices;
167 m_pVertices = nullptr;
168 }
169
170 if (m_iMaxNumIndices > 0)
171 file.Read(m_pIndices, m_iMaxNumIndices * sizeof(uint16_t));
172
173 if (m_iNumCollapses > 0)
174 {
175 // +1을 한 이유 : PMeshInstance::SplitOne() 함수에서 부득이하게 포인터가 경계선을 가르키게 해야 하는 경우가 있어서.
176 m_pCollapses = new __EdgeCollapse[m_iNumCollapses + 1];
177 file.Read(m_pCollapses, m_iNumCollapses * sizeof(__EdgeCollapse));
178
179 // 위의 +1을 한이유와 같음. 만약을 대비해 마지막 데이타를 초기화 해둠
180 // NOLINTNEXTLINE(cppcoreguidelines-pro-bounds-pointer-arithmetic)
181 memset(m_pCollapses + m_iNumCollapses, 0, sizeof(__EdgeCollapse));
182
183#ifdef _DEBUG
184 bool bFixed = false;
185#endif
186 for (int i = 0; i < m_iNumCollapses; i++)
187 {
188 if (m_pCollapses[i].iIndexChanges < 0)
189 {
190 m_pCollapses[i].iIndexChanges = 0;
191#ifdef _DEBUG
192 bFixed = true;
193#endif
194 }
195 }
196
197#ifdef _DEBUG
198 if (bFixed)
199 ::MessageBox(s_hWndBase, "잘못된 Progressive Mesh 수정", m_szName.c_str(), MB_OK);
200#endif
201 }
202
203 if (m_iTotalIndexChanges > 0)
204 {
205 m_pAllIndexChanges = new int[m_iTotalIndexChanges];
206 file.Read(m_pAllIndexChanges, m_iTotalIndexChanges * sizeof(int));
207 }
208
209 __ASSERT(m_pLODCtrlValues == nullptr && m_iLODCtrlValueCount == 0,
210 "Invalid Level of detail control value");
211
212 file.Read(&m_iLODCtrlValueCount, sizeof(m_iLODCtrlValueCount));
213 if (m_iLODCtrlValueCount > 0)
214 {
215 m_pLODCtrlValues = new __LODCtrlValue[m_iLODCtrlValueCount];
216 file.Read(m_pLODCtrlValues, m_iLODCtrlValueCount * sizeof(__LODCtrlValue));
217 }
218
219 FindMinMax();
220
221 return true;
222}

245void CN3FXPMesh::Render()
246{
247 s_lpD3DDev->SetFVF(FVF_XYZCOLORT1);
248
249 const int iPCToRender = 1000; // primitive count to render
250 if (m_iMaxNumIndices > 3)
251 {
252 int iPC = m_iMaxNumIndices / 3;
253
254 int iLC = iPC / iPCToRender;
255 int i = 0;
256 for (i = 0; i < iLC; ++i)
257 {
258 s_lpD3DDev->DrawIndexedPrimitiveUP(D3DPT_TRIANGLELIST, 0, m_iMaxNumVertices,
259 iPCToRender, m_pIndices + i * iPCToRender * 3, D3DFMT_INDEX16, m_pColorVertices,
260 sizeof(__VertexXyzColorT1));
261 }
262
263 int iRPC = iPC % iPCToRender;
264 if (iRPC > 0)
265 s_lpD3DDev->DrawIndexedPrimitiveUP(D3DPT_TRIANGLELIST, 0, m_iMaxNumVertices, iRPC,
266 m_pIndices + i * iPCToRender * 3, D3DFMT_INDEX16, m_pColorVertices,
267 sizeof(__VertexXyzColorT1));
268 }

271void CN3FXPMesh::FindMinMax()
272{
273 if (m_iMaxNumVertices <= 0)
274 {
275 m_vMin.Zero();
276 m_vMax.Zero();
277 m_fRadius = 0;
278 return;
279 }
280
281 // 최소, 최대 점을 찾는다.
282 m_vMin.Set(FLT_MAX, FLT_MAX, FLT_MAX);
283 m_vMax.Set(-FLT_MAX, -FLT_MAX, -FLT_MAX);
284
285 for (int i = 0; i < m_iMaxNumVertices; i++)
286 {
287 if (m_pColorVertices[i].x < m_vMin.x)
288 m_vMin.x = m_pColorVertices[i].x;
289 if (m_pColorVertices[i].y < m_vMin.y)
290 m_vMin.y = m_pColorVertices[i].y;
291 if (m_pColorVertices[i].z < m_vMin.z)
292 m_vMin.z = m_pColorVertices[i].z;
293
294 if (m_pColorVertices[i].x > m_vMax.x)
295 m_vMax.x = m_pColorVertices[i].x;
296 if (m_pColorVertices[i].y > m_vMax.y)
297 m_vMax.y = m_pColorVertices[i].y;
298 if (m_pColorVertices[i].z > m_vMax.z)
299 m_vMax.z = m_pColorVertices[i].z;
300 }
301
302 // 최대 최소값을 갖고 반지름 계산한다..
303 m_fRadius = (m_vMax - m_vMin).Magnitude() * 0.5f;
304}
