50AIServerApp::~AIServerApp()
51{
52 spdlog::info("AIServerApp::~AIServerApp: Shutting down, releasing resources.");
53 _socketManager.Shutdown();
54 spdlog::info("AIServerApp::~AIServerApp: SocketManager stopped.");
55
56 // wait for all of these threads to be fully shut down.
57 spdlog::info("AIServerApp::~AIServerApp: Waiting for worker threads to fully shut down.");
58
59 if (_checkAliveThread != nullptr)
60 {
61 spdlog::info("AIServerApp::~AIServerApp: Shutting down CheckAliveThread...");
62
63 _checkAliveThread->shutdown();
64
65 spdlog::info("AIServerApp::~AIServerApp: CheckAliveThread stopped.");
66 }
67
68 if (!_npcThreads.empty())
69 {
70 spdlog::info(
71 "AIServerApp::~AIServerApp: Shutting down {} NPC threads...", _npcThreads.size());
72
73 for (CNpcThread* npcThread : _npcThreads)
74 npcThread->shutdown(false);
75
76 for (CNpcThread* npcThread : _npcThreads)
77 npcThread->join();
78
79 spdlog::info("AIServerApp::~AIServerApp: NPC threads stopped.");
80 }
81
82 if (_zoneEventThread != nullptr)
83 {
84 spdlog::info("AIServerApp::~AIServerApp: Shutting down ZoneEventThread...");
85
86 _zoneEventThread->shutdown();
87
88 spdlog::info("AIServerApp::~AIServerApp: ZoneEventThread stopped.");
89 }
90
91 spdlog::info("AIServerApp::~AIServerApp: All worker threads stopped, freeing caches.");
92
93 for (MAP* map : _zones)
94 delete map;
95 _zones.clear();
96
97 delete _zoneEventThread;
98 _zoneEventThread = nullptr;
99
100 for (CNpcThread* npcThread : _npcThreads)
101 delete npcThread;
102 _npcThreads.clear();
103
104 if (_npcItem.m_ppItem != nullptr)
105 {
106 for (int i = 0; i < _npcItem.m_nRow; i++)
107 {
108 delete[] _npcItem.m_ppItem[i];
109 _npcItem.m_ppItem[i] = nullptr;
110 }
111
112 delete[] _npcItem.m_ppItem;
113 _npcItem.m_ppItem = nullptr;
114 }
115
116 for (int i = 0; i < MAX_USER; i++)
117 {
118 delete _users[i];
119 _users[i] = nullptr;
120 }
121
122 spdlog::info("AIServerApp::~AIServerApp: All resources safely released.");
123
124 ConnectionManager::Destroy();
125}

447bool AIServerApp::GetNpcItemTable()
448{
449 using ModelType = model::MonsterItem;
450
451 std::vector<ModelType*> rows;
452
453 recordset_loader::Vector<ModelType> loader(rows);
454 if (!loader.Load_ForbidEmpty(true))
455 {
456 spdlog::error("AIServerApp::GetNpcItemTable: load failed - {}", loader.GetError().Message);
457 return false;
458 }
459
460 _npcItem.m_nField = loader.GetColumnCount();
461 _npcItem.m_nRow = static_cast<int>(rows.size());
462
463 if (rows.empty())
464 return false;
465
466 _npcItem.m_ppItem = new int*[_npcItem.m_nRow];
467 for (int i = 0; i < _npcItem.m_nRow; i++)
468 _npcItem.m_ppItem[i] = new int[_npcItem.m_nField];
469
470 for (size_t i = 0; i < rows.size(); i++)
471 {
472 ModelType* row = rows[i];
473
474 _npcItem.m_ppItem[i][0] = row->MonsterId;
475 _npcItem.m_ppItem[i][1] = row->ItemId[0];
476 _npcItem.m_ppItem[i][2] = row->DropChance[0];
477 _npcItem.m_ppItem[i][3] = row->ItemId[1];
478 _npcItem.m_ppItem[i][4] = row->DropChance[1];
479 _npcItem.m_ppItem[i][5] = row->ItemId[2];
480 _npcItem.m_ppItem[i][6] = row->DropChance[2];
481 _npcItem.m_ppItem[i][7] = row->ItemId[3];
482 _npcItem.m_ppItem[i][8] = row->DropChance[3];
483 _npcItem.m_ppItem[i][9] = row->ItemId[4];
484 _npcItem.m_ppItem[i][10] = row->DropChance[4];
485
486 delete row;
487 }
488
489 rows.clear();
490
491 spdlog::info("AIServerApp::GetNpcItemTable: K_MONSTER_ITEM loaded");
492 return true;
493}

560// Npc Thread 를 만든다.
561bool AIServerApp::CreateNpcThread()
562{
563 _totalNpcCount = 0; // DB에 있는 수
564 _loadedNpcCount = 0;
565
566 std::vector<model::NpcPos*> rows;
567 if (!LoadNpcPosTable(rows))
568 {
569 spdlog::error("AIServerApp::CreateNpcThread: K_NPCPOS load failed");
570 return false;
571 }
572
573 for (model::NpcPos* row : rows)
574 delete row;
575 rows.clear();
576
577 int step = 0;
578 int nThreadNumber = 0;
579 CNpcThread* pNpcThread = nullptr;
580
581 for (auto& [_, pNpc] : _npcMap)
582 {
583 if (step == 0)
584 pNpcThread = new CNpcThread();
585
586 pNpcThread->m_pNpc[step] = pNpc;
587 pNpcThread->m_pNpc[step]->m_sThreadNumber = nThreadNumber;
588 pNpcThread->m_pNpc[step]->Init();
589
590 ++step;
591
592 if (step == NPC_NUM)
593 {
594 pNpcThread->m_sThreadNumber = nThreadNumber++;
595 _npcThreads.push_back(pNpcThread);
596 step = 0;
597 }
598 }
599
600 if (step != 0)
601 {
602 pNpcThread->m_sThreadNumber = nThreadNumber++;
603 _npcThreads.push_back(pNpcThread);
604 }
605
606 if (_zoneEventThread == nullptr)
607 _zoneEventThread = new ZoneEventThread();
608
609 spdlog::info("AIServerApp::CreateNpcThread: Monsters/NPCs loaded: {}", _totalNpcCount);
610 return true;
611}

613bool AIServerApp::LoadNpcPosTable(std::vector<model::NpcPos*>& rows)
614{
615 CRoomEvent* pRoom = nullptr;
616
617 recordset_loader::Vector<model::NpcPos> loader(rows);
618 if (!loader.Load_ForbidEmpty(true))
619 {
620 spdlog::error("AIServerApp::LoadNpcPosTable: failed - {}", loader.GetError().Message);
621 return false;
622 }
623
624 int nSerial = _mapEventNpcCount;
625
626 for (model::NpcPos* row : rows)
627 {
628 bool bMoveNext = true;
629 int nPathSerial = 1;
630 int nNpcCount = 0;
631
632 do
633 {
634 int nMonsterNumber = row->NumNpc;
635 int nServerNum = GetServerNumber(row->ZoneId);
636
637 if (_serverZoneType == nServerNum || _serverZoneType == UNIFY_ZONE)
638 {
639 for (int j = 0; j < nMonsterNumber; j++)
640 {
641 CNpc* pNpc = new CNpc();
642 pNpc->m_sNid = nSerial++; // 서버 내에서의 고유 번호
643 pNpc->m_sSid = (int16_t) row->NpcId; // MONSTER(NPC) Serial ID
644
645 pNpc->m_byMoveType = row->ActType;
646 pNpc->m_byInitMoveType = row->ActType;
647 pNpc->m_byDirection = row->Direction;
648
649 model::Npc* pNpcTable = nullptr;
650
651 if (row->ActType >= 0 && row->ActType < 100)
652 {
653 pNpcTable = _monTableMap.GetData(pNpc->m_sSid);
654 }
655 else if (row->ActType >= 100)
656 {
657 pNpc->m_byMoveType = row->ActType - 100;
658 //pNpc->m_byInitMoveType = row->ActType - 100;
659
660 pNpcTable = _npcTableMap.GetData(pNpc->m_sSid);
661 }
662
663 pNpc->m_byBattlePos = 0;
664
665 if (pNpc->m_byMoveType >= 2)
666 {
667 pNpc->m_byBattlePos = myrand(1, 3);
668 pNpc->m_byPathCount = nPathSerial++;
669 }
670
671 pNpc->InitPos();
672
673 if (pNpcTable == nullptr)
674 {
675 spdlog::error(
676 "AIServerApp::LoadNpcPosTable: npc not found [serial={}, npcId={}]",
677 pNpc->m_sNid, pNpc->m_sSid);
678 break;
679 }
680
681 if (bMoveNext)
682 {
683 bMoveNext = false;
684 nNpcCount = row->NumNpc;
685 }
686
687 pNpc->Load(pNpcTable, true);
688
689 //////// MONSTER POS ////////////////////////////////////////
690 pNpc->m_sCurZone = row->ZoneId;
691
692 float fRandom_X = 0.0f, fRandom_Z = 0.0f;
693
694 // map에 몬스터의 위치를 랜덤하게 위치시킨다.. (테스트 용 : 수정-DB에서 읽어오는데로 몬 위치 결정)
695 int nRandom = abs(row->LeftX - row->RightX);
696 if (nRandom <= 1)
697 {
698 fRandom_X = (float) row->LeftX;
699 }
700 else
701 {
702 if (row->LeftX < row->RightX)
703 fRandom_X = (float) myrand(row->LeftX, row->RightX);
704 else
705 fRandom_X = (float) myrand(row->RightX, row->LeftX);
706 }
707
708 nRandom = abs(row->TopZ - row->BottomZ);
709 if (nRandom <= 1)
710 {
711 fRandom_Z = (float) row->TopZ;
712 }
713 else
714 {
715 if (row->TopZ < row->BottomZ)
716 fRandom_Z = (float) myrand(row->TopZ, row->BottomZ);
717 else
718 fRandom_Z = (float) myrand(row->BottomZ, row->TopZ);
719 }
720
721 pNpc->m_fCurX = fRandom_X;
722 pNpc->m_fCurY = 0;
723 pNpc->m_fCurZ = fRandom_Z;
724
725 if (row->RespawnTime < 15)
726 {
727 spdlog::warn("AIServerApp::LoadNpcPosTable: RegTime below minimum value of "
728 "15s [npcId={}, serial={}, npcName={}, RegTime={}]",
729 pNpc->m_sSid, pNpc->m_sNid + NPC_BAND, pNpc->m_strName,
730 row->RespawnTime);
731 // TODO: Set this to 15 in separate ticket and comment on it deviating from official behavior
732 row->RespawnTime = 30;
733 }
734
735 pNpc->m_sRegenTime = row->RespawnTime * 1000; // 초(DB)단위-> 밀리세컨드로
736
737 pNpc->m_sMaxPathCount = row->PathPointCount;
738
739 if (pNpc->m_byMoveType == 2 || pNpc->m_byMoveType == 3)
740 {
741 if (row->PathPointCount == 0 || !row->Path.has_value())
742 {
743 spdlog::error("AIServerApp::LoadNpcPosTable: NPC expects path to be "
744 "set [zoneId={} serial={}, npcId={}, npcName={}, "
745 "moveType={}, pathCount={}]",
746 row->ZoneId, pNpc->m_sNid + NPC_BAND, pNpc->m_sSid, pNpc->m_strName,
747 pNpc->m_byMoveType, pNpc->m_sMaxPathCount);
748 return false;
749 }
750 }
751
752 int index = 0;
753
754 if (row->PathPointCount != 0 && row->Path.has_value())
755 {
756 // The path is a series of points (x,z), each in the form ("%04d%04d", x, z)
757 // As such, we expect there to be at least 8 characters per point.
758 constexpr size_t CharactersPerPoint = 8;
759
760 const std::string& path = *row->Path;
761 if ((row->PathPointCount * CharactersPerPoint) > path.size())
762 {
763 spdlog::error("LoadNpcPosTable: NPC expects a larger path for this "
764 "PathPointCount [zoneId={} serial={} npcId={} npcName={} "
765 "moveType={}, pathCount={}]",
766 row->ZoneId, row->PathPointCount, pNpc->m_sNid + NPC_BAND,
767 pNpc->m_sSid, pNpc->m_strName, pNpc->m_byMoveType,
768 pNpc->m_sMaxPathCount);
769 return false;
770 }
771
772 for (int l = 0; l < row->PathPointCount; l++)
773 {
774 char szX[5] {}, szZ[5] {};
775 GetString(szX, path.c_str(), 4, index);
776 GetString(szZ, path.c_str(), 4, index);
777 pNpc->m_PathList.pPattenPos[l].x = atoi(szX);
778 pNpc->m_PathList.pPattenPos[l].z = atoi(szZ);
779 // TRACE(_T(" l=%d, x=%d, z=%d\n"), l, pNpc->m_PathList.pPattenPos[l].x, pNpc->m_PathList.pPattenPos[l].z);
780 }
781 }
782
783 pNpc->m_nInitMinX = pNpc->m_nLimitMinX = row->LeftX;
784 pNpc->m_nInitMinY = pNpc->m_nLimitMinZ = row->TopZ;
785 pNpc->m_nInitMaxX = pNpc->m_nLimitMaxX = row->RightX;
786 pNpc->m_nInitMaxY = pNpc->m_nLimitMaxZ = row->BottomZ;
787 // dungeon work
788 pNpc->m_byDungeonFamily = row->DungeonFamily;
789 pNpc->m_bySpecialType = row->SpecialType;
790 pNpc->m_byRegenType = row->RegenType;
791 pNpc->m_byTrapNumber = row->TrapNumber;
792
793 if (pNpc->m_byDungeonFamily > 0)
794 {
795 pNpc->m_nLimitMinX = row->LimitMinX;
796 pNpc->m_nLimitMinZ = row->LimitMinZ;
797 pNpc->m_nLimitMaxX = row->LimitMaxX;
798 pNpc->m_nLimitMaxZ = row->LimitMaxZ;
799 }
800
801 pNpc->m_ZoneIndex = -1;
802
803 MAP* pMap = nullptr;
804 for (size_t i = 0; i < _zones.size(); i++)
805 {
806 if (_zones[i]->m_nZoneNumber == pNpc->m_sCurZone)
807 {
808 pNpc->m_ZoneIndex = static_cast<int16_t>(i);
809 pMap = _zones[i];
810 break;
811 }
812 }
813
814 if (pMap == nullptr)
815 {
816 spdlog::error("AIServerApp::LoadNpcPosTable: NPC invalid zone [npcId:{}, "
817 "npcZoneId:{}]",
818 pNpc->m_sSid, pNpc->m_sCurZone);
819 return false;
820 }
821
822 //pNpc->Init();
823 //_npcMap.Add(pNpc);
824 if (!_npcMap.PutData(pNpc->m_sNid, pNpc))
825 {
826 spdlog::error("AIServerApp::LoadNpcPosTable: Npc PutData Fail [serial={}]",
827 pNpc->m_sNid);
828 delete pNpc;
829 pNpc = nullptr;
830 }
831
832 if (pNpc != nullptr && pMap->m_byRoomEvent > 0 && pNpc->m_byDungeonFamily > 0)
833 {
834 pRoom = pMap->m_arRoomEventArray.GetData(pNpc->m_byDungeonFamily);
835 if (pRoom == nullptr)
836 {
837 spdlog::error(
838 "AIServerApp::LoadNpcPosTable: No RoomEvent for NPC dungeonFamily: "
839 "serial={}, npcId={}, npcName={}, dungeonFamily={}, zoneId={}",
840 pNpc->m_sNid + NPC_BAND, pNpc->m_sSid, pNpc->m_strName,
841 pNpc->m_byDungeonFamily, pNpc->m_ZoneIndex);
842 return false;
843 }
844
845 int* pInt = new int;
846 *pInt = pNpc->m_sNid;
847 if (!pRoom->m_mapRoomNpcArray.PutData(pNpc->m_sNid, pInt))
848 {
849 delete pInt;
850 spdlog::error("AIServerApp::LoadNpcPosTable: MapRoomNpcArray.PutData "
851 "failed for NPC: [serial={}, npcId={}]",
852 pNpc->m_sNid, pNpc->m_sSid);
853 }
854 }
855
856 _totalNpcCount = nSerial;
857
858 if (--nNpcCount > 0)
859 continue;
860
861 bMoveNext = true;
862 nNpcCount = 0;
863 }
864 }
865 }
866 while (!bMoveNext);
867 }
868
869 return true;
870}

1220void AIServerApp::SendCompressedData(int nZone)
1221{
1222 if (_compressedPacketCount <= 0 || _compressedPacketIndex <= 0)
1223 {
1224 _compressedPacketCount = 0;
1225 _compressedPacketIndex = 0;
1226 spdlog::error("AIServerApp::SendCompressData: count={}, index={}", _compressedPacketCount,
1227 _compressedPacketIndex);
1228 return;
1229 }
1230
1231 int sendIndex = 0;
1232 unsigned int comp_data_len = 0;
1233 uint32_t crc_value = 0;
1234
1235 char sendBuffer[32000] {};
1236 uint8_t comp_buff[32000] {};
1237
1238 comp_data_len = lzf_compress(
1239 _compressedPacketBuffer, _compressedPacketIndex, comp_buff, sizeof(comp_buff));
1240
1241 assert(comp_data_len != 0 && comp_data_len <= sizeof(comp_buff));
1242
1243 if (comp_data_len == 0 || comp_data_len > sizeof(comp_buff))
1244 {
1245 spdlog::error("AIServerApp::SendCompressedData: Failed to compress packet");
1246 return;
1247 }
1248
1249 crc_value = crc32(reinterpret_cast<uint8_t*>(_compressedPacketBuffer), _compressedPacketIndex);
1250
1251 SetByte(sendBuffer, AG_COMPRESSED_DATA, sendIndex);
1252 SetShort(sendBuffer, (int16_t) comp_data_len, sendIndex);
1253 SetShort(sendBuffer, (int16_t) _compressedPacketIndex, sendIndex);
1254 SetDWORD(sendBuffer, crc_value, sendIndex);
1255 SetShort(sendBuffer, (int16_t) _compressedPacketCount, sendIndex);
1256 SetString(sendBuffer, reinterpret_cast<const char*>(comp_buff), comp_data_len, sendIndex);
1257
1258 Send(sendBuffer, sendIndex, nZone);
1259
1260 _compressedPacketCount = 0;
1261 _compressedPacketIndex = 0;
1262}
