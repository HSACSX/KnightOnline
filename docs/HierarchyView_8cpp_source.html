104void CHierarchyView::OnSelchanged(NMHDR* pNMHDR, LRESULT* pResult)
105{
106 NM_TREEVIEW* pNMTreeView = (NM_TREEVIEW*) pNMHDR;
107 if (nullptr == pNMTreeView->itemOld.hItem)

111 if (hItem == nullptr)

117 GetDocument()->SetSelectedUI(pUI);

119 *pResult = 0;
120}

142// tree item 정보 갱신하기
143void CHierarchyView::UpdateTreeItem(HTREEITEM hParent, CN3UIBase* pUIBase)
144{
145 if (pUIBase == nullptr)
146 return;
147 eUI_TYPE eUIType = pUIBase->UIType();
148 CString str;
149
150 CN3UIBase* pParentUI = pUIBase->GetParent();
151 switch (eUIType)
152 {
153 case UI_TYPE_BASE:
154 str = _T("Base");
155 break;
156 case UI_TYPE_BUTTON:
157 {
158 str = _T("Button");
159 if (pParentUI && UI_TYPE_SCROLLBAR == pParentUI->UIType())
160 { // 부모가 스크롤의 버튼일 경우
161 if (CN3UIScrollBar::BTN_LEFTUP == pUIBase->GetReserved())
162 str = _T("Left/Up Button");
163 else if (CN3UIScrollBar::BTN_RIGHTDOWN == pUIBase->GetReserved())
164 str = _T("Right/Down Button");
165 }
166 }
167 break;
168 case UI_TYPE_STATIC:
169 str = _T("Static");
170 break;
171 case UI_TYPE_PROGRESS:
172 str = _T("Progress");
173 break;
174 case UI_TYPE_IMAGE:
175 {
176 str = _T("Image");
177 if (pParentUI == nullptr)
178 break;
179 if (UI_TYPE_IMAGE == pParentUI->UIType())
180 {
181 ASSERT(UISTYLE_IMAGE_ANIMATE
182 & pParentUI->m_dwStyle); // 부모는 반드시 Animate image여야 한다.
183 str.Format("Image (%d)", pUIBase->GetReserved());
184 }
185 else if (UI_TYPE_BUTTON == pParentUI->UIType())
186 {
187 if (CN3UIButton::BS_NORMAL == pUIBase->GetReserved())
188 str = _T("Normal Image");
189 else if (CN3UIButton::BS_DOWN == pUIBase->GetReserved())
190 str = _T("Down Image");
191 else if (CN3UIButton::BS_ON == pUIBase->GetReserved())
192 str = _T("On Image");
193 else if (CN3UIButton::BS_DISABLE == pUIBase->GetReserved())
194 str = _T("Disable Image");
195 }
196 else if (UI_TYPE_STATIC == pParentUI->UIType() || UI_TYPE_EDIT == pParentUI->UIType())
197 {
198 str = _T("Background Image");
199 }
200 else if (UI_TYPE_PROGRESS == pParentUI->UIType())
201 {
202 if (CN3UIProgress::IMAGETYPE_BKGND == pUIBase->GetReserved())
203 str = _T("Background Image");
204 else if (CN3UIProgress::IMAGETYPE_FRGND == pUIBase->GetReserved())
205 str = _T("Foreground Image");
206 }
207 else if (UI_TYPE_TRACKBAR == pParentUI->UIType())
208 {
209 if (CN3UITrackBar::IMAGETYPE_BKGND == pUIBase->GetReserved())
210 str = _T("Background Image");
211 else if (CN3UITrackBar::IMAGETYPE_THUMB == pUIBase->GetReserved())
212 str = _T("Thumb Image");
213 }
214
215 if (UISTYLE_IMAGE_ANIMATE & pUIBase->m_dwStyle)
216 str = "(Animate)" + str;
217 }
218 break;
219 case UI_TYPE_SCROLLBAR:
220 str = _T("Scrollbar");
221 break;
222 case UI_TYPE_STRING:
223 str = _T("String");
224 break;
225 case UI_TYPE_TRACKBAR:
226 str = _T("Trackbar");
227 break;
228 case UI_TYPE_EDIT:
229 str = _T("Edit");
230 break;
231
232 //cerberus 02, 01,02
233 case UI_TYPE_AREA:
234 str = _T("Area");
235 break;
236 case UI_TYPE_ICONSLOT:
237 str = _T("Icon Slot");
238 break;
239 case UI_TYPE_LIST:
240 str = _T("List");
241 break;
242
243 default:
244 str = _T("Unknown");
245 break;
246 }
247
248 str += " - ";
249 str += pUIBase->m_szID.c_str(); // 이름을 붙여준다... !!
250
251 HTREEITEM hItem = GetTreeCtrl().InsertItem(str, hParent); // insert
252 GetTreeCtrl().SetItemData(hItem, (DWORD_PTR) pUIBase); // pointer 저장
253 GetTreeCtrl().Expand(hItem, TVE_EXPAND); // expand
254
255 // update child
256 for (CN3UIBase* pChild : pUIBase->m_Children)
257 UpdateTreeItem(hItem, pChild);
258}

260// UIBase 포인터로 tree item 선택하기
261void CHierarchyView::SelectObject(HTREEITEM hItem, CN3UIBase* pUIBase)
262{
263 if (pUIBase == nullptr || hItem == nullptr)
264 return;
265
266 if (hItem != TVI_ROOT)
267 {
268 if (GetTreeCtrl().GetItemData(hItem) == (DWORD_PTR) pUIBase)
269 {
270 GetTreeCtrl().SelectItem(hItem);
271 return;
272 }
273 }
274
275 HTREEITEM hChild = GetTreeCtrl().GetChildItem(hItem);
276 while (hChild != nullptr)
277 {
278 SelectObject(hChild, pUIBase);
279 hChild = GetTreeCtrl().GetNextItem(hChild, TVGN_NEXT);
280 }
281}
