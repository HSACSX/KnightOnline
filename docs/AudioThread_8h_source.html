32/// \class AudioThread
33/// \brief Manages active audio playback and streamed decoding.
34///
35/// AudioThread runs a dedicated background thread that processes
36/// queued audio handles, updates playback state, handles fades,
37/// manages the decoder thread for streamed assets, and invokes
38/// callbacks. This allows safe asynchronous audio playback
39/// while maintaining real-time update of all sounds.
40class AudioThread : public Thread
41{
42public:
43 /// Callback type executed on the audio thread with an AudioHandle.
44 using AudioCallback = std::function<void(AudioHandle*)>;
45
46 /// Tuple representing a queued operation in the audio thread.
47 ///
48 /// Consists of:
49 /// - e_AudioQueueType: the type of operation.
50 /// - shared_ptr<AudioHandle>: the handle affected by the operation.
51 /// - AudioCallback: optional callback function.
52 using QueueType = std::tuple<
53 e_AudioQueueType,
54 std::shared_ptr<AudioHandle>,
55 AudioCallback>;
56
57 /// Constructs the audio thread object.
58 AudioThread();
59
60 /// Stops the thread and cleans up resources.
61 ~AudioThread() override;
62
63 /// Main loop executed by the audio thread.
64 ///
65 /// Processes queued handle operations, updates playback states,
66 /// applies fades, and schedules decoding of streamed audio.
67 void thread_loop() override;
68
69 /// Queues an audio handle for playback and decoding.
70 ///
71 /// Marks the handle as managed and ensures it is tracked by
72 /// the audio thread until removed.
73 ///
74 /// \param handle Audio handle to add.
75 void Add(std::shared_ptr<AudioHandle> handle);
76
77 /// Queues a callback to be executed on the audio thread.
78 ///
79 /// \param handle Audio handle context for the callback.
80 /// \param callback Function to invoke on the audio thread.
81 void QueueCallback(std::shared_ptr<AudioHandle> handle, AudioCallback callback);
82
83 /// Queues an audio handle for removal from the audio thread.
84 ///
85 /// Stops playback, releases resources, and marks the handle as

88 /// \param handle Audio handle to remove.
89 void Remove(std::shared_ptr<AudioHandle> handle);
90
91private:
92 /// Resets a handle for playback, stopping any current source.
93 ///
94 /// For streamed handles, ensures buffers are allocated and

98 /// \param alreadyManaged true if the handle is already active in the thread.

100
101 /// Updates the decoder state for a streamed handle.
102 ///
103 /// Processes any available decoded chunks and queues them to
104 /// available OpenAL buffers for playback.

109
110 /// Updates playback state for an audio handle.
111 ///
112 /// This method handles start delays, fade-in, fade-out, looping, and
113 /// stopping, updating the handle's internal timer and applying gain
114 /// changes as needed.
115 ///
116 /// The logic was originally part of \ref CN3SndObj::Tick(), but has been
117 /// moved here into the audio thread to allow asynchronous, thread-safe
118 /// playback updates. It no longer resides in CN3SndObj::Tick().
119 ///
120 /// \param handle Audio handle to update.
121 /// \param elapsedTime Time in seconds since the last tick.
122 void tick_sound(std::shared_ptr<AudioHandle>& handle, float elapsedTime);
123
124 /// Sets the OpenAL gain for a handle and updates its locally stored
125 /// CurrentGain.
126 ///
127 /// \param handle Audio handle to update.
128 /// \param gain Gain value in range [0.0, 1.0].
129 void set_gain(std::shared_ptr<AudioHandle>& handle, float gain);
130
131 /// Starts playback of an audio handle.
132 ///
133 /// Updates state, marks StartedPlaying, and triggers OpenAL playback.
134 ///
135 /// \param handle Audio handle to play.
136 void play_impl(std::shared_ptr<AudioHandle>& handle);
137
138 /// Stops playback and releases OpenAL buffers for a handle.
139 ///
140 /// For streamed handles, also removes them from the decoder thread.
141 ///
142 /// \param handle Audio handle to remove.
143 void remove_impl(std::shared_ptr<AudioHandle>& handle);
144
145protected:
