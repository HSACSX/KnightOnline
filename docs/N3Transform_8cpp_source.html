102void CN3Transform::Tick(float fFrm)
103{
104 if (FRAME_SELFPLAY == fFrm)
105 {
106 m_fFrmCur += s_fSecPerFrm;
107 if (m_fFrmCur < 0)
108 m_fFrmCur = 0.0f;
109 if (m_fFrmCur >= m_fFrmWhole)
110 m_fFrmCur = 0.0f;
111 fFrm = m_fFrmCur;
112 }
113 else
114 {
115 m_fFrmCur = fFrm;
116 }
117
118 bool bNdeedReCalcMatrix = this->TickAnimationKey(m_fFrmCur);
119
120 if (m_dwType & OBJ_JOINT)
121 return; // Joint 일 경우는 행렬을 계산하는 방법이 다르기 땜시 넘어간다..
122
123 if (bNdeedReCalcMatrix)
124 this->ReCalcMatrix();
125}

127void CN3Transform::ReCalcMatrix()
128{
129 m_Matrix.Scale(m_vScale);
130 if (m_qRot.w != 0)
131 m_Matrix *= m_qRot;
132 m_Matrix.PosSet(m_vPos);

135bool CN3Transform::TickAnimationKey(float fFrm)
136{
137 // 에니메이션 키
138 int nKCP = m_KeyPos.Count();
139 int nKCR = m_KeyRot.Count();
140 int nKCS = m_KeyScale.Count();
141 if (nKCP <= 0 && nKCR <= 0 && nKCS <= 0)
142 return false;
143
144 bool bNeedReCalcMatrix = false;
145 if (m_KeyPos.DataGet(fFrm, m_vPos) == true)
146 bNeedReCalcMatrix = true;
147 if (m_KeyRot.DataGet(fFrm, m_qRot) == true)
148 bNeedReCalcMatrix = true;
149 if (m_KeyScale.DataGet(fFrm, m_vScale) == true)
150 bNeedReCalcMatrix = true;
151
152 return bNeedReCalcMatrix;
153}
