111uint32_t CN3UIWndBase::MouseProc(uint32_t dwFlags, const POINT& ptCur, const POINT& ptOld)
112{
113 uint32_t dwRet = UI_MOUSEPROC_NONE;
114 if (!m_bVisible)
115 return dwRet;
116
117 // 아이콘을 가진 윈도우는 자기 윈도우 영역을 벗어 났을때도 자식의 메시지를 받을수 있어야 한다.. ^^
118
119 // UI 움직이는 코드
120 if (UI_STATE_COMMON_MOVE == m_eState)
121 {
122 if (dwFlags & UI_MOUSE_LBCLICKED)
123 {
124 SetState(UI_STATE_COMMON_NONE);
125 }
126 else
127 {
128 MoveOffset(ptCur.x - ptOld.x, ptCur.y - ptOld.y);
129 }
130 dwRet |= UI_MOUSEPROC_DONESOMETHING;
131 return dwRet;
132 }
133
134 if (false == IsIn(ptCur.x, ptCur.y)) // 영역 밖이면
135 {
136 if (false == IsIn(ptOld.x, ptOld.y))
137 {
138 if (GetState() != UI_STATE_ICON_MOVING)
139 return dwRet; // 이전 좌표도 영역 밖이면
140 }
141 dwRet |= UI_MOUSEPROC_PREVINREGION; // 이전 좌표는 영역 안이었다.
142 }
143 else
144 {
145 // tool tip 관련
146 if (s_pTooltipCtrl != nullptr)
147 s_pTooltipCtrl->SetText(m_szToolTip, m_crToolTip);
148 }
149
150 dwRet |= UI_MOUSEPROC_INREGION; // 이번 좌표는 영역 안이다.
151
152 // child에게 메세지 전달
153 for (UIListItor itor = m_Children.begin(); m_Children.end() != itor; ++itor)
154 {
155 CN3UIBase* pChild = (*itor);
156 uint32_t dwChildRet = pChild->MouseProc(dwFlags, ptCur, ptOld);
157 if (UI_MOUSEPROC_DONESOMETHING & dwChildRet)
158 {
159 // 이경우에는 먼가 포커스를 받은 경우이다.
160 dwRet |= (UI_MOUSEPROC_CHILDDONESOMETHING | UI_MOUSEPROC_DONESOMETHING);
161 return dwRet;
162 }
163 }
164
165 // UI 움직이는 코드
166 if (UI_STATE_COMMON_MOVE != m_eState && PtInRect(&m_rcMovable, ptCur) && (dwFlags & UI_MOUSE_LBCLICK))
167 {
168 // 인벤토리 윈도우이고 상거래 중이면..
169 if ((UIType() == UI_TYPE_ICON_MANAGER) && (m_eUIWnd == UIWND_INVENTORY)
170 && (CGameProcedure::s_pProcMain->m_pUITransactionDlg->IsVisible()))
171 return dwRet;
172 SetState(UI_STATE_COMMON_MOVE);
173 dwRet |= UI_MOUSEPROC_DONESOMETHING;
174 return dwRet;
175 }
176
177 return dwRet;
178}
