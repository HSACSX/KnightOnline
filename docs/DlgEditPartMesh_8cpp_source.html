151void CDlgEditPartMesh::OnPartMeshBtnSave()
152{
153 UpdateData(TRUE);
154
155 if(m_strPathName == _T(""))
156 {
157 OnPartMeshBtnSaveAs();
158 return;
159 }
160
161 FILE* file = fopen((LPCTSTR)m_strPathName, "w");
162 if(!file)
163 {
164 MessageBox("n3fxpart파일 생성 실패..-.-;;","ERR02",MB_OK);
165 return;
166 }
167
168 fprintf(file,"<N3FXPART>\n");
169 fprintf(file,"<type> mesh\n");
170 fprintf(file,"<shape_name> %s\n", (LPCTSTR)m_strShapeName);
171 CString TexUVMove;
172 m_CBTexUVMove.GetLBText(m_CBTexUVMove.GetCurSel(), TexUVMove);
173 fprintf(file,"<texture_move> %s %5.4f\n", (LPCTSTR)TexUVMove, m_fTexUVVelocity);
174 fprintf(file,"<scale_velocity> %5.4f %5.4f %5.4f\n", m_fScaleX, m_fScaleY, m_fScaleZ);
175 fprintf(file,"<scale> %5.4f %5.4f %5.4f\n", m_fBaseScaleX, m_fBaseScaleY, m_fBaseScaleZ);
176 fprintf(file,"<scale_accelerate> %5.4f %5.4f %5.4f\n", m_fScaleAccelX, m_fScaleAccelY, m_fScaleAccelZ);
177
178 fprintf(file,"<tex_fps> %5.4f\n", m_fTexVelocity);
179
180 if(m_bTexLoop) fprintf(file,"<tex_loop> true\n");
181 else fprintf(file,"<tex_loop> false\n");
182
183 if(m_bAlphaBlend) fprintf(file,"<alpha> true\n");
184 else fprintf(file,"<alpha> false\n");
185
186 if(m_bOnGround) fprintf(file,"<on_ground> true\n");
187 else fprintf(file,"<on_ground> false\n");
188
189 if(m_bDoubleSide) fprintf(file,"<doubleside> true\n");
190 else fprintf(file,"<doubleside> false\n");
191
192 if(m_bLight) fprintf(file,"<light> true\n");
193 else fprintf(file,"<light> false\n");
194
195 if(m_bZBuffer) fprintf(file,"<zbuffer> true\n");
196 else fprintf(file,"<zbuffer> false\n");
197
198 if(m_bZWrite) fprintf(file,"<zwrite> true\n");
199 else fprintf(file,"<zwrite> false\n");
200
201 fprintf(file,"<mesh_fps> %5.4f\n", m_fMeshFPS);
202
203 fprintf(file,"<acceleration> %5.4f %5.4f %5.4f\n", m_fAccelX, m_fAccelY, m_fAccelZ);
204 fprintf(file,"<position0> %5.4f %5.4f %5.4f\n", m_fPosX, m_fPosY, m_fPosZ);
205 fprintf(file,"<velocity> %5.4f %5.4f %5.4f\n", m_fMoveX, m_fMoveY, m_fMoveZ);
206 fprintf(file,"<rot_velocity> %5.4f %5.4f %5.4f\n", m_fRotX, m_fRotY, m_fRotZ);
207
208 fprintf(file,"<life> %5.4f\n", m_fLife);
209 fprintf(file,"<fadeout> %5.4f\n", m_fFadeOut);
210 fprintf(file,"<fadein> %5.4f\n", m_fFadeIn);
211
212 CString strBlend;
213 m_CBSrcBlend.GetLBText(m_CBSrcBlend.GetCurSel(), strBlend);
214 fprintf(file,"<src_blend> %s\n", (LPCTSTR)strBlend);
215
216 m_CBDestBlend.GetLBText(m_CBDestBlend.GetCurSel(), strBlend);
217 fprintf(file,"<dest_blend> %s\n", (LPCTSTR)strBlend);
218
219 fprintf(file,"<end>\n");
220 fclose(file);
221}

223void CDlgEditPartMesh::OnPartMeshBtnSaveAs()
224{

235 pBaseFileAccess->FileNameSet((LPCTSTR)PathName);
236
237 m_strPathName.Empty();

248bool CDlgEditPartMesh::LoadPartScript(const char* szPath)
249{
250 m_strPathName = szPath;
251 CN3FXPartMesh* pPart = new CN3FXPartMesh;
252 if(!pPart->DecodeScriptFile(szPath))
253 {
254 delete pPart;
255 return false;
256 }
257
258 //////////////////////////////////////////////////
259 //각 컨트롤 셋팅...
260 m_bOnGround = pPart->m_bOnGround;
261 m_bAlphaBlend = pPart->m_bAlpha;
262
263 if(D3DCULL_NONE==pPart->m_dwDoubleSide) m_bDoubleSide = TRUE;
264 else m_bDoubleSide = FALSE;
265
266 m_bLight = pPart->m_dwLight;

268 m_bZWrite = pPart->m_dwZWrite;

273
274 if(pPart->m_cTextureMoveDir==1) m_fTexUVVelocity = pPart->m_fv;
275 else if(pPart->m_cTextureMoveDir==2 ) m_fTexUVVelocity = -pPart->m_fv;
276 else if(pPart->m_cTextureMoveDir==3) m_fTexUVVelocity = pPart->m_fu;
277 else if(pPart->m_cTextureMoveDir==4 ) m_fTexUVVelocity = -pPart->m_fu;
278 else m_fTexUVVelocity = 0.0f;
279
280 m_fMeshFPS = pPart->m_fMeshFPS;
281
282 m_fScaleZ = pPart->m_vScaleVel.z;
283 m_fScaleY = pPart->m_vScaleVel.y;
284 m_fScaleX = pPart->m_vScaleVel.x;
285
286 m_fScaleAccelX = pPart->m_vScaleAccel.x;
287 m_fScaleAccelY = pPart->m_vScaleAccel.y;
288 m_fScaleAccelZ = pPart->m_vScaleAccel.z;
289
290 m_fAccelX = pPart->m_vAcceleration.x;
291 m_fAccelY = pPart->m_vAcceleration.y;
292 m_fAccelZ = pPart->m_vAcceleration.z;
293
294 m_fBaseScaleX = pPart->m_vUnitScale.x;
295 m_fBaseScaleY = pPart->m_vUnitScale.y;
296 m_fBaseScaleZ = pPart->m_vUnitScale.z;
297
298 m_fFadeOut = pPart->m_fFadeOut;
299 m_fFadeIn = pPart->m_fFadeIn;
300 m_fLife = pPart->m_fLife;
301 m_fMoveX = pPart->m_vVelocity.x;
302 m_fMoveY = pPart->m_vVelocity.y;
303 m_fMoveZ = pPart->m_vVelocity.z;
304
305 m_fPosX = pPart->m_vPos.x;
306 m_fPosY = pPart->m_vPos.y;
307 m_fPosZ = pPart->m_vPos.z;
308
309 m_fRotX = pPart->m_vRotVelocity.x;
310 m_fRotY = pPart->m_vRotVelocity.y;

312

317 UpdateData(FALSE);
318 //
319 //////////////////////////////////////////////////
320
321 delete pPart;
322 return true;

374void CDlgEditPartMesh::OnPartMeshBtnLoadShape()
375{
376 DWORD dwFlags = OFN_EXPLORER | OFN_FILEMUSTEXIST | OFN_LONGNAMES | OFN_HIDEREADONLY;
377 CFileDialog dlg(TRUE, "N3Shape", nullptr, dwFlags, "N3Shape File(*.n3shape)|*.n3shape||", nullptr);
378 if(dlg.DoModal() == IDCANCEL) return;
379
380 CString PathName = dlg.GetPathName();
381
382 CN3BaseFileAccess* pBaseFileAccess = new CN3BaseFileAccess;
383 pBaseFileAccess->FileNameSet((LPCTSTR)PathName);
384 PathName = pBaseFileAccess->FileName().c_str();
385
386 if( (PathName[0]=='F' || PathName[0]=='f') &&
387 (PathName[1]=='X' || PathName[1]=='x') &&
388 (PathName[2]=='/' || PathName[2]=='\\') )

390 m_strShapeName = PathName;
