77void CN3SPart::Tick(const __Matrix44& mtxParent, const __Quaternion& qRot,
78 float fScale) // timeGetTime 으로 얻은 값을 넣으면 Texture Animation 을 컨트롤 한다..
79{
80 CN3PMesh* pPMesh = m_PMInst.GetMesh();
81 if (nullptr == pPMesh)
82 return;
83
84 m_bOutOfCameraRange = FALSE;
85
86 // 카메라와 멀리 떨어지면 지나간다..
87 __Vector3 vCenter = (this->Min() + this->Max()) * 0.5f;
88 if (s_CameraData.IsOutOfFrustum(
89 vCenter, this->Radius() * fScale)) // 카메라 사면체 바깥이면 지나간다..
90 {
91 m_bOutOfCameraRange = TRUE;
92 return;
93 }
94
95 float fDist = (vCenter - s_CameraData.vEye).Magnitude();
96 float fLOD = fDist * s_CameraData.fFOV / fScale;
97 // float fLOD = fDist + fDist * (s_CameraData.fFOV - 1.0f) / 3.0f;
98 // float fLOD = fDist * s_CameraData.fFOV * (512.0f / s_CameraData.fFP);
99
100 // 카메라 거리에 따라 LOD 수준을 조절한다.
101 // fLOD *= 256.0f / s_CameraData.fFP;
102 m_PMInst.SetLOD(fLOD);
103
104 int iTC = static_cast<int>(m_TexRefs.size());
105 if (iTC > 1) // 텍스처 에니메이션
106 {
107 m_fTexIndex += CN3Base::s_fSecPerFrm * m_fTexFPS;
108 if (m_fTexIndex >= iTC)
109 m_fTexIndex -= (iTC * m_fTexIndex)
110 / iTC; // 정수로 나누면 소숫점만 남기게 된다??(하여튼 비슷해~)
111 }
112
113 if (m_Mtl.nRenderFlags & RF_BOARD_Y) // 카메라를 바라봐야하는 거면..
114 {
115 __Vector3 vPos = m_vPivot * mtxParent;
116 __Vector3 vDir = s_CameraData.vEye - vPos;
117 if (vDir.x > 0.0f)
118 m_Matrix.RotationY(-atanf(vDir.z / vDir.x) - (__PI * 0.5f));
119 else
120 m_Matrix.RotationY(-atanf(vDir.z / vDir.x) + (__PI * 0.5f));
121
122 // 부모 회전과 반대로 회전을 시킨다..
123 float fAngle;
124 qRot.AxisAngle(vDir, fAngle);
125 if (fAngle != 0)
126 {
127 __Quaternion qRotI;
128 qRotI.RotationAxis(vDir, -fAngle);
129 m_Matrix *= qRotI;
130 }
131
132 m_Matrix *= mtxParent;
133 m_Matrix.PosSet(vPos);
134 }
135
136 if (m_Mtl.nRenderFlags & RF_WINDY) // 바람에 살짝 날려야 하면..
137 {
138 m_fTimeToSetWind -= CN3Base::s_fSecPerFrm;
139
140 if (m_fTimeToSetWind <= 0)
141 {
142 // m_vWindFactorToReach.x = 0.05f - (0.1f * (rand()%100) / 100.0f);
143 // m_vWindFactorToReach.y = 0.05f - (0.1f * (rand()%100) / 100.0f); // 위아래로는 조금만 불게 한다.
144 // m_vWindFactorToReach.z = 0.05f - (0.1f * (rand()%100) / 100.0f);
145
146 m_fWindFactorToReach = (rand() % 100) / 100.0f;
147 m_fTimeToSetWind = 3.0f * ((rand() % 100) / 100.0f); // 바람이 지속될 값..
148 }
149 else if (m_fWindFactorToReach != m_fWindFactorCur)
150 // else if(m_vWindFactorToReach != m_vWindFactorCur)
151 {
152 // float fFactor = s_fSecPerFrm * (m_vWindFactorToReach - m_vWindFactorCur).Magnitude();
153 float fFactor = s_fSecPerFrm * std::abs(m_fWindFactorToReach - m_fWindFactorCur);
154
155 // if(m_vWindFactorCur.x < m_vWindFactorToReach.x) m_vWindFactorCur.x += fFactor;
156 // if(m_vWindFactorCur.x > m_vWindFactorToReach.x) m_vWindFactorCur.x -= fFactor;
157 // if(m_vWindFactorCur.y < m_vWindFactorToReach.y) m_vWindFactorCur.y += fFactor;
158 // if(m_vWindFactorCur.y > m_vWindFactorToReach.y) m_vWindFactorCur.y -= fFactor;
159 // if(m_vWindFactorCur.z < m_vWindFactorToReach.z) m_vWindFactorCur.z += fFactor;
160 // if(m_vWindFactorCur.z > m_vWindFactorToReach.z) m_vWindFactorCur.z -= fFactor;
161 if (m_fWindFactorCur < m_fWindFactorToReach)
162 m_fWindFactorCur += fFactor;
163 if (m_fWindFactorCur > m_fWindFactorToReach)
164 m_fWindFactorCur -= fFactor;
165
166 // if(std::abs(m_vWindFactorToReach.x - m_vWindFactorCur.x) < fFactor) m_vWindFactorCur.x = m_vWindFactorToReach.x;
167 // if(std::abs(m_vWindFactorToReach.y - m_vWindFactorCur.y) < fFactor) m_vWindFactorCur.y = m_vWindFactorToReach.y;
168 // if(std::abs(m_vWindFactorToReach.z - m_vWindFactorCur.z) < fFactor) m_vWindFactorCur.z = m_vWindFactorToReach.z;
169 if (std::abs(m_fWindFactorToReach - m_fWindFactorCur) < fFactor)
170 m_fWindFactorCur = m_fWindFactorToReach;
171
172 __Vector3 vPos = m_vPivot * mtxParent;
173 // m_Matrix.Rotation(CN3Base::s_vWindFactor * m_fWindFactorCur);
174 m_Matrix.Rotation(__Vector3(0.05f, 0.02f, 0.05f) * m_fWindFactorCur);
175 m_Matrix *= mtxParent;
176 m_Matrix.PosSet(vPos);
177 }
178 }
179}

181void CN3SPart::Render()
182{
183 if (m_bOutOfCameraRange || m_PMInst.GetNumVertices() <= 0)
184 return;
185
186#ifdef _DEBUG
187 CN3Base::s_RenderInfo.nShape_Part++; // Rendering Information Update...
188#endif
189
190 LPDIRECT3DTEXTURE9 lpTex = nullptr;
191 int iTC = static_cast<int>(m_TexRefs.size());
192 if (iTC > 0)
193 {
194 int iTexIndex = (int) m_fTexIndex;
195 if (iTexIndex >= 0 && iTexIndex < iTC && m_TexRefs[iTexIndex])
196 lpTex = m_TexRefs[iTexIndex]->Get();
197 }
198
199 if (m_Mtl.nRenderFlags & RF_ALPHABLENDING) // Alpha 사용
200 {
201 __AlphaPrimitive* pAP = s_AlphaMgr.Add();
202 if (pAP)
203 {
204 pAP->bUseVB = FALSE;
205 pAP->dwBlendDest = m_Mtl.dwDestBlend;
206 pAP->dwBlendSrc = m_Mtl.dwSrcBlend;
207 pAP->dwFVF = FVF_VNT1;
208 pAP->dwPrimitiveSize = sizeof(__VertexT1);
209 pAP->fCameraDistance = (s_CameraData.vEye - m_Matrix.Pos()).Magnitude();
210 pAP->lpTex = lpTex;
211 pAP->ePrimitiveType = D3DPT_TRIANGLELIST;
212 pAP->nPrimitiveCount = m_PMInst.GetNumIndices() / 3;
213 pAP->nRenderFlags = m_Mtl.nRenderFlags;
214 pAP->nVertexCount = m_PMInst.GetNumVertices();
215 pAP->MtxWorld = m_Matrix;
216 pAP->pVertices = m_PMInst.GetVertices();
217 pAP->pwIndices = m_PMInst.GetIndices();
218 }
219
220 return; // 렌더링 안하지롱.
221 }
222
223 // static uint32_t dwAlpha, dwFog, dwCull;
224 DWORD dwFog, dwCull;
225 // s_lpD3DDev->GetRenderState(D3DRS_ALPHABLENDENABLE, &dwAlpha);
226 // if(TRUE != dwAlpha) s_lpD3DDev->SetRenderState(D3DRS_ALPHABLENDENABLE, TRUE);
227 // s_lpD3DDev->SetRenderState(D3DRS_SRCBLEND, m_Mtl.dwSrcBlend);
228 // s_lpD3DDev->SetRenderState(D3DRS_DESTBLEND, m_Mtl.dwDestBlend);
229
230#ifdef _DEBUG
231 CN3Base::s_RenderInfo.nShape_Polygon += m_PMInst.GetNumIndices()
232 / 3; // Rendering Information Update...
233#endif
234
235 if (m_Mtl.nRenderFlags & RF_NOTUSEFOG) // Fog 무시..
236 {
237 s_lpD3DDev->GetRenderState(D3DRS_FOGENABLE, &dwFog);
238 if (TRUE == dwFog)
239 s_lpD3DDev->SetRenderState(D3DRS_FOGENABLE, FALSE);
240 }
241 if (m_Mtl.nRenderFlags & RF_DOUBLESIDED) // Render Flags -
242 {
243 s_lpD3DDev->GetRenderState(D3DRS_CULLMODE, &dwCull);
244 s_lpD3DDev->SetRenderState(D3DRS_CULLMODE, D3DCULL_NONE);
245 }
246
247 s_lpD3DDev->SetMaterial(&m_Mtl); // 재질 설정..
248 s_lpD3DDev->SetTexture(0, lpTex);
249 if (nullptr != lpTex)
250 {
251 s_lpD3DDev->SetTextureStageState(0, D3DTSS_COLOROP, m_Mtl.dwColorOp);
252 s_lpD3DDev->SetTextureStageState(0, D3DTSS_COLORARG1, m_Mtl.dwColorArg1);
253 s_lpD3DDev->SetTextureStageState(0, D3DTSS_COLORARG2, m_Mtl.dwColorArg2);
254 }
255 else
256 {
257 s_lpD3DDev->SetTextureStageState(0, D3DTSS_COLOROP, D3DTOP_SELECTARG1);
258 s_lpD3DDev->SetTextureStageState(0, D3DTSS_COLORARG1, D3DTA_DIFFUSE);
259 }
260
261 // 로딩할때 미리 계산해 놓은 월드 행렬 적용..
262 s_lpD3DDev->SetTransform(D3DTS_WORLD, m_Matrix.toD3D());
263
264 m_PMInst.Render();
265
266 // if((m_Mtl.nRenderFlags & RF_ALPHABLENDING) && FALSE == dwAlpha) s_lpD3DDev->SetRenderState(D3DRS_ALPHABLENDENABLE, FALSE);
267 if ((m_Mtl.nRenderFlags & RF_NOTUSEFOG) && TRUE == dwFog)
268 s_lpD3DDev->SetRenderState(D3DRS_FOGENABLE, TRUE); // 안개 사용하지 않는다..
269 if ((m_Mtl.nRenderFlags & RF_DOUBLESIDED) && D3DCULL_NONE != dwCull)
270 s_lpD3DDev->SetRenderState(D3DRS_CULLMODE, dwCull);
271}

274void CN3SPart::RenderSelected(bool bWireFrame)
275{
276 if (m_bOutOfCameraRange || m_PMInst.GetNumVertices() <= 0)
277 return;
278
279#ifdef _DEBUG
280 CN3Base::s_RenderInfo.nShape_Part++; // Rendering Information Update...

290 lpTex = m_TexRefs[iTexIndex]->Get();
291 }
292
293 s_lpD3DDev->SetTexture(0, lpTex);
294 s_lpD3DDev->SetTextureStageState(0, D3DTSS_COLOROP, D3DTOP_SELECTARG1);

296 s_lpD3DDev->SetTextureStageState(0, D3DTSS_COLORARG1, D3DTA_TEXTURE);
297 else
298 s_lpD3DDev->SetTextureStageState(0, D3DTSS_COLORARG1, D3DTA_DIFFUSE);
299
300 s_lpD3DDev->SetTransform(D3DTS_WORLD, m_Matrix.toD3D());
301 if (bWireFrame)
302 {
303 __Vector3 vLines[4];
304
305 int nFC = m_PMInst.GetNumIndices() / 3;
306 uint16_t* pwIndices = m_PMInst.GetIndices();
307 __VertexT1* pVs = m_PMInst.GetVertices();
308
309 for (int i = 0; i < nFC; i++)
310 {
311 vLines[0] = pVs[pwIndices[i * 3 + 0]];
312 vLines[1] = pVs[pwIndices[i * 3 + 1]];
313 vLines[2] = pVs[pwIndices[i * 3 + 2]];
314 vLines[3] = vLines[0];
315 CN3Base::RenderLines(vLines, 3, 0xff00ff00);
316 }
317 }
318 else
319 m_PMInst.Render();
320}

336bool CN3SPart::Load(File& file)
337{
338 int nL = 0;
339 char szFN[256];
340
341 file.Read(&m_vPivot, sizeof(__Vector3));
342
343 file.Read(&nL, 4); // Mesh FileName
344 file.Read(szFN, nL);
345 szFN[nL] = '\0'; // 메시 파일 이름..
346 this->MeshSet(szFN);
347
348 file.Read(&m_Mtl, sizeof(__Material)); // 재질
349
350 int iTC = 0;
351 file.Read(&iTC, 4);
352 file.Read(&m_fTexFPS, 4);
353 m_TexRefs.clear();
354 this->TexAlloc(iTC); // Texture Pointer Pointer 할당..
355 for (int j = 0; j < iTC; j++) // Texture Count 만큼 파일 이름 읽어서 텍스처 부르기..
356 {
357 file.Read(&nL, 4);
358 if (nL > 0)
359 {
360 file.Read(szFN, nL);
361 szFN[nL] = '\0'; // 텍스처 파일 이름..
362 m_TexRefs[j] = s_MngTex.Get(szFN, true, s_Options.iTexLOD_Shape);
363 }
364 }
365
366 return true;
367}

756void CN3Shape::FindMinMax()
757{
758 m_vMin.Zero();
759 m_vMax.Zero();
760 m_fRadius = 0;
761
762 if (m_Parts.empty())
763 return;
764
765 m_vMin.Set(FLT_MAX, FLT_MAX, FLT_MAX);
766 m_vMax.Set(-FLT_MAX, -FLT_MAX, -FLT_MAX);
767 __Vector3 vMinTmp(0, 0, 0);
768 __Vector3 vMaxTmp(0, 0, 0);
769
770 // 가장 큰 지점찾기..
771 __Matrix44 mtxWI = m_Matrix.Inverse(); // World Matrix Inverse
772 for (CN3SPart* pPart : m_Parts)
773 {
774 vMinTmp = pPart->Min() * mtxWI; // 월드 상의 최소값을 로컬 좌표로 바꾸어준다..
775 vMaxTmp = pPart->Max() * mtxWI; // 월드 상의 최대값을 로컬 좌표로 바꾸어준다..
776
777 if (vMinTmp.x < m_vMin.x)
778 m_vMin.x = vMinTmp.x;
779 if (vMinTmp.y < m_vMin.y)
780 m_vMin.y = vMinTmp.y;
781 if (vMinTmp.z < m_vMin.z)
782 m_vMin.z = vMinTmp.z;
783 if (vMaxTmp.x > m_vMax.x)
784 m_vMax.x = vMaxTmp.x;
785 if (vMaxTmp.y > m_vMax.y)
786 m_vMax.y = vMaxTmp.y;
787 if (vMaxTmp.z > m_vMax.z)
788 m_vMax.z = vMaxTmp.z;
789 }
790
791 // 최대 최소값을 갖고 반지름 계산한다..
792 m_fRadius = (m_vMax - m_vMin).Magnitude() * 0.5f;
793}

795int CN3Shape::CheckCollisionPrecisely(
796 bool bIgnoreBoxCheck, int ixScreen, int iyScreen, __Vector3* pVCol, __Vector3* pVNormal)
797{
798 // Compute the vector of the pick ray in screen space
799 __Vector3 vTmp;
800
801 vTmp.x = (((2.0f * ixScreen) / (CN3Base::s_CameraData.vp.Width)) - 1)
802 / CN3Base::s_CameraData.mtxProjection.m[0][0];
803 vTmp.y = -(((2.0f * iyScreen) / (CN3Base::s_CameraData.vp.Height)) - 1)
804 / CN3Base::s_CameraData.mtxProjection.m[1][1];
805 vTmp.z = 1.0f;
806
807 // Transform the screen space pick ray into 3D space
808 __Matrix44* pMtxVI = &CN3Base::s_CameraData.mtxViewInverse;
809 __Vector3 vPos, vDir;
810 vDir.x = vTmp.x * pMtxVI->m[0][0] + vTmp.y * pMtxVI->m[1][0] + vTmp.z * pMtxVI->m[2][0];
811 vDir.y = vTmp.x * pMtxVI->m[0][1] + vTmp.y * pMtxVI->m[1][1] + vTmp.z * pMtxVI->m[2][1];
812 vDir.z = vTmp.x * pMtxVI->m[0][2] + vTmp.y * pMtxVI->m[1][2] + vTmp.z * pMtxVI->m[2][2];
813 vPos = pMtxVI->Pos();
814
815 return this->CheckCollisionPrecisely(bIgnoreBoxCheck, vPos, vDir, pVCol, pVNormal);
816}
817
818int CN3Shape::CheckCollisionPrecisely(bool bIgnoreBoxCheck, const __Vector3& vPos,
819 const __Vector3& vDir, __Vector3* pVCol, __Vector3* pVNormal)
820{
821 // 박스 체크 먼저한다..
822 if (!bIgnoreBoxCheck && !_CheckCollisionByBox(vPos, vDir, m_vMin * m_Matrix, m_vMax * m_Matrix))
823 return -1;
824
825 __Vector3 vPos2 = vPos, vDir2 = vDir;
826 int partCount = static_cast<int>(m_Parts.size());
827 for (int i = 0; i < partCount; i++)
828 {
829 CN3SPart* pPart = m_Parts[i];
830
831 CN3PMeshInstance* pPMI = pPart->MeshInstance();
832 if (pPMI == nullptr)
833 continue;
834
835 __VertexT1* pVs = pPMI->GetVertices();
836 uint16_t* pwIs = pPMI->GetIndices();
837 int nIndexCount = pPMI->GetNumIndices();
838
839 int nFC = nIndexCount / 3; // Face Count
840
841 // Face 수가 24 개보다 많은 경우 일단 박스체크를 한다..
842 if (nFC > 64 && !::_CheckCollisionByBox(vPos, vDir, pPart->Min(), pPart->Max()))
843 continue;
844
845 __Matrix44 mtxWI = pPart->m_Matrix.Inverse(); // World Matrix Inverse
846
847 vPos2 = vPos * mtxWI;
848 mtxWI.PosSet(0, 0, 0);
849 vDir2 = vDir * mtxWI; // 역행렬로 회전..
850
851 int nCI0, nCI1, nCI2;
852 for (int j = 0; j < nFC; j++) // 각각의 Face 마다 충돌체크..
853 {
854 nCI0 = pwIs[j * 3 + 0];
855 nCI1 = pwIs[j * 3 + 1];
856 nCI2 = pwIs[j * 3 + 2];
857
858 if (!_IntersectTriangle(vPos2, vDir2, pVs[nCI0], pVs[nCI1], pVs[nCI2]))
859 continue;
860
861 float fT, fU, fV;
862 ::_IntersectTriangle(vPos2, vDir2, pVs[nCI0], pVs[nCI1], pVs[nCI2], fT, fU, fV, pVCol);
863 if (pVCol != nullptr)
864 (*pVCol) *= pPart->m_Matrix;
865
866 if (pVNormal != nullptr)
867 {
868 pVNormal->Cross(pVs[nCI1] - pVs[nCI0], pVs[nCI2] - pVs[nCI1]);
869 pVNormal->Normalize();
870
871 mtxWI = pPart->m_Matrix.Inverse(); // World Matrix Inverse
872 mtxWI.PosSet(0, 0, 0); // 역행렬로 회전..
873 *pVNormal *= mtxWI; // 역행렬로 회전..
874 }
875
876 return i;
877 }
878 }
879
880 return -1;
881}

883bool CN3Shape::MakeCollisionMeshByParts() // 충돌 메시를 박스로 만든다...
884{
885 int iVC = 0, iIC = 0;
886 for (CN3SPart* pPart : m_Parts)
887 {
888 CN3PMesh* pPMesh = pPart->Mesh();
889 if (pPMesh == nullptr)
890 continue;
891
892 iVC += 8;
893 iIC += 36;
894 }
895
896 if (iVC <= 0 || iIC <= 0)
897 return false;
898
899 CN3VMesh* pVMesh = new CN3VMesh();
900 pVMesh->CreateVertices(iVC);
901 pVMesh->CreateIndex(iIC);
902
903 __Vector3* pVDest = pVMesh->Vertices();
904 uint16_t* pwIDest = pVMesh->Indices();
905
906 __Matrix44 mtxI = m_Matrix.Inverse();
907 CN3VMesh VMTmp;
908
909 iVC = 0;
910 iIC = 0;
911 for (CN3SPart* pPart : m_Parts)
912 {
913 CN3PMesh* pPMesh = pPart->Mesh();
914 if (pPMesh == nullptr)
915 continue;
916
917 VMTmp.CreateCube(pPMesh->Min(), pPMesh->Max());
918
919 __Vector3* pVSrc = VMTmp.Vertices();
920 uint16_t* pwISrc = VMTmp.Indices();
921
922 pPart->Tick(m_Matrix, m_qRot, 1.0f);
923
924 __Matrix44 mtxPart = pPart->m_Matrix;
925 mtxPart *= mtxI;
926
927 for (int j = 0; j < 8; j++)
928 pVDest[iVC + j] = pVSrc[j] * mtxPart;
929
930 for (int j = 0; j < 36; j++)
931 pwIDest[iIC + j] = pwISrc[j] + iVC;
932
933 iVC += 8;
934 iIC += 36;
935 }
936
937 if (iVC <= 0 || iIC <= 0)
938 {
939 delete pVMesh;
940 return false;
941 }
942
943 int iCount = CN3Base::s_MngVMesh.Count();
944 std::string buff = fmt::format(
945 "{}_collision_{}.n3vmesh", m_szFileName.c_str(), iCount); // 임시로 이름일 짓고..

953
