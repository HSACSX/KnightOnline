9/// \enum e_AudioAssetType
10/// \brief Describes how an audio asset is stored and consumed by the audio engine.
11///
12/// This enum is used to distinguish between fully-loaded audio buffers and
13/// streamed audio assets, which may be decoded incrementally at runtime.
14enum e_AudioAssetType : uint8_t
15{
16 /// Audio data is fully loaded into memory and played from a shared OpenAL buffer.
17 AUDIO_ASSET_BUFFERED = 0,
18
19 /// Audio data is memory mapped from file and streamed incremenetally (being decoded in chunks) during playback.
20 AUDIO_ASSET_STREAMED,
21
22 /// Placeholder type for undetermined audio asset types.
23 AUDIO_ASSET_UNKNOWN
24};

43/// \class AudioAsset
44/// \brief Base class representing a loadable audio asset.
45///
46/// AudioAsset provides common metadata and lifetime management for audio
47/// resources used by the sound engine. Concrete implementations define how
48/// the audio data is loaded, decoded, and supplied to the playback backend
49/// (e.g. buffered or streamed assets).
50class AudioAsset
51{
52public:
53 /// Specifies how the audio asset is stored and consumed.
54 /// This is used for runtime identification of class types,
55 /// in lieu of dynamic_cast() (which would require the use of RTTI).
56 e_AudioAssetType Type = AUDIO_ASSET_UNKNOWN;
57
58 /// Decoder used to convert encoded audio data into PCM samples.
59 e_AudioDecoderType DecoderType = AUDIO_DECODER_UNKNOWN;
60
61 /// Source filename of the audio asset.
62 std::string Filename;
63
64 /// Reference count used to manage shared ownership of the asset.
65 /// Once this hits 0 (i.e. nothing references it), the asset will be
66 /// immediately removed from its pool and unloaded.
67 uint32_t RefCount = 0;
68
69 /// OpenAL audio format (e.g. AL_FORMAT_MONO16), or -1 if uninitialized.
70 int32_t AlFormat = -1;
71
72 /// Sample rate of the decoded audio data, in Hz.
73 int32_t SampleRate = 0;
74
75public:
76 /// Loads the audio asset from a file.
77 ///
78 /// Implementations are responsible for parsing the file and preparing
79 /// the asset for playback, including populating format and sample rate
80 /// information.
81 ///
82 /// \param filename Path to the audio file to load.
83 /// \return True if the asset was successfully loaded; false otherwise.
84 virtual bool LoadFromFile(const std::string& filename) = 0;
85
86 /// Virtual destructor to allow safe deletion via base-class pointers.
87 virtual ~AudioAsset()

89 }
