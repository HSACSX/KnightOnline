241bool CUIKnightsOperation::MsgRecv_KnightsList(Packet& pkt)
242{
243 m_iPageCur = pkt.read<int16_t>();
244 int iKC = pkt.read<int16_t>();
245 int iID, iNameLength, iMemberCount, iPoint;
246 std::string szName, szChiefName;
247 for(int i = 0; i < iKC; i++)
248 {
249 iID = pkt.read<int16_t>();
250 iNameLength = pkt.read<int16_t>();
251 pkt.readString(szName, iNameLength);
252 iMemberCount = pkt.read<int16_t>();
253 iNameLength = pkt.read<int16_t>();
254 pkt.readString(szChiefName, iNameLength);
255 iPoint = pkt.read<uint32_t>();
256
257 this->KnightsListAdd(iID, szName, szChiefName, iMemberCount, iPoint); // UI 에 추가..
258 }
259 this->KnightsListUpdate(); // List 에 다 넣었으면 UI Update!!
260 this->EnableKnightsUIs(true); // Disable 된 버튼들 Enable 시킨다.
261
262 return true;
263}

265void CUIKnightsOperation::MsgSend_KnightsCreate()
266{
267 if(nullptr == m_pEdit_KnightsName) return;
268
269 std::string szKnightsName = m_pEdit_KnightsName->GetString();
270 if(szKnightsName.empty()) // 이름이 없으면 에러..
271 {
272 std::string szMsg = fmt::format_text_resource(IDS_ERR_KNIGHTS_CREATE_FAILED_NAME_EMPTY);
273 CGameProcedure::s_pProcMain->MsgOutput(szMsg, 0xffff00ff);
274 return;
275 }
276
277 int iOffset = 0;
278 uint8_t byBuff[128];
279
280 CAPISocket::MP_AddByte(byBuff, iOffset, WIZ_KNIGHTS_PROCESS);
281 CAPISocket::MP_AddByte(byBuff, iOffset, N3_SP_KNIGHTS_CREATE); // 생성 Send - s1(Name Length) str1 | Recv - b1(1:성공 0:실패)
282 CAPISocket::MP_AddShort(byBuff, iOffset, (int16_t)szKnightsName.size());
283 CAPISocket::MP_AddString(byBuff, iOffset, szKnightsName);
284
285 CGameProcedure::s_pSocket->Send(byBuff, iOffset);
286}

299void CUIKnightsOperation::MsgSend_KnightsJoin()
300{
301 if(nullptr == m_pList_Knights) return;
302 int iCurSel = m_pList_Knights->GetCurSel();
303 if (iCurSel < 0

305 return;
306
307 auto it = m_KnightsListExt.begin();
