25/// \struct SoundSettings
26/// \brief Represents per-sound playback settings, such as volume.
27///
28/// It is shared between the sound instance and the audio handle, allowing,
29/// for example, persistent volume settings across subsequent plays from the
30/// same sound instance across different handles.
31struct SoundSettings
32{
33 /// Indicates whether playback should loop.
34 bool IsLooping = false;
35
36 /// Current gain applied to the sound, in the range [0.0, 1.0].
37 float CurrentGain = 0.0f;
38
39 /// Maximum gain that can be applied to the sound, in the range [0.0, 1.0].
40 /// Used as the target gain for fades and playback.
41 float MaxGain = 1.0f;
42};

44/// \enum e_SndType
45/// \brief Type of sound, used to determine playback behavior.
46enum e_SndType
47{
48 /// Standard 2D sound, unaffected by 3D spatialization.
49 SNDTYPE_2D = 0,

51 /// 3D sound that is spatialized based on position and listener.
52 SNDTYPE_3D,

54 /// Streamed sound, decoded on the fly.
55 SNDTYPE_STREAM,
56
57 /// Unknown or uninitialized sound type.
58 SNDTYPE_UNKNOWN

61/// \enum e_SndState
62/// \brief Current playback state of a sound.
63///
64/// These states are used by the audio system to manage the lifecycle
65/// of a sound handle, including fades, delays, and stopping.
66enum e_SndState
67{
68 /// Initial state, before playback starts.
69 SNDSTATE_INITIAL = 0,
70
71 /// Playback has stopped. Once in this state, the handle is no longer
72 /// considered active and will be removed from the \ref AudioThread.
73 SNDSTATE_STOP,
74
75 /// Playback is delayed, waiting for the start time to elapse.
76 SNDSTATE_DELAY,
77
78 /// Sound is fading in.
79 SNDSTATE_FADEIN,
80
81 /// Sound is actively playing at full volume.
82 SNDSTATE_PLAY,
83
84 /// Sound is fading out. Once the fade-out completes, playback
85 /// will automatically stop and the handle state will transition to SNDSTATE_STOP.
86 SNDSTATE_FADEOUT
87};
