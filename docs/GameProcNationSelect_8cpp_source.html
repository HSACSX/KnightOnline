95bool CGameProcNationSelect::ProcessPacket(Packet& pkt)
96{
97 size_t rpos = pkt.rpos();
98 if (CGameProcedure::ProcessPacket(pkt))
99 return true;
100
101 pkt.rpos(rpos);
102
103 int iCmd = pkt.read<uint8_t>(); // 커멘드 파싱..
104 switch ( iCmd ) // 커멘드에 다라서 분기..
105 {
106 case WIZ_SEL_NATION: // 캐릭터 선택 메시지..
107 {
108 int iNation = pkt.read<uint8_t>(); // 국가 - 0 실패.. 1 - 카루스 2 - 엘모라드..
109
110 if( 0 == iNation ) s_pPlayer->m_InfoBase.eNation = NATION_NOTSELECTED; // 아직 국가를 선택하지 않았다..
111 else if( 1 == iNation ) s_pPlayer->m_InfoBase.eNation = NATION_KARUS;
112 else if( 2 == iNation ) s_pPlayer->m_InfoBase.eNation = NATION_ELMORAD;
113 }
114 return true;
115 }
116
117 return false;
118}
