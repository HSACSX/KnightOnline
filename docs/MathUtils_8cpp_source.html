183 const __Vector3& vPos,
184 const __Matrix44& mtxView, const __Matrix44& mtxProjection,
185 int iViewportWidth, int iViewportHeight)
186{
187 __Matrix44 matVP = mtxView * mtxProjection;

191 _POINT pt;
192 float fScreenZ = (v.z / v.w);

195 pt.x = -1;

197 return pt;

206void _Convert2D_To_3DCoordinate(
207 int ixScreen, int iyScreen,
208 const __Matrix44& mtxView, const __Matrix44& mtxPrj,
209 int iViewportWidth, int iViewportHeight,
210 __Vector3& vPosResult, __Vector3& vDirResult)
211{
212 // Compute the vector of the pick ray in screen space
213 __Vector3 vTmp;
214
215 vTmp.x = (((2.0f * ixScreen) / iViewportWidth) - 1) / mtxPrj.m[0][0];
216 vTmp.y = -(((2.0f * iyScreen) / iViewportHeight) - 1) / mtxPrj.m[1][1];
217 vTmp.z = 1.0f;
218
219 // Transform the screen space pick ray into 3D space
220 __Matrix44 mtxVI = mtxView.Inverse();
221 vDirResult.x = vTmp.x * mtxVI.m[0][0] + vTmp.y * mtxVI.m[1][0] + vTmp.z * mtxVI.m[2][0];
222 vDirResult.y = vTmp.x * mtxVI.m[0][1] + vTmp.y * mtxVI.m[1][1] + vTmp.z * mtxVI.m[2][1];
223 vDirResult.z = vTmp.x * mtxVI.m[0][2] + vTmp.y * mtxVI.m[1][2] + vTmp.z * mtxVI.m[2][2];
224 vPosResult = mtxVI.Pos();
225}
