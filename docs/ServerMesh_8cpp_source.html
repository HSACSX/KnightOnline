236void CServerMesh::AutoConcMesh(D3DCOLOR color, float left, float right, float bottom, float top, float low, float high, int iStart)
237{
238 // left, right : x
239 // bottom, top : z
240 // low, high : y
241
242 switch (iStart)
243 {
244 case 0:
245 case 6:
246 case 12:
247 case 18:
248 m_vSMesh[iStart].Set(left, low, bottom);
249 m_vSMesh[iStart + 1].Set(left, high, bottom);
250 m_vSMesh[iStart + 2].Set(left, low, top);
251 m_vSMesh[iStart + 3].Set(left, high, top);
252 m_vSMesh[iStart + 4].Set(left, low, top);
253 m_vSMesh[iStart + 5].Set(left, high, bottom);
254 break;
255
256 case 24:
257 case 30:
258 case 36:
259 case 42:
260 m_vSMesh[iStart].Set(left, low, top);
261 m_vSMesh[iStart + 1].Set(left, high, top);
262 m_vSMesh[iStart + 2].Set(right, low, top);
263 m_vSMesh[iStart + 3].Set(right, high, top);
264 m_vSMesh[iStart + 4].Set(right, low, top);
265 m_vSMesh[iStart + 5].Set(left, high, top);
266 break;
267 }

270void CServerMesh::Render()
271{
272 __Matrix44 mtxWorld;
273 mtxWorld.Identity();
274 s_lpD3DDev->SetTransform(D3DTS_WORLD, mtxWorld.toD3D());
275 s_lpD3DDev->SetRenderState(D3DRS_CULLMODE, D3DCULL_NONE);
276
277 DWORD dwFillPrev;
278 s_lpD3DDev->GetRenderState(D3DRS_FILLMODE, &dwFillPrev);
279
280 s_lpD3DDev->SetRenderState(D3DRS_FILLMODE, D3DFILL_SOLID);
281 s_lpD3DDev->SetTextureStageState(0, D3DTSS_COLOROP, D3DTOP_SELECTARG1);
282 s_lpD3DDev->SetTextureStageState(0, D3DTSS_COLORARG1, D3DTA_TEXTURE);
283 s_lpD3DDev->SetTexture(0, nullptr);
284
285 s_lpD3DDev->SetFVF(D3DFVF_XYZ);
286 s_lpD3DDev->DrawPrimitiveUP(D3DPT_TRIANGLELIST, 16, &m_vSMesh, sizeof(__Vector3));
287
288 s_lpD3DDev->SetRenderState(D3DRS_FILLMODE, dwFillPrev);
289 s_lpD3DDev->SetRenderState(D3DRS_CULLMODE, D3DCULL_CCW);
