120{

123
124 s_lpD3DDev->SetTransform(D3DTS_WORLD, matWorld.toD3D());
125 s_lpD3DDev->SetRenderState(D3DRS_ZENABLE, D3DZB_TRUE);

128 s_lpD3DDev->SetRenderState(D3DRS_DITHERENABLE, TRUE);

130
131 s_lpD3DDev->SetRenderState(D3DRS_ALPHATESTENABLE, TRUE);
132 s_lpD3DDev->SetRenderState(D3DRS_ALPHAFUNC, D3DCMP_GREATER);
133
134 s_lpD3DDev->SetRenderState(D3DRS_FILLMODE, D3DFILL_SOLID);
135 s_lpD3DDev->SetRenderState(D3DRS_CULLMODE, D3DCULL_CCW);
136

138
139 for (int i = 0; i < 8; ++i)
140 {
141 s_lpD3DDev->SetTexture(i, nullptr);
142 s_lpD3DDev->SetSamplerState(i, D3DSAMP_MINFILTER, D3DTEXF_LINEAR);
143 s_lpD3DDev->SetSamplerState(i, D3DSAMP_MAGFILTER, D3DTEXF_LINEAR);

174//-----------------------------------------------------------------------------
175bool CN3Eng::Reset(bool bWindowed, uint32_t dwWidth, uint32_t dwHeight, uint32_t dwBPP)
176{
177 if (s_lpD3DDev == nullptr)
178 return false;
179 if (dwWidth <= 0 || dwHeight <= 0)
180 return false;
181
182 if (dwWidth == s_DevParam.BackBufferWidth && dwHeight == s_DevParam.BackBufferHeight)
183 {
184 if (0 == dwBPP)
185 return false;
186 if (16 == dwBPP && D3DFMT_R5G6B5 == s_DevParam.BackBufferFormat)
187 return false;
188 if (24 == dwBPP && D3DFMT_R8G8B8 == s_DevParam.BackBufferFormat)
189 return false;
190 if (32 == dwBPP && D3DFMT_X8R8G8B8 == s_DevParam.BackBufferFormat)
191 return false;
192 }
193
194 D3DPRESENT_PARAMETERS DevParamBackUp;
195 memcpy(&DevParamBackUp, &s_DevParam, sizeof(D3DPRESENT_PARAMETERS));
196
197 D3DFORMAT BBFormat = D3DFMT_UNKNOWN;
198
199 if (bWindowed)
200 {
201 D3DDISPLAYMODE dm;
202 m_lpD3D->GetAdapterDisplayMode(D3DADAPTER_DEFAULT, &dm);
203
204 BBFormat = dm.Format;
205 }
206 else
207 {
208 if (16 == dwBPP)
209 BBFormat = D3DFMT_R5G6B5;
210 else if (24 == dwBPP)
211 BBFormat = D3DFMT_R8G8B8;
212 else if (32 == dwBPP)
213 BBFormat = D3DFMT_X8R8G8B8;
214 }
215
216 s_DevParam.Windowed = bWindowed;
217 s_DevParam.BackBufferWidth = dwWidth;
218 s_DevParam.BackBufferHeight = dwHeight;
219 s_DevParam.BackBufferFormat = BBFormat;
220
221 int nMC = m_DeviceInfo.nModeCount;
222 for (int i = 0; i < nMC; i++)
223 {
224 if (m_DeviceInfo.pModes[i].Format == s_DevParam.BackBufferFormat)
225 {
226 FindDepthStencilFormat(0, m_DeviceInfo.DevType, m_DeviceInfo.pModes[i].Format,
227 &s_DevParam.AutoDepthStencilFormat);
228
229 m_nModeActive = i;
230
231 break;
232 }
233 }
234
235 if (D3D_OK != s_lpD3DDev->Reset(&s_DevParam))
236 {
237#ifdef _N3GAME
238 CLogWriter::Write("CNEng::Reset - Insufficient video memory");
239#endif
240 memcpy(&s_DevParam, &DevParamBackUp, sizeof(D3DPRESENT_PARAMETERS));
241
242 if (D3D_OK != s_lpD3DDev->Reset(&s_DevParam))
243 {
244#ifdef _N3GAME
245 CLogWriter::Write("CNEng::Reset - Insufficient video memory");
246#endif
247 }
248
249 return false;
250 }
251
252 RECT rcView = { 0, 0, (int) dwWidth, (int) dwHeight };
253
254 SetViewPort(rcView);
255 SetDefaultEnvironment();
256
257 return true;
258}

490BOOL CN3Eng::FindDepthStencilFormat(
491 UINT iAdapter, D3DDEVTYPE DeviceType, D3DFORMAT TargetFormat, D3DFORMAT* pDepthStencilFormat)
492{
493 int nDSC = 6;
494 D3DFORMAT DepthFmts[] = { D3DFMT_D32, D3DFMT_D24S8, D3DFMT_D24X4S4, D3DFMT_D24X8, D3DFMT_D16,
495 D3DFMT_D15S1 };
496
497 HRESULT rval = 0;
498 for (int i = 0; i < nDSC; i++)
499 {
500 rval = m_lpD3D->CheckDeviceFormat(iAdapter, DeviceType, TargetFormat, D3DUSAGE_DEPTHSTENCIL,
501 D3DRTYPE_SURFACE, DepthFmts[i]);
502 if (D3D_OK == rval)
503 {
504 rval = m_lpD3D->CheckDepthStencilMatch(
505 iAdapter, DeviceType, TargetFormat, TargetFormat, DepthFmts[i]);
506 if (D3D_OK == rval)
507 {
508 *pDepthStencilFormat = DepthFmts[i];
509 return TRUE;
510 }
511 }
512 }
513
514 return FALSE;
515}

596void CN3Eng::WaitForDeviceRestoration()
597{
598 while (true)
599 {
600 HRESULT rval = s_lpD3DDev->TestCooperativeLevel();
601 if (rval == D3DERR_DEVICENOTRESET || rval == D3D_OK)
602 {
603 rval = s_lpD3DDev->Reset(&s_DevParam);
604 if (rval == D3D_OK)
605 {
606#ifdef _N3GAME
607 CLogWriter::Write("Device reset succeeded");
608#endif
609 SetDefaultEnvironment();
610 break;
611 }
612
