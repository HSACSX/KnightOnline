37BOOL CProgressBar::Create(LPCTSTR strMessage, int nSize, int MaxValue)
38{
39 // 상태바를 얻음
40 CStatusBar* pStatusBar = GetStatusBar();
41 if (pStatusBar == nullptr)
42 return FALSE;
43
44 // 상태바 위에 프로그레스 컨트롤생성
45 if (!CProgressCtrl::Create(WS_CHILD | WS_VISIBLE, CRect(0, 0, 0, 0), pStatusBar, 1))
46 return FALSE;
47
48 // 프로그레스 컨트롤의 범위와 스텝 설정
49 SetRange(0, MaxValue);
50 SetStep(1);
51
52 m_strMessage = strMessage;
53 m_nSize = nSize;
54
55 // 위치와 크기 조정
56 Resize();
57 return TRUE;
58}

60CStatusBar* CProgressBar::GetStatusBar()
61{
62 CFrameWnd* pFrame = (CFrameWnd*) AfxGetMainWnd();
63 if (pFrame == nullptr || pFrame->IsKindOf(RUNTIME_CLASS(CFrameWnd)) == FALSE)
64 return nullptr;
65
66 CStatusBar* pBar = (CStatusBar*) pFrame->GetMessageBar();
67 if (pBar == nullptr || pBar->IsKindOf(RUNTIME_CLASS(CStatusBar)) == FALSE)
68 return nullptr;
69
70 return pBar;
71}

73void CProgressBar::Resize()
74{
75 CStatusBar* pStatusBar = GetStatusBar();
76 if (pStatusBar == nullptr)
77 return;
78
79 // 텍스트 출력
80 if (::IsWindow(m_hWnd) && IsWindowVisible())
81 {
82 pStatusBar->SetWindowText(m_strMessage);
83 pStatusBar->UpdateWindow();
84 }
85
86 // 텍스트가 차지하는 영역 계산
87 CClientDC dc(pStatusBar);
88 CFont* pOldFont = dc.SelectObject(pStatusBar->GetFont());
89 CSize size = dc.GetTextExtent(m_strMessage);
90 int margin = dc.GetTextExtent(_T(" ")).cx * 2;
91 dc.SelectObject(pOldFont);
92
93 // 프로그레스 컨트롤이 그려질 영역 계산
94 CRect rc;
95 pStatusBar->GetItemRect(0, rc);
96 rc.left = size.cx + 2 * margin;
97 rc.right = rc.left + (rc.right - rc.left) * m_nSize / 100;
98 if (rc.right < rc.left)
99 rc.right = rc.left;
100
101 // 상태바 상하로 10%의 여백을 둠
102 int Height = rc.bottom - rc.top;
103 rc.bottom -= Height / 10;
104 rc.top += Height / 10;
105
106 // 프로그레스 컨트롤의 위치와 크기를 재조정
107 if (::IsWindow(m_hWnd) && (rc != m_Rect))
108 MoveWindow(&rc);
109 m_Rect = rc;
110}
