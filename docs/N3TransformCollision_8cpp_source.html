268void CN3TransformCollision::FindMinMax()
269{
270 m_vMin.Set(0, 0, 0);
271 m_vMax.Set(0, 0, 0);
272 m_fRadius = 0.0f;
273
274 if (nullptr == m_pMeshCollision || m_pMeshCollision->VertexCount() <= 0)
275 return;
276
277 m_vMin.Set(FLT_MAX, FLT_MAX, FLT_MAX);
278 m_vMax.Set(-FLT_MAX, -FLT_MAX, -FLT_MAX);
279
280 int nVC = m_pMeshCollision->VertexCount();
281 __Vector3* pVs = m_pMeshCollision->Vertices();
282 for (int i = 0; i < nVC; i++)
283 {
284 if (pVs[i].x < m_vMin.x)
285 m_vMin.x = pVs[i].x;
286 if (pVs[i].y < m_vMin.y)
287 m_vMin.y = pVs[i].y;
288 if (pVs[i].z < m_vMin.z)
289 m_vMin.z = pVs[i].z;
290 if (pVs[i].x > m_vMax.x)
291 m_vMax.x = pVs[i].x;
292 if (pVs[i].y > m_vMax.y)
293 m_vMax.y = pVs[i].y;
294 if (pVs[i].z > m_vMax.z)
295 m_vMax.z = pVs[i].z;
296 }
297
298 // 최대 최소값을 갖고 반지름 계산한다..
299 m_fRadius = (m_vMax - m_vMin).Magnitude() * 0.5f;
300}
