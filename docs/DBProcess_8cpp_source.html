63int CDBProcess::AccountLogin(const char* accountId, const char* password)
64{
65 // TODO: Restore this, but it should be handled ideally in its own database, or a separate stored procedure.
66 // As we're currently using a singular database (and we expect people to be using our database), and we have
67 // no means of syncing this currently, we'll temporarily hack this to fetch and handle basic auth logic
68 // without a procedure.
69 db::SqlBuilder<model::TbUser> sql;
70 sql.IsWherePK = true;
71
72 try
73 {
74 db::ModelRecordSet<model::TbUser> recordSet;
75
76 auto stmt = recordSet.prepare(sql);
77 if (stmt == nullptr)
78 {
79 throw db::ApplicationError("DBProcess::AccountLogin: statement could not be allocated");
80 }
81
82 stmt->bind(0, accountId);
83 recordSet.execute();
84
85 if (!recordSet.next())
86 return AUTH_NOT_FOUND;
87
88 model::TbUser user {};
89 recordSet.get_ref(user);
90
91 if (user.Password != password)
92 {
93 // Use AUTH_NOT_FOUND here instead of AUTH_INVALID_PW
94 // to ensure attackers have no way of identifying real accounts to bruteforce passwords on.
95 return AUTH_NOT_FOUND;
96 }
97
98 if (user.Authority == AUTHORITY_BLOCK_USER)
99 {
100 return AUTH_BANNED;
101 }
102 }
103 catch (const nanodbc::database_error& dbErr)
104 {
105 db::utils::LogDatabaseError(dbErr, "DBProcess::AccountLogin()");
106 return AUTH_FAILED;
107 }
108
109 return AUTH_OK;
110}

112bool CDBProcess::InsertVersion(
113 int version, const char* fileName, const char* compressName, int historyVersion)
114{
115 using ModelType = model::Version;
116
117 db::SqlBuilder<ModelType> sql;
118 std::string insert = sql.InsertString();
119 try
120 {
121 auto conn = db::ConnectionManager::CreatePoolConnection(
122 ModelType::DbType(), DB_PROCESS_TIMEOUT);
123 if (conn == nullptr)
124 return false;
125
126 nanodbc::statement stmt = conn->CreateStatement(insert);
127 stmt.bind(0, &version);
128 stmt.bind(1, fileName);
129 stmt.bind(2, compressName);
130 stmt.bind(3, &historyVersion);
131
132 nanodbc::result result = stmt.execute();
133 if (result.affected_rows() > 0)
134 return true;
135 }
136 catch (const nanodbc::database_error& dbErr)
137 {
138 db::utils::LogDatabaseError(dbErr, "DBProcess::InsertVersion()");
139 return false;
140 }
141
142 return false;
143}

173bool CDBProcess::LoadUserCountList()
174{
175 VersionManagerApp* appInstance = VersionManagerApp::instance();
176 try
177 {
178 db::ModelRecordSet<model::Concurrent> recordSet;
179 recordSet.open();
180
181 while (recordSet.next())
182 {
183 model::Concurrent concurrent = recordSet.get();
184
185 int serverId = concurrent.ServerId - 1;
186 if (serverId >= static_cast<int>(appInstance->ServerList.size()))
187 continue;
188
189 appInstance->ServerList[serverId]->sUserCount = concurrent.Zone1Count
190 + concurrent.Zone2Count
191 + concurrent.Zone3Count;
192 }
193
194 return true;
195 }
196 catch (const nanodbc::database_error& dbErr)
197 {
198 db::utils::LogDatabaseError(dbErr, "DBProcess::LoadUserCountList()");
199 }
200
201 return false;
202}
