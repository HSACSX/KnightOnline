24std::shared_ptr<TcpClientSocket> TcpClientSocketManager::AcquireSocket()
25{
26 int socketId = -1;
27 std::shared_ptr<TcpClientSocket> tcpSocket;
28
29 // NOTE: Handle the guarding externally so it's clear what's guarded and what's not,
30 // which is critical when dealing with code needing to be fairly high performance here.
31 {
32 std::lock_guard<std::recursive_mutex> lock(_mutex);
33 tcpSocket = std::static_pointer_cast<TcpClientSocket>(
34 TcpSocketManager::AcquireSocket(socketId));
35 }
36
37 if (socketId == -1)
38 {
39 spdlog::error(
40 "TcpClientSocketManager({})::AcquireSocket: socketId list is empty", _implName);
41 return nullptr;
42 }
43
44 // This should never happen.
45 // If it does, the associated socket ID was never removed from the list so we don't have to restore it.
46 if (tcpSocket == nullptr)
47 {
48 spdlog::error("TcpClientSocketManager({})::AcquireSocket: null socket [socketId:{}]",
49 _implName, socketId);
50 return nullptr;
51 }
52
53 return tcpSocket;
54}

56bool TcpClientSocketManager::ProcessClose(TcpSocket* tcpSocket)
57{
58 std::lock_guard<std::recursive_mutex> lock(_mutex);
59 if (tcpSocket->GetState() == CONNECTION_STATE_DISCONNECTED)
60 return false;
61
62 // NOTE: Unlike server sockets, client sockets don't get removed
63 // from the manager upon disconnect.
64 // This allows the user to reconnect to the same socket instance.
65 tcpSocket->CloseProcess();
66 return true;
67}
