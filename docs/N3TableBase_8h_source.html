113template <class Type>
114bool CN3TableBase<Type>::Load(File& file)
115{
116 Release();
117
118 // data(column) 의 구조가 어떻게 되어 있는지 읽기
119 int iDataTypeCount = 0;
120 file.Read(&iDataTypeCount, 4); // (엑셀에서 column 수)
121
122 std::vector<int> offsets;
123 __ASSERT(iDataTypeCount > 0, "Data Type 이 0 이하입니다.");
124 if (iDataTypeCount > 0)
125 {
126 m_DataTypes.insert(m_DataTypes.begin(), iDataTypeCount, DT_NONE);
127 file.Read(&m_DataTypes[0], sizeof(DATA_TYPE) * iDataTypeCount); // 각각의 column에 해당하는 data type
128
129 if (!MakeOffsetTable(offsets))
130 {
131 __ASSERT(0, "can't make offset table");
132 return FALSE; // structure변수에 대한 offset table 만들어주기
133 }
134
135 int iSize = offsets[iDataTypeCount]; // MakeOffstTable 함수에서 리턴되는 값중 m_iDataTypeCount번째에 이 함수의 실제 사이즈가 들어있다.
136 if (sizeof(Type) != iSize // 전체 type의 크기와 실제 구조체의 크기와 다르거나
137 || DT_DWORD != m_DataTypes[0]) // 맨 처음의 데이타가 DT_DWORD형이 아닐때(맨처음은 고유한 ID이므로)
138 {
139 m_DataTypes.clear();
140 __ASSERT(0, "DataType is mismatch or DataSize is incorrect!!");
141 return false;
142 }
143 }
144
145 // row 가 몇줄인지 읽기
146 int iRC;
147 file.Read(&iRC, sizeof(iRC));
148
149 Type Data;
150 for (int i = 0; i < iRC; i++)
151 {
152 for (int j = 0; j < iDataTypeCount; j++)
153 ReadData(file, m_DataTypes[j], (char*) (&Data) + offsets[j]);
154
155 uint32_t dwKey = *((uint32_t*) (&Data));
156 auto pt = m_Datas.insert(std::make_pair(dwKey, Data));
157
158 __ASSERT(pt.second, "CN3TableBase<Type> : Key 중복 경고.");
159 }
160
161 return true;
162}

164// structure는 4바이트 정렬하여서 메모리를 잡는다. 따라서 아래 함수가 필요하다.
165// 아래 함수로 OffsetTable을 만들어 쓴 후에는 만드시 리턴값을 delete [] 를 해주어야 한다.
166template <class Type>
167bool CN3TableBase<Type>::MakeOffsetTable(std::vector<int>& offsets)
168{
169 if (m_DataTypes.empty())
170 return false;
171
172 static constexpr int StructAlignment = alignof(Type);
173
174 int iDataTypeCount = (int) m_DataTypes.size();
175
176 offsets.clear();
177 offsets.resize(iDataTypeCount + 1);
178 offsets[0] = 0;
179
180 int iPrevDataSize = SizeOf(m_DataTypes[0]);
181 for (int i = 1; i < iDataTypeCount; i++)
182 {
183 int iCurDataSize = SizeOf(m_DataTypes[i]);
184 int iPreviousOffset = offsets[i - 1];
185
186 int modulo = (iCurDataSize % StructAlignment);
187 if (0 == modulo)
188 {
189 modulo = (iPreviousOffset + iPrevDataSize) % StructAlignment;
190 if (0 == modulo)
191 offsets[i] = iPreviousOffset + iPrevDataSize;
192 else
193 offsets[i] = ((int) (iPreviousOffset + iPrevDataSize + (StructAlignment - 1)) / StructAlignment) * StructAlignment;
194 }
195 else if (1 == modulo)
196 {
197 offsets[i] = iPreviousOffset + iPrevDataSize;
198 }
199 else if (2 == modulo)
200 {
201 modulo = ((iPreviousOffset + iPrevDataSize) % 2);
202 if (0 == modulo)
203 offsets[i] = iPreviousOffset + iPrevDataSize;
204 else
205 offsets[i] = iPreviousOffset + iPrevDataSize + 1; // NOTE: Effectively this is (2 - modulo), but modulo can only be 1 here.
206 }
207 else if (4 == modulo)
208 {
209 modulo = ((iPreviousOffset + iPrevDataSize) % 4);
210 if (0 == modulo)
211 offsets[i] = iPreviousOffset + iPrevDataSize;
212 else
213 offsets[i] = iPreviousOffset + iPrevDataSize + (4 - modulo);
214 }
215 else
216 {
217 __ASSERT(0, "");
218 }
219
220 iPrevDataSize = iCurDataSize;
221 }
222
223 offsets[iDataTypeCount] = ((int) (offsets[iDataTypeCount - 1] + iPrevDataSize + (StructAlignment - 1)) / StructAlignment) * StructAlignment;
224 return true;
225}
