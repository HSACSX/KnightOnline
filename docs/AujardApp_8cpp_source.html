150bool AujardApp::OnStart()
151{
152 if (!_dbAgent.InitDatabase())
153 return false;
154
155 if (!LoadItemTable())
156 return false;
157
158 // Attempt to open all shared memory queues & blocks first.
159 // If it fails (memory not yet available), we'll run the _smqOpenThread to periodically check
160 // until it can finally be opened.
161 if (!AttemptOpenSharedMemory())
162 {
163 spdlog::info("AujardApp::OnStart: shared memory unavailable, waiting for memory to become available");
164 _smqOpenThread->start();
165 }
166 else
167 {
168 OnSharedMemoryOpened();
169 }
170
171 return true;
172}

174bool AujardApp::AttemptOpenSharedMemory()
175{
176 bool openedAll = true;
177
178 if (!LoggerRecvQueue.IsOpen()
179 && !LoggerRecvQueue.Open(SMQ_LOGGERSEND))
180 openedAll = false;
181
182 if (!LoggerSendQueue.IsOpen()
183 && !LoggerSendQueue.Open(SMQ_LOGGERRECV))
184 openedAll = false;
185
186 // NOTE: This looks unsafe, but the server isn't started up until all of this finishes,
187 // so nothing else is using _dbAgent.UserData yet.
188 if (_dbAgent.UserData.empty()
189 && !InitSharedMemory())
190 openedAll = false;
191
192 return openedAll;
193}

250void AujardApp::SelectCharacter(const char* buffer)
251{
252 int index = 0, userId = -1, sendIndex = 0, idLen1 = 0, idLen2 = 0, tempUserId = -1,
253 packetIndex = 0;
254 uint8_t init = 0x01;
255 char sendBuff[256] = {},
256 accountId[MAX_ID_SIZE + 1] = {},
257 charId[MAX_ID_SIZE + 1] = {};
258
259 _USER_DATA* user = nullptr;
260
261 userId = GetShort(buffer, index);
262 idLen1 = GetShort(buffer, index);
263 GetString(accountId, buffer, idLen1, index);
264 idLen2 = GetShort(buffer, index);
265 GetString(charId, buffer, idLen2, index);
266 init = GetByte(buffer, index);
267 packetIndex = GetDWORD(buffer, index);
268
269 spdlog::debug("AujardApp::SelectCharacter: acctId={}, charId={}, index={}",
270 accountId, charId, packetIndex);
271
272 _recvPacketCount++; // packet count
273
274 if (userId < 0
275 || userId >= MAX_USER)
276 goto fail_return;
277
278 if (strlen(accountId) == 0)
279 goto fail_return;
280
281 if (strlen(charId) == 0)
282 goto fail_return;
283
284 if (GetUserPtr(charId, tempUserId) != nullptr)
285 {
286 SetShort(sendBuff, tempUserId, sendIndex);
287 SetShort(sendBuff, idLen1, sendIndex);
288 SetString(sendBuff, accountId, idLen1, sendIndex);
289 SetShort(sendBuff, idLen2, sendIndex);
290 SetString(sendBuff, charId, idLen2, sendIndex);
291 UserLogOut(sendBuff);
292 return;
293 }
294
295 if (!_dbAgent.LoadUserData(accountId, charId, userId))
296 goto fail_return;
297
298 if (!_dbAgent.LoadWarehouseData(accountId, userId))
299 goto fail_return;
300
301 user = _dbAgent.UserData[userId];
302 if (user == nullptr)
303 goto fail_return;
304
305 if (strcpy_safe(user->m_Accountid, accountId) != 0)
306 {
307 spdlog::error("AujardApp::SelectCharacter: accountId too long (len: {}, val: {})",
308 std::strlen(accountId), accountId);
309 // it didn't fail here before and we don't currently return anything upstream
310 // if this exposes any problems we'll have to decide how to handle it then
311 }
312
313 SetByte(sendBuff, WIZ_SEL_CHAR, sendIndex);
314 SetShort(sendBuff, userId, sendIndex);
315 SetByte(sendBuff, 0x01, sendIndex);
316 SetByte(sendBuff, init, sendIndex);
317
318 _packetCount++;
319
320 if (LoggerSendQueue.PutData(sendBuff, sendIndex) != SMQ_OK)
321 {
322 spdlog::error("AujardApp::SelectCharacter: Packet Drop: WIZ_SEL_CHAR");
323 return;
324 }
325
326 ++_sendPacketCount;
327 return;
328
329fail_return:
330 sendIndex = 0;
331 SetByte(sendBuff, WIZ_SEL_CHAR, sendIndex);
332 SetShort(sendBuff, userId, sendIndex);
333 SetByte(sendBuff, 0x00, sendIndex);
334
335 LoggerSendQueue.PutData(sendBuff, sendIndex);
336}

368bool AujardApp::HandleUserLogout(int userId, uint8_t saveType, bool forceLogout)
369{
370 bool logoutResult = false;
371
372 // make sure UserData[userId] value is valid
373 _USER_DATA* pUser = _dbAgent.UserData[userId];
374 if (pUser == nullptr || std::strlen(pUser->m_id) == 0)
375 {
376 spdlog::error("AujardApp::HandleUserLogout: Invalid logout: UserData[{}] is not in use", userId);
377 return false;
378 }
379
380 // only call AccountLogout for non-zone change logouts
381 if (pUser->m_bLogout != 2 || forceLogout)
382 logoutResult = _dbAgent.AccountLogout(pUser->m_Accountid);
383 else
384 logoutResult = true;
385
386 // update UserData (USERDATA/WAREHOUSE)
387 bool userdataSuccess = HandleUserUpdate(userId, *pUser, saveType);
388
389 // Log results
390 bool success = userdataSuccess && logoutResult;
391 if (!success)
392 {
393 spdlog::error("AujardApp::HandleUserLogout: Invalid Logout: {}, {} (UserData: {}, Logout: {})",
394 pUser->m_Accountid, pUser->m_id, userdataSuccess, logoutResult);
395 return false;
396 }
397
398 spdlog::debug("AujardApp::HandleUserLogout: Logout: {}, {} (UserData: {}, Logout: {})",
399 pUser->m_Accountid, pUser->m_id, userdataSuccess, logoutResult);
400
401 // reset the object stored in UserData[userId] before returning
402 // this will reset data like accountId/charId, so logging must
403 // occur beforehand
404 _dbAgent.ResetUserData(userId);
405
406 return success;
407}

409bool AujardApp::HandleUserUpdate(int userId, const _USER_DATA& user, uint8_t saveType)
410{
411 auto sleepTime = 10ms;
412 int updateWarehouseResult = 0, updateUserResult = 0,
413 retryCount = 0, maxRetry = 10;
414
415 // attempt updates
416 updateUserResult = _dbAgent.UpdateUser(user.m_id, userId, saveType);
417
418 std::this_thread::sleep_for(sleepTime);
419
420 updateWarehouseResult = _dbAgent.UpdateWarehouseData(user.m_Accountid, userId, saveType);
421
422 // TODO: Seems like the following two loops could/should just be combined
423 // retry handling for update user/warehouse
424 for (retryCount = 0; !updateWarehouseResult || !updateUserResult; retryCount++)
425 {
426 if (retryCount >= maxRetry)
427 {
428 spdlog::error("AujardApp::HandleUserUpdate: UserData Save Error: [accountId={} charId={} (W:{},U:{})]",
429 user.m_Accountid, user.m_id, updateWarehouseResult, updateUserResult);
430 break;
431 }
432 // only retry the calls that fail - they're both updating using UserData[userId]->dwTime, so they should sync fine
433 if (!updateUserResult)
434 {
435 updateUserResult = _dbAgent.UpdateUser(user.m_id, userId, saveType);
436 }
437
438 std::this_thread::sleep_for(sleepTime);
439
440 if (!updateWarehouseResult)
441 {
442 updateWarehouseResult = _dbAgent.UpdateWarehouseData(user.m_Accountid, userId, saveType);
443 }
444 }
445
446 // Verify saved data/timestamp
447 updateWarehouseResult = _dbAgent.CheckUserData(user.m_Accountid, user.m_id, 1, user.m_dwTime, user.m_iBank);
448 updateUserResult = _dbAgent.CheckUserData(user.m_Accountid, user.m_id, 2, user.m_dwTime, user.m_iExp);
449 for (retryCount = 0; !updateWarehouseResult || !updateUserResult; retryCount++)
450 {
451 if (retryCount >= maxRetry)
452 {
453 if (!updateWarehouseResult)
454 {
455 spdlog::error("AujardApp::HandleUserUpdate: Warehouse Save Check Error [accountId={} charId={} (W:{})]",
456 user.m_Accountid, user.m_id, updateWarehouseResult);
457 }
458
459 if (!updateUserResult)
460 {
461 spdlog::error("AujardApp::HandleUserUpdate: UserData Save Check Error: [accountId={} charId={} (U:{})]",
462 user.m_Accountid, user.m_id, updateUserResult);
463 }
464 break;
465 }
466
467 if (!updateWarehouseResult)
468 {
469 _dbAgent.UpdateWarehouseData(user.m_Accountid, userId, saveType);
470 updateWarehouseResult = _dbAgent.CheckUserData(user.m_Accountid, user.m_id, 1, user.m_dwTime, user.m_iBank);
471 }
472
473 std::this_thread::sleep_for(sleepTime);
474
475 if (!updateUserResult)
476 {
477 _dbAgent.UpdateUser(user.m_id, userId, saveType);
478 updateUserResult = _dbAgent.CheckUserData(user.m_Accountid, user.m_id, 2, user.m_dwTime, user.m_iExp);
479 }
480 }
481
482 return static_cast<bool>(updateUserResult) && static_cast<bool>(updateWarehouseResult);
483}
