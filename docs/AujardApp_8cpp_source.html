138bool AujardApp::OnStart()
139{
140 if (!_dbAgent.InitDatabase())
141 return false;
142
143 if (!LoadItemTable())
144 return false;
145
146 // Attempt to open all shared memory queues & blocks first.
147 // If it fails (memory not yet available), we'll run the _smqOpenThread to periodically check
148 // until it can finally be opened.
149 if (!AttemptOpenSharedMemory())
150 {
151 spdlog::info("AujardApp::OnStart: shared memory unavailable, waiting for memory to become "
152 "available");
153 _smqOpenThread->start();
154 }
155 else
156 {
157 OnSharedMemoryOpened();
158 }
159
160 return true;
161}

163bool AujardApp::AttemptOpenSharedMemory()
164{
165 bool openedAll = true;
166
167 if (!LoggerRecvQueue.IsOpen() && !LoggerRecvQueue.Open(SMQ_LOGGERSEND))
168 openedAll = false;
169
170 if (!LoggerSendQueue.IsOpen() && !LoggerSendQueue.Open(SMQ_LOGGERRECV))
171 openedAll = false;
172
173 // NOTE: This looks unsafe, but the server isn't started up until all of this finishes,
174 // so nothing else is using _dbAgent.UserData yet.
175 if (_dbAgent.UserData.empty() && !InitSharedMemory())
176 openedAll = false;
177
178 return openedAll;
179}

181void AujardApp::AttemptOpenSharedMemoryThreadTick()
182{
183 if (!AttemptOpenSharedMemory())
184 return;
185
186 // Shared memory is open, this thread doesn't need to exist anymore.
187 _smqOpenThread->shutdown(false);
188
189 // Run the server
190 OnSharedMemoryOpened();

236void AujardApp::SelectCharacter(const char* buffer)
237{
238 int index = 0, userId = -1, sendIndex = 0, idLen1 = 0, idLen2 = 0, tempUserId = -1,
239 packetIndex = 0;
240 uint8_t init = 0x01;
241 char sendBuffer[256] {}, accountId[MAX_ID_SIZE + 1] {}, charId[MAX_ID_SIZE + 1] {};
242
243 _USER_DATA* user = nullptr;
244
245 userId = GetShort(buffer, index);
246 idLen1 = GetShort(buffer, index);
247 GetString(accountId, buffer, idLen1, index);
248 idLen2 = GetShort(buffer, index);
249 GetString(charId, buffer, idLen2, index);
250 init = GetByte(buffer, index);
251 packetIndex = GetDWORD(buffer, index);
252
253 spdlog::debug("AujardApp::SelectCharacter: acctId={}, charId={}, index={}", accountId, charId,
254 packetIndex);
255
256 _recvPacketCount++; // packet count
257
258 if (userId < 0 || userId >= MAX_USER || strlen(accountId) == 0 || strlen(charId) == 0)
259 {
260 SendSelectCharacterFailed(userId);
261 return;
262 }
263
264 if (GetUserPtr(charId, tempUserId) != nullptr)
265 {
266 SetShort(sendBuffer, tempUserId, sendIndex);
267 SetShort(sendBuffer, idLen1, sendIndex);
268 SetString(sendBuffer, accountId, idLen1, sendIndex);
269 SetShort(sendBuffer, idLen2, sendIndex);
270 SetString(sendBuffer, charId, idLen2, sendIndex);
271 UserLogOut(sendBuffer);
272 return;
273 }
274
275 if (!_dbAgent.LoadUserData(accountId, charId, userId)
276 || !_dbAgent.LoadWarehouseData(accountId, userId))
277 {
278 SendSelectCharacterFailed(userId);
279 return;
280 }
281
282 user = _dbAgent.UserData[userId];
283 if (user == nullptr)
284 {
285 SendSelectCharacterFailed(userId);
286 return;
287 }
288
289 if (strcpy_safe(user->m_Accountid, accountId) != 0)
290 {
291 spdlog::error("AujardApp::SelectCharacter: accountId too long (len: {}, val: {})",
292 std::strlen(accountId), accountId);
293 // it didn't fail here before and we don't currently return anything upstream
294 // if this exposes any problems we'll have to decide how to handle it then
295 }
296
297 SetByte(sendBuffer, WIZ_SEL_CHAR, sendIndex);
298 SetShort(sendBuffer, userId, sendIndex);
299 SetByte(sendBuffer, 0x01, sendIndex);
300 SetByte(sendBuffer, init, sendIndex);
301
302 _packetCount++;
303
304 if (LoggerSendQueue.PutData(sendBuffer, sendIndex) != SMQ_OK)
305 {
306 spdlog::error("AujardApp::SelectCharacter: Packet Drop: WIZ_SEL_CHAR");
307 return;
308 }
309
310 ++_sendPacketCount;
311}

349bool AujardApp::HandleUserLogout(int userId, uint8_t saveType, bool forceLogout)
350{
351 bool logoutResult = false;
352
353 // make sure UserData[userId] value is valid
354 _USER_DATA* pUser = _dbAgent.UserData[userId];
355 if (pUser == nullptr || std::strlen(pUser->m_id) == 0)
356 {
357 spdlog::error(
358 "AujardApp::HandleUserLogout: Invalid logout: UserData[{}] is not in use", userId);
359 return false;
360 }
361
362 // only call AccountLogout for non-zone change logouts
363 if (pUser->m_bLogout != 2 || forceLogout)
364 logoutResult = _dbAgent.AccountLogout(pUser->m_Accountid);
365 else
366 logoutResult = true;
367
368 // update UserData (USERDATA/WAREHOUSE)
369 bool userdataSuccess = HandleUserUpdate(userId, *pUser, saveType);
370
371 // Log results
372 bool success = userdataSuccess && logoutResult;
373 if (!success)
374 {
375 spdlog::error(
376 "AujardApp::HandleUserLogout: Invalid Logout: {}, {} (UserData: {}, Logout: {})",
377 pUser->m_Accountid, pUser->m_id, userdataSuccess, logoutResult);
378 return false;
379 }
380
381 spdlog::debug("AujardApp::HandleUserLogout: Logout: {}, {} (UserData: {}, Logout: {})",
382 pUser->m_Accountid, pUser->m_id, userdataSuccess, logoutResult);
383
384 // reset the object stored in UserData[userId] before returning
385 // this will reset data like accountId/charId, so logging must
386 // occur beforehand
387 _dbAgent.ResetUserData(userId);
388
389 return success;
390}

392bool AujardApp::HandleUserUpdate(int userId, const _USER_DATA& user, uint8_t saveType)
393{
394 auto sleepTime = 10ms;
395 int updateWarehouseResult = 0, updateUserResult = 0, retryCount = 0, maxRetry = 10;
396
397 // attempt updates
398 updateUserResult = _dbAgent.UpdateUser(user.m_id, userId, saveType);
399
400 std::this_thread::sleep_for(sleepTime);
401
402 updateWarehouseResult = _dbAgent.UpdateWarehouseData(user.m_Accountid, userId, saveType);
403
404 // TODO: Seems like the following two loops could/should just be combined
405 // retry handling for update user/warehouse
406 for (retryCount = 0; !updateWarehouseResult || !updateUserResult; retryCount++)
407 {
408 if (retryCount >= maxRetry)
409 {
410 spdlog::error("AujardApp::HandleUserUpdate: UserData Save Error: [accountId={} "
411 "charId={} (W:{},U:{})]",
412 user.m_Accountid, user.m_id, updateWarehouseResult, updateUserResult);
413 break;
414 }
415 // only retry the calls that fail - they're both updating using UserData[userId]->dwTime, so they should sync fine
416 if (!updateUserResult)
417 {
418 updateUserResult = _dbAgent.UpdateUser(user.m_id, userId, saveType);
419 }
420
421 std::this_thread::sleep_for(sleepTime);
422
423 if (!updateWarehouseResult)
424 {
425 updateWarehouseResult = _dbAgent.UpdateWarehouseData(
426 user.m_Accountid, userId, saveType);
427 }
428 }
429
430 // Verify saved data/timestamp
431 updateWarehouseResult = _dbAgent.CheckUserData(
432 user.m_Accountid, user.m_id, 1, user.m_dwTime, user.m_iBank);
433 updateUserResult = _dbAgent.CheckUserData(
434 user.m_Accountid, user.m_id, 2, user.m_dwTime, user.m_iExp);
435 for (retryCount = 0; !updateWarehouseResult || !updateUserResult; retryCount++)
436 {
437 if (retryCount >= maxRetry)
438 {
439 if (!updateWarehouseResult)
440 {
441 spdlog::error("AujardApp::HandleUserUpdate: Warehouse Save Check Error "
442 "[accountId={} charId={} (W:{})]",
443 user.m_Accountid, user.m_id, updateWarehouseResult);
444 }
445
446 if (!updateUserResult)
447 {
448 spdlog::error("AujardApp::HandleUserUpdate: UserData Save Check Error: "
449 "[accountId={} charId={} (U:{})]",
450 user.m_Accountid, user.m_id, updateUserResult);
451 }
452 break;
453 }
454
455 if (!updateWarehouseResult)
456 {
457 _dbAgent.UpdateWarehouseData(user.m_Accountid, userId, saveType);
458 updateWarehouseResult = _dbAgent.CheckUserData(
459 user.m_Accountid, user.m_id, 1, user.m_dwTime, user.m_iBank);
460 }
461
462 std::this_thread::sleep_for(sleepTime);
463
464 if (!updateUserResult)
465 {
466 _dbAgent.UpdateUser(user.m_id, userId, saveType);
467 updateUserResult = _dbAgent.CheckUserData(
468 user.m_Accountid, user.m_id, 2, user.m_dwTime, user.m_iExp);
469 }
470 }
471
472 return static_cast<bool>(updateUserResult) && static_cast<bool>(updateWarehouseResult);
473}
