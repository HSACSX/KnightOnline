46void CUIQuestTalk::Open(Packet& pkt)
47{
48 m_iNumTalk = 0;
49 m_iCurTalk = 0;
50
51 // NOTE(srmeier): two -1s before text ids
52 int index = pkt.read<uint32_t>();
53 index = pkt.read<uint32_t>();
54
55 for (int i = 0; i < MAX_STRING_TALK; i++)
56 {
57 m_szTalk[i] = "";
58
59 index = pkt.read<uint32_t>();
60 __TABLE_QUEST_TALK* pTbl_Quest_Talk = CGameBase::s_pTbl_QuestTalk.Find(index);
61 if (pTbl_Quest_Talk)
62 {
63 m_szTalk[i] = pTbl_Quest_Talk->szTalk;
64 CGameBase::ConvertPipesToNewlines(m_szTalk[i]);
65 m_iNumTalk++;
66 }
67 }
68
69 m_pTextTalk->SetString(m_szTalk[m_iCurTalk]);
70
71 // reset scrollbar position
72 if (m_pScrollBar != nullptr)
73 m_pScrollBar->SetCurrentPos(0);
74
75 SetVisible(true);
76}

78bool CUIQuestTalk::ReceiveMessage(CN3UIBase* pSender, uint32_t dwMsg)
79{
80 if (dwMsg == UIMSG_BUTTON_CLICK)
81 {
82 if (pSender == m_pBtnOk)
83 {
84 m_iCurTalk++;
85 if (m_iCurTalk >= m_iNumTalk)
86 {
87 m_iCurTalk = 0;
88 SetVisible(false);
89 }
90 else
91 {
92 CGameBase::ConvertPipesToNewlines(m_szTalk[m_iCurTalk]);
93 m_pTextTalk->SetString(m_szTalk[m_iCurTalk]);
94 }
95 }
96 else if (pSender == m_pBtnClose)
97 {
98 SetVisible(false);
99 }
100 }
101 else if (dwMsg == UIMSG_SCROLLBAR_POS)
102 {
103 if (pSender == m_pScrollBar)
104 {
105 UpdateTextForScroll();
106 return true;
107 }
108 }
109
110 return true;
111}

184void CUIQuestTalk::UpdateTextForScroll()
185{
186 if (m_pTextTalk == nullptr || m_pScrollBar == nullptr)
187 return;
188
189 // scrollbar's current position
190 const int iScrollPosition = m_pScrollBar->GetCurrentPos();
191
192 // total number of lines of text
193 const int iTotalLineCount = m_pTextTalk->GetLineCount();
194
195 // max number of lines visible in text area
196 const int iVisibleLineCount = 8;
197
198 const int iMaxScrollableLines = iTotalLineCount - iVisibleLineCount;
199 m_pScrollBar->SetRangeMax(iMaxScrollableLines);
200
201 // return if text is shorter than or equal to the visible line count
202 if (iTotalLineCount <= iVisibleLineCount)
203 return;
204
205 // limit check for the line which displayed first, topline
206 int iTopLine = std::clamp(iScrollPosition, 0, iTotalLineCount - iVisibleLineCount);
207 m_pTextTalk->SetStartLine(iTopLine);
208}
