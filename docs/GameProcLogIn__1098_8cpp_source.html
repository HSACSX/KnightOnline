340void CGameProcLogIn_1098::MsgRecv_AccountLogIn(int iCmd, Packet& pkt)
341{
342 int iResult = pkt.read<uint8_t>(); // Recv - b1(0:실패 1:성공 2:ID없음 3:PW틀림 4:서버점검중)
343 if (1 == iResult) // 접속 성공..
344 {
345 // 모든 메시지 박스 닫기..
346 MessageBoxClose(-1);
347 m_pUILogIn->OpenServerList(); // 서버 리스트 읽기..
348 }
349 else if (2 == iResult) // ID 가 없어서 실패한거면..
350 {
351 if (iCmd == LS_LOGIN_REQ)

353 std::string szMsg = fmt::format_text_resource(IDS_NOACCOUNT_RETRY_MGAMEID);

364 }
365 }
366 else if (3 == iResult) // PassWord 실패
367 {
368 std::string szMsg = fmt::format_text_resource(IDS_WRONG_PASSWORD);
369 std::string szTmp = fmt::format_text_resource(IDS_CONNECT_FAIL);

372 else if (4 == iResult) // 서버 점검 중??

374 std::string szMsg = fmt::format_text_resource(IDS_SERVER_CONNECT_FAIL);

379 {
380 int iLen = pkt.read<int16_t>();
381 if (iLen > 0)
382 {
383 std::string szIP;
384 pkt.readString(szIP, iLen);
385 uint32_t dwPort = pkt.read<int16_t>();
386

396 CAPISocket::MP_AddString(Buff, iOffset2, s_szAccount); // Recv s1, str1(IP) s1(port) | Send s1, str1(ID)

490bool CGameProcLogIn_1098::ProcessPacket(Packet & pkt)
491{
492 size_t rpos = pkt.rpos();
493 if (CGameProcedure::ProcessPacket(pkt))
494 return true;
495
496 pkt.rpos(rpos);
497
498 s_pPlayer->m_InfoBase.eNation = NATION_UNKNOWN;
499 int iCmd = pkt.read<uint8_t>(); // 커멘드 파싱..
500 s_pPlayer->m_InfoBase.eNation = NATION_UNKNOWN;
501 switch (iCmd) // 커멘드에 다라서 분기..
502 {
503 case LS_SERVERLIST: // 접속하면 바로 보내준다..
504 MsgRecv_GameServerGroupList(pkt);
505 return true;
506
507 case LS_LOGIN_REQ: // 계정 접속 성공..
508 case LS_MGAME_LOGIN: // MGame 계정 접속 성공..
509 MsgRecv_AccountLogIn(iCmd, pkt);
510 return true;
511
512 case LS_NEWS:
513 // act as if it's handled
514 return true;
515 }
516
517 return false;
518}

520void CGameProcLogIn_1098::ConnectToGameServer() // 고른 게임 서버에 접속
521{
522 if (m_fTimeUntilNextGameConnectionAttempt > 0.0f)
523 return;
524
525 __GameServerInfo GSI;
526 if (!m_pUILogIn->ServerInfoGetCur(GSI))
527 return; // 서버를 고른다음..
528
529 const char* ip = GSI.szIP.c_str();
530 int port = SOCKET_PORT_GAME;
531
