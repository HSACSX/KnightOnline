34void CPlayerNPC::Tick()
35{
36 if(m_pShapeExtraRef) // 오브젝트이면..
37 {
38 CPlayerBase::Tick();
39 return;
40 }
41
42 if(m_fTimeAfterDeath > 0) // 죽어야하거나 죽은 넘이면...
43 {
44 if(m_fTimeAfterDeath > 3.0f)
45 this->Action(PSA_DYING, false); // 5 초가 지나야 죽는다.
46 CPlayerBase::Tick(); // 회전, 지정된 에니메이션 Tick 및 색깔 지정 처리.. 등등..
47 return;
48 }
49
50 __Vector3 vPos = m_Chr.Pos();
51 if( m_vPosFromServer.x != vPos.x || m_vPosFromServer.z != vPos.z ) // 조금 더 가야한다.
52 {
53 if(m_fMoveSpeedPerSec == 0)
54 {
55 __Vector3 vT1(m_vPosFromServer.x, 0, m_vPosFromServer.z), vT2(vPos.x, 0, vPos.z);
56 m_fMoveSpeedPerSec = (vT2 - vT1).Magnitude();
57 }
58
59 __Vector3 vOffset = m_vPosFromServer - vPos; vOffset.y = 0;
60 __Vector3 vDir = vOffset; vDir.Normalize(); // 방향..
61
62 float fSpeedAbsolute = (m_fMoveSpeedPerSec > 0) ? m_fMoveSpeedPerSec : -m_fMoveSpeedPerSec; // 속도 절대값
63 float fDist = vOffset.Magnitude(); // 거리
64 if(fDist < fSpeedAbsolute * CN3Base::s_fSecPerFrm) // 움직이는 거리가 거의 다온거면..
65 {
66 vPos.x = m_vPosFromServer.x; // 위치를 고정해주고..
67 vPos.z = m_vPosFromServer.z; // 위치를 고정해주고..
68// m_fMoveSpeedPerSec = 0; // 움직이는 속도를 0으로!
69 this->ActionMove(PSM_STOP); // 움직이는 모션 처리..
70 }
71 else
72 {
73 float fYaw = (m_fMoveSpeedPerSec < 0) ? ::_Yaw2D(-vDir.x, -vDir.z) : ::_Yaw2D(vDir.x, vDir.z); // 방향을 계산해서..
74 this->RotateTo(fYaw, false); // 진행 방향으로 돌리고..
75
76 e_StateMove eMove = PSM_STOP; // 움직임...
77 // 플레이어면 걷는 속도가 기준 나머지는 덩치에 반비례..
78 float fStandWalk = ((PLAYER_OTHER == m_ePlayerType) ? (MOVE_SPEED_WHEN_WALK * 2.0f) : (MOVE_SPEED_WHEN_WALK * m_Chr.Radius() * 2.0f));
79 if(m_fMoveSpeedPerSec < 0) eMove = PSM_WALK_BACKWARD; // 뒤로 걷기..
80 else if(m_fMoveSpeedPerSec < fStandWalk) eMove = PSM_WALK; // 앞으로 걷기..
81 else eMove = PSM_RUN; // if(fDN > 5.0f) // 다음 위치의 거리가 일정 이상이면 뛰어간다.
82 this->ActionMove(eMove); // 움직이는 모션 처리..
83
84 vPos += vDir * (fSpeedAbsolute * s_fSecPerFrm); // 이동..
85 }
86
87 float fYTerrain = ACT_WORLD->GetHeightWithTerrain(vPos.x, vPos.z); // 지면의 높이값..
88 float fYMesh = ACT_WORLD->GetHeightNearstPosWithShape(vPos, 1.0f); // 충돌 체크 오브젝트의 높이값..
89 if(fYMesh != -FLT_MAX && fYMesh > fYTerrain && fYMesh < m_fYNext + 1.0f) m_fYNext = fYMesh; // 올라갈수 있는 오브젝트이고 높이값이 지면보다 높으면.
90 else m_fYNext = fYTerrain;
91 this->PositionSet(vPos, false); // 위치 최종 적용..
92 }
93
94 // 공격 중이거나 스킬 사용중이면..
95 if (PSA_ATTACK == m_eState
96 || PSA_SPELLMAGIC == m_eState)
97 {
98 CPlayerBase* pTarget = this->TargetPointerCheck(false);
99 CPlayerBase::ProcessAttack(pTarget); // 공격에 관한 루틴 처리.. 에니메이션 세팅과 충돌만 처리할뿐 패킷은 처리 안한다..
100 }
101
102 CPlayerBase::Tick(); // 회전. 이동, 에니메이션 틱.. 상태 바뀜 등을 처리한다.
103}

105void CPlayerNPC::MoveTo(float fPosX, float fPosY, float fPosZ, float fSpeed, int iMoveMode)
106{
107 m_vPosFromServer.Set(fPosX, fPosY, fPosZ);
108 if(m_pShapeExtraRef) return; // 오브젝트 형식이면 움직일수가 없다..
109
110 // iMoveMode : 현재 움직이는 상태.. 0 -정지 1 움직임시작 2 - 1초마다 한번 연속움직임..
111 if(0 == iMoveMode)
112 {
113 }
114 else if(iMoveMode) // 움직임 시작.. // 계속 움직임
115 {
116 m_fMoveSpeedPerSec = fSpeed;
117 __Vector3 vPos = m_Chr.Pos(); vPos.y = 0;
118 __Vector3 vPosS(fPosX, 0, fPosZ);
119
120 if(fSpeed) m_fMoveSpeedPerSec *= ((vPosS - vPos).Magnitude() / (fSpeed * PACKET_INTERVAL_MOVE)) * 0.85f; // 속도보간.. 동기화때문에 그런다.. 약간 줄여주는 이유는 멈칫하는걸 방지하기 위해서이다..
121 else m_fMoveSpeedPerSec = ((vPosS - vPos).Magnitude() / (fSpeed * PACKET_INTERVAL_MOVE)) * 0.85f; // 속도보간.. 동기화때문에 그런다.. 약간 줄여주는 이유는 멈칫하는걸 방지하기 위해서이다..
122 if(fSpeed < 0) m_fMoveSpeedPerSec *= -1.0f; // 뒤로 간다..
123 }
124 else
125 {
126// __ASSERT(0, "Invalid Move Mode");
127 }
128}
