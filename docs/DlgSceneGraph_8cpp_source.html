75void CDlgSceneGraph::UpdateTreeItem(HTREEITEM hParent, CN3Base* pBase)
76{
77 if (pBase == nullptr)
78 return;
79
80 DWORD dwType = pBase->Type();
81 int nItem = 0;
82 if (dwType & OBJ_SCENE)
83 nItem = 0;
84 else if (dwType & OBJ_CAMERA)
85 nItem = 1;
86 else if (dwType & OBJ_LIGHT)
87 nItem = 2;
88 else if (dwType & OBJ_SHAPE)
89 nItem = 3;
90 else if (dwType & OBJ_CHARACTER)
91 nItem = 4;
92 else if (dwType & OBJ_MESH)
93 nItem = 5;
94 else if (dwType & OBJ_MESH_PROGRESSIVE)
95 nItem = 6;
96 else if (dwType & OBJ_MESH_INDEXED)
97 nItem = 7;
98 else if (dwType & OBJ_JOINT)
99 nItem = 8;
100 else if (dwType & OBJ_SKIN)
101 nItem = 9;
102 else if (dwType & OBJ_TRANSFORM)
103 nItem = 10;
104
105 HTREEITEM hItem = m_Tree.InsertItem(TVIF_IMAGE | TVIF_SELECTEDIMAGE | TVIF_TEXT,
106 pBase->m_szName.c_str(), nItem, nItem, 0, 0, 0, hParent, nullptr);
107 m_Tree.SetItemData(hItem, (DWORD_PTR) pBase);
108
109 if (dwType & OBJ_SCENE)
110 {
111 CString str;
112 str.Format("Scene : %s", m_pSceneRef->m_szName.c_str());
113 m_Tree.SetItemText(hItem, str);
114
115 int i = 0;
116 HTREEITEM hItem2 = nullptr;
117
118 if (m_dwFlag & OBJ_CAMERA)
119 {
120 hItem2 = m_Tree.InsertItem("Camera", hItem);
121 int nCC = m_pSceneRef->CameraCount();
122 for (i = 0; i < nCC; i++)
123 this->UpdateTreeItem(hItem2, m_pSceneRef->CameraGet(i));
124 }
125
126 if (m_dwFlag & OBJ_LIGHT)
127 {
128 hItem2 = m_Tree.InsertItem("Light", hItem);
129 int nLC = m_pSceneRef->LightCount();
130 for (i = 0; i < nLC; i++)
131 this->UpdateTreeItem(hItem2, m_pSceneRef->LightGet(i));
132 }
133
134 if (m_dwFlag & OBJ_SHAPE)
135 {
136 hItem2 = m_Tree.InsertItem("Shape", hItem);
137 int nSC = m_pSceneRef->ShapeCount();
138 for (i = 0; i < nSC; i++)
139 this->UpdateTreeItem(hItem2, m_pSceneRef->ShapeGet(i));
140 m_Tree.SortChildren(hItem2);
141 }
142
143 if (m_dwFlag & OBJ_CHARACTER)
144 {
145 hItem2 = m_Tree.InsertItem("Character", hItem);
146 int nCC2 = m_pSceneRef->ChrCount();
147 for (i = 0; i < nCC2; i++)
148 this->UpdateTreeItem(hItem2, m_pSceneRef->ChrGet(i));
149 }
150
151 if (m_dwFlag & OBJ_MESH)
152 {
153 hItem2 = m_Tree.InsertItem("Mesh Resource", hItem);
154 int nMC = m_pSceneRef->s_MngMesh.Count();
155 for (i = 0; i < nMC; i++)
156 this->UpdateTreeItem(hItem2, m_pSceneRef->s_MngMesh.Get(i));
157 }
158
159 if (m_dwFlag & OBJ_MESH_PROGRESSIVE)
160 {
161 hItem2 = m_Tree.InsertItem("Progressive Mesh Resource", hItem);
162 int nPMC = m_pSceneRef->s_MngPMesh.Count();
163 for (i = 0; i < nPMC; i++)
164 this->UpdateTreeItem(hItem2, m_pSceneRef->s_MngPMesh.Get(i));
165 }
166
167 if (m_dwFlag & OBJ_JOINT)
168 {
169 hItem2 = m_Tree.InsertItem("Joint Resource", hItem);
170 int nJC = m_pSceneRef->s_MngJoint.Count();
171 for (i = 0; i < nJC; i++)
172 this->UpdateTreeItem(hItem2, m_pSceneRef->s_MngJoint.Get(i));
173 }
174 }
175 m_Tree.Expand(hItem, TVE_EXPAND);
176}

218void CDlgSceneGraph::SelectObject(HTREEITEM hItem, void* pItemData)
219{
220 if (hItem == nullptr)
221 return;
222
223 if (hItem != TVI_ROOT)
224 {
225 if (m_Tree.GetItemData(hItem) == (DWORD_PTR) pItemData)
226 {
227 m_Tree.SelectItem(hItem);
228 return;
229 }
230 }
231
232 HTREEITEM hChild = m_Tree.GetChildItem(hItem);
233 while (hChild != nullptr)
234 {
235 SelectObject(hChild, pItemData);
236 hChild = m_Tree.GetNextItem(hChild, TVGN_NEXT);
237 }
238}

240void CDlgSceneGraph::OnPaint()
241{
242 CPaintDC dc(this); // device context for painting
243
244 CWnd* pWnd = GetDlgItem(IDC_PREVIEW);
245 if (pWnd && m_Tree.GetSafeHwnd())
246 {
247 HTREEITEM hItem = m_Tree.GetSelectedItem();
248
249 if (hItem)
250 {
251 LPARAM lParam = m_Tree.GetItemData(hItem);
252 CN3Base* pObj = (CN3Base*) lParam;
253 if (pObj)
254 {
255 DWORD dwType = pObj->Type();
256 if (dwType & OBJ_CHARACTER || dwType & OBJ_SHAPE)
257 m_pMapMng->RenderObjectToWindow((CN3TransformCollision*) pObj, pWnd->m_hWnd);
258 }
259 else
260 {
261 CRect rc;
262 pWnd->GetClientRect(&rc);
263 pWnd->ClientToScreen(&rc);
264 ScreenToClient(&rc);
265 dc.FillSolidRect(&rc, RGB(192, 192, 192));
266 }
267 }
268 }
269
270 // Do not call CDialog::OnPaint() for painting messages
271}
