85void CNpc::NpcInOut(uint8_t Type, float fx, float fz, float fy)
86{
87 int sendIndex = 0;
88 char buff[1024] {};
89
90 C3DMap* pMap = m_pMain->GetMapByIndex(m_sZoneIndex);
91 if (pMap == nullptr)
92 {
93 spdlog::error("Npc::NpcInOut: no map found for zoneIndex={} [serial={} npcId={} npcName={} "
94 "x={} z={}]",
95 m_sZoneIndex, m_sNid, m_sSid, m_strName, m_sRegion_X, m_sRegion_Z);
96 return;
97 }
98
99 if (Type == NPC_OUT)
100 {
101 pMap->RegionNpcRemove(m_sRegion_X, m_sRegion_Z, m_sNid);
102 }
103 else
104 {
105 pMap->RegionNpcAdd(m_sRegion_X, m_sRegion_Z, m_sNid);
106
107 m_fCurX = fx;
108 m_fCurZ = fz;
109 m_fCurY = fy;
110 }
111
112 SetByte(buff, WIZ_NPC_INOUT, sendIndex);
113 SetByte(buff, Type, sendIndex);
114 SetShort(buff, m_sNid, sendIndex);
115
116 if (Type == NPC_OUT)
117 {
118 m_pMain->Send_Region(buff, sendIndex, m_sCurZone, m_sRegion_X, m_sRegion_Z);
119 return;
120 }
121
122 GetNpcInfo(buff, sendIndex);
123 m_pMain->Send_Region(buff, sendIndex, m_sCurZone, m_sRegion_X, m_sRegion_Z);
124}

126void CNpc::RegisterRegion()
127{
128 int iRegX = 0, iRegZ = 0, old_region_x = 0, old_region_z = 0;
129 iRegX = (int) (m_fCurX / VIEW_DISTANCE);
130 iRegZ = (int) (m_fCurZ / VIEW_DISTANCE);
131
132 if (m_sRegion_X != iRegX || m_sRegion_Z != iRegZ)
133 {
134 C3DMap* pMap = m_pMain->GetMapByIndex(m_sZoneIndex);
135 if (pMap == nullptr)
136 return;
137
138 old_region_x = m_sRegion_X;
139 old_region_z = m_sRegion_Z;
140 pMap->RegionNpcRemove(m_sRegion_X, m_sRegion_Z, m_sNid);
141 m_sRegion_X = iRegX;
142 m_sRegion_Z = iRegZ;
143 pMap->RegionNpcAdd(m_sRegion_X, m_sRegion_Z, m_sNid);
144
145 RemoveRegion(old_region_x - m_sRegion_X,
146 old_region_z - m_sRegion_Z); // delete npc 는 계산 방향이 진행방향의 반대...
147 InsertRegion(m_sRegion_X - old_region_x,
148 m_sRegion_Z - old_region_z); // add npc 는 계산 방향이 진행방향...
149 }

255int CNpc::GetRegionNpcList(int region_x, int region_z, char* buff, int& t_count)
256{
257 // 포인터 참조하면 안됨
258 if (!m_pMain->m_bPointCheckFlag)
259 return 0;
260
261 int buff_index = 0;
262
263 C3DMap* pMap = m_pMain->GetMapByIndex(m_sZoneIndex);
264 if (pMap == nullptr)
265 return 0;
266
267 if (region_x < 0 || region_z < 0 || region_x > pMap->GetXRegionMax()
268 || region_z > pMap->GetZRegionMax())
269 return 0;
270
271 CRegion* region = &pMap->m_ppRegion[region_x][region_z];
272
273 std::lock_guard<std::recursive_mutex> lock(g_region_mutex);
274
275 for (const auto& [_, pNid] : region->m_RegionNpcArray)
276 {
277 int nid = *pNid;
278 if (nid < 0)
279 continue;
280
281 CNpc* pNpc = m_pMain->m_NpcMap.GetData(nid);
282 if (pNpc != nullptr)
283 {
284 SetShort(buff, pNpc->m_sNid, buff_index);
285 ++t_count;
286 }
287 }
288
289 return buff_index;
290}
