13bool FileReader::OpenExisting(const std::filesystem::path& path)
14{
15 // Close any existing file handle and reset read states.
16 Close();
17
18 // Open and map the given file into memory for reading.
19 auto handleResult = llfio::mapped_file({}, path.native(), llfio::handle::mode::read,
20 llfio::handle::creation::open_existing, llfio::handle::caching::all,
21 llfio::handle::flag::none);
22 if (!handleResult)
23 return false;
24
25 _mappedFileHandle = std::move(std::move(handleResult).value());
26 _address = _mappedFileHandle.address();
27 _path = path;
28 _open = true;
29
30 // Internally the size is always fetched on load (as we don't supply a size) and is what's reserved.
31 // If we can get away with it, this approach is very cheap, but we should just be careful.
32 // The next best way to get this is _mappedFileHandle.underlying_file_maximum_extent(), but
33 // we'd rather not use a second syscall unless we absolutely have to.
34 _size = static_cast<uint64_t>(_mappedFileHandle.capacity());
35
36 return true;
37}

45bool FileReader::Read(void* buffer, size_t bytesToRead, size_t* bytesRead /*= nullptr*/)
46{
47 // Ensure bytesRead is always initialised, regardless of whether this succeeds or fails.
48 if (bytesRead != nullptr)
49 *bytesRead = 0;
50
51 // Fail on invalid buffers.
52 if (buffer == nullptr)
53 return false;
54
55 // Nothing to read, so just skip ahead and consider this successful.
56 if (bytesToRead == 0)
57 return true;
58
59 assert(_mappedFileHandle.is_valid());
60 assert(_offset <= _size);
61
62 // Determine how many bytes are remaining to be read.
63 const size_t remainingBytes = static_cast<size_t>(_size - _offset);
64
65 // If we're trying to read more than is available, we should still
66 // attempt the read, copying only what's actually available.
67 const size_t bytesToCopy = std::min(bytesToRead, remainingBytes);
68
69 // Read operations can only be considered successful if we actually
70 // manage to read at least 1 byte.
71 // If there's nothing available to read, we must fail.
72 if (bytesToCopy == 0)
73 return false;
74
75 // Read directly from our memory mapped buffer.
76 std::memcpy(buffer, static_cast<uint8_t*>(_address) + _offset, bytesToCopy);
77
78 // Advance the current file offset by how much we actually read.
79 _offset += bytesToCopy;
80
81 // Update the caller with the number bytes we actually read.
82 if (bytesRead != nullptr)
83 *bytesRead = bytesToCopy;
84
85 // We read at least 1 byte, even if it wasn't the full amount.
86 return true;
87}

96bool FileReader::Seek(int64_t offset, int origin)
97{
98 // File not opened, we cannot seek.
99 if (!IsOpen())
100 return false;
101
102 int64_t newOffset = offset;
103
104 switch (origin)
105 {
106 // Explicitly set to the given offset
107 case SEEK_SET:
108 break;
109
110 // Set is relative to the current offset
111 case SEEK_CUR:
112 newOffset += static_cast<int64_t>(_offset);
113 break;
114
115 // Set is relative to the end offset (i.e. the size)
116 case SEEK_END:
117 newOffset += static_cast<int64_t>(_size);
118 break;
119
120 default:
121 // Unsupported seek type.
122 return false;
123 }
124
125 // New offset would be before the start of the file, so it is invalid and should fail.
126 if (newOffset < 0)
127 return false;
128
129 // New offset is after the end of the file, so it is invalid and shoul fail.
130 if (static_cast<uint64_t>(newOffset) > _size)
131 return false;
132
133 // Update the current file offset.
134 _offset = static_cast<uint64_t>(newOffset);
135
136 // The file offset is valid, so we can consider this request successful.
137 return true;
138}
