118 /// \brief opens a connection to the model's database, resets the recordset's state,
119 /// and prepares a new statement for the pending lookup.
120 ///
121 /// \throws db::DatasourceConfigNotFoundException
122 /// \throws nanodbc::database_error
123 std::shared_ptr<nanodbc::statement> prepare(SqlBuilder<ModelType>& filterObj) noexcept(false)
124 {
125 _columnCount = 0;
126 _rowCount.reset();
127
128 if (_poolConn == nullptr)
129 _poolConn = ConnectionManager::CreatePoolConnection(ModelType::DbType());
130
131 if (_fetchRowCount)
132 _selectCountQuery = filterObj.SelectCountString();
133 else
134 _selectCountQuery.clear();
135
136 std::string query = filterObj.SelectString();
137
138 _selectLimit = filterObj.Limit;
139 _stmt = std::make_shared<nanodbc::statement>(_poolConn->CreateStatement(query));
140
141 return _stmt;
142 }

144 /// \brief executes the existing prepared statement using the existing connection.
145 ///
146 /// \see next(), get()
147 /// \throws std::logic_error
148 /// \throws nanodbc::database_error
149 void execute() noexcept(false)
150 {
151 if (_stmt == nullptr)
152 {
153 throw std::logic_error("Statement not initialized");
154 }
155
156 if (_fetchRowCount)
157 {
158 nanodbc::statement stmt = _poolConn->CreateStatement(_selectCountQuery);
159 nanodbc::result result = stmt.execute();
160
161 int64_t rowCount = 0;
162 if (result.next())
163 {
164 rowCount = result.get<int64_t>(0);
165 if (_selectLimit > 0)
166 rowCount = (std::min)(rowCount, _selectLimit); // NOTE: allow for Windows.h defining min
167
168 _rowCount = rowCount;
169 }
170 }
171
172 _result = _stmt->execute();
173 _columnCount = _result.columns();
174
175 Model::IndexColumnNameBindings<BoundModelType>(
176 _result,
177 _columnCount,
178 _bindingIndex);
179 }
