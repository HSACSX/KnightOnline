189void CN3CEDoc::OnFileSaveAsOneFolder()
190{
191 // TODO: Add your specialized creation code here
192 DWORD dwFlags = OFN_HIDEREADONLY | OFN_OVERWRITEPROMPT;
193 CFileDialog dlg(FALSE, "N3Chr", nullptr, dwFlags, "Character Data File(*.N3Chr)|*.N3Chr||", nullptr);

195 CString szFullFileName = dlg.GetPathName();
196
197 char szPath[512]; // 경로를 잘 가려야 한다..
198 lstrcpy(szPath, szFullFileName);
199 int nFind = 0;
200 for(int i = lstrlen(szPath) - 1; i >= 0; i--)
201 {
202 if('\\' == szPath[i] || '/' == szPath[i]) nFind++;
203 if(nFind >= 2)
204 {
205 szPath[i] = '\0';
206 break;
207 }

210 char szDrive[_MAX_DRIVE], szDir[_MAX_DIR];
211 _splitpath(szFullFileName, szDrive, szDir, nullptr, nullptr);

218 {
219 char szFN[256];

235 pPlug->PMesh()->SaveToFile(szFN);

295void CN3CEDoc::OnToolOptimizeAnimationData()
296{
297 CN3Chr* pChr = m_Scene.ChrGet(0);
298 if(nullptr == pChr) return;
299 CN3AnimControl* pAniCtrl = pChr->AniCtrl();
300 if(nullptr == pAniCtrl) return;
301 CN3Joint* pJointSrc = pChr->Joint();
302 if(nullptr == pJointSrc) return;
303
304 float fSamplingRate = 30.0f / pJointSrc->m_KeyRot.SamplingRate();
305 if( fSamplingRate != 6.0f &&
306 fSamplingRate != 3.0f &&
307 fSamplingRate != 2.0f &&
308 fSamplingRate != 1.0f )
309 {
310 MessageBox(AfxGetMainWnd()->m_hWnd, "SamplingRate 는 5, 10, 15, 30 만 지원됩니다.", "SamplingRate 오류", MB_OK);
311 return;
312 }
313
314 MessageBox(AfxGetMainWnd()->m_hWnd, "원본 Joint 와 Animation File 을 다른 이름으로 저장해 둡니다. 이 파일은 나중에 다시 편집할때 필요합니다.", "Joint 및 Animation 파일 다른이름으로 저장", MB_OK);
315
316 CString FileName;
317 DWORD dwFlags = OFN_HIDEREADONLY | OFN_OVERWRITEPROMPT;
318 CFileDialog dlg1(FALSE, "N3Anim", nullptr, dwFlags, "Animation Data(*.N3Anim)|*.n3Anim||", nullptr);
319 if(dlg1.DoModal() == IDCANCEL) return;
320 FileName = dlg1.GetPathName();
321
322 std::string szFN(FileName);
323 std::string szFNPrev = pAniCtrl->FileName();
324 pAniCtrl->SaveToFile(szFN); // 저장..
325 pAniCtrl->FileNameSet(szFNPrev);
326
327 CFileDialog dlg2(FALSE, "N3Joint", nullptr, dwFlags, "Joint File(*.N3Joint)|*.N3Joint||", nullptr);
328 if(dlg2.DoModal() == IDCANCEL) return;
329 FileName = dlg2.GetPathName();
330
331 szFNPrev = pJointSrc->FileName();
332 pChr->Tick(0.0f); // 초기위치로 Tick
333 szFN = FileName;
334 pJointSrc->SaveToFile(szFN); // 원래걸 다른 이름으로 저장..
335 pJointSrc->FileNameSet(szFNPrev);
336
337 int nK = pAniCtrl->Count();
338 CN3Joint JointDest;
339
340 pChr->Tick(0.0f); // 초기위치로 Tick
341 JointDest.CopyExceptAnimationKey(pJointSrc); // 뼈대 구조 및 값들만 복사한다음..
342 JointDest.AddKey(pJointSrc, 0, 0); // 바인딩 포즈 키를 추가하고 복사한다..
343
344 int iAniCount = pAniCtrl->Count();
345 std::vector<float> fFrmOffsets(iAniCount, 0);
346 std::vector<float> fFrmStarts(iAniCount, 0);
347 for(int i = 0; i < iAniCount; i++) // Animation Key 추가.
348 {
349 __AnimData* pAD = pAniCtrl->DataGet(i);
350 if(pAD->fFrmEnd < pAD->fFrmStart) pAD->fFrmEnd = pAD->fFrmStart; // 같은 프레임으로 만든다..
351
352 int iS = (int)(pAD->fFrmStart / fSamplingRate);
353 int iE = (int)(pAD->fFrmEnd / fSamplingRate); // 반올림하려고 0.5f 넣었다..
354
355 if(iE < iS) continue;
356
357 JointDest.AddKey(pJointSrc, iS, iE); // 키를 추가하고 복사한다..
358 fFrmOffsets[i] = (iE - iS + 1) * fSamplingRate; // 각 데이터의 Offset 계산..
359 fFrmStarts[i] = pAD->fFrmStart - (iS * fSamplingRate);
360 }
361
362 float fOffsetCur = fSamplingRate;
363 for(int i = 0; i < iAniCount; i++) // Animation Key Frame 다시 계산..
364 {
365 __AnimData* pAD = pAniCtrl->DataGet(i);
366
367 pAD->Offset(fOffsetCur - pAD->fFrmStart + fFrmStarts[i]); // 모든 데이터를 Offset만큼 값을 적용..
368
369 fOffsetCur += fFrmOffsets[i];
370 }
371
372 pAniCtrl->SaveToFile();
373 JointDest.SaveToFile();
374
375 pChr->Tick(0.0f); // 초기위치로 Tick
376 pChr->JointSet("");
377 pChr->JointSet(JointDest.FileName());
378
379 UpdateAllViews(nullptr); // 모든 뷰 초기화..
380}
