42bool CPlayerOther::Init(e_Race eRace, int iFace, int iHair, uint32_t* pdwItemIDs, int* piItenDurabilities)
43{
44 if(nullptr == pdwItemIDs || nullptr == piItenDurabilities) return false;
45
46 m_InfoBase.eRace = eRace;
47 m_InfoExt.iFace = iFace;
48 m_InfoExt.iHair = iHair;
49
50 // 이제 패킷에 따라 캐릭터를 치장..(?) 시켜준다.. 아이템장착, 무기 장착등...
51 __TABLE_PLAYER_LOOKS* pLooks = s_pTbl_UPC_Looks.Find(eRace); // 테이블에서 기본 스킨 ..
52 if(nullptr == pLooks)
53 {
54 CLogWriter::Write("CPlayerOther::Init() Basic Resource Pointer is NULL Race({})",
55 static_cast<int>(eRace));
56 return false;
57 }
58 this->InitChr(pLooks); // 관절 세팅..
59
60 for(int i = 0; i < MAX_ITEM_SLOT_OPC; i++)
61 {
62 std::string szFN;
63 e_PartPosition ePart = PART_POS_UNKNOWN;
64 e_PlugPosition ePlug = PLUG_POS_UNKNOWN;
65 e_ItemSlot eSlot = ITEM_SLOT_UNKNOWN;
66 __TABLE_ITEM_BASIC* pItem = nullptr;
67 __TABLE_ITEM_EXT* pItemExt = nullptr;
68
69 if(0 == pdwItemIDs[i])
70 {
71 if(0 == i) { ePart = PART_POS_UPPER; szFN = pLooks->szPartFNs[0]; }
72 else if(1 == i) { ePart = PART_POS_LOWER; szFN = pLooks->szPartFNs[1]; }
73// else if(2 == i) { ePart = PART_POS_HAIR_HELMET; szFN = pLooks->szPartFNs[5]; }
74 else if(3 == i) { ePart = PART_POS_HANDS; szFN = pLooks->szPartFNs[3]; }
75 else if(4 == i) { ePart = PART_POS_FEET; szFN = pLooks->szPartFNs[4]; }
76 else if(5 == i) { } // 망토
77// else if(6 == i) { ePlug = PLUG_POS_RIGHTHAND; }
78// else if(7 == i) { ePlug = PLUG_POS_LEFTHAND; }
79 }
80 else
81 {
82 pItem = s_pTbl_Items_Basic.Find(pdwItemIDs[i]/1000*1000); // 유저 플레이어 아이템 얻기..
83 if(pItem && pItem->byExtIndex >= 0 && pItem->byExtIndex < MAX_ITEM_EXTENSION)
84 pItemExt = s_pTbl_Items_Exts[pItem->byExtIndex].Find(pdwItemIDs[i]%1000);
85 if(nullptr == pItem || nullptr == pItemExt)
86 {
87 __ASSERT(0, "NULL Item!!!");
88 continue;
89 }
90
91
92 e_ItemType eType = CGameBase::MakeResrcFileNameForUPC(pItem, pItemExt, &szFN, nullptr, ePart, ePlug, m_InfoBase.eRace); // 리소스 파일 이름을 만들고..
93
94 if(0 == i) { ePart = PART_POS_UPPER; eSlot = ITEM_SLOT_UPPER; }
95 else if(1 == i) { ePart = PART_POS_LOWER; eSlot = ITEM_SLOT_LOWER; }
96 else if(2 == i) { ePart = PART_POS_HAIR_HELMET; eSlot = ITEM_SLOT_HEAD; }
97 else if(3 == i) { ePart = PART_POS_HANDS; eSlot = ITEM_SLOT_GLOVES; }
98 else if(4 == i) { ePart = PART_POS_FEET; eSlot = ITEM_SLOT_SHOES; }
99 else if(5 == i) { } // 망토
100 else if(6 == i) { ePlug = PLUG_POS_RIGHTHAND; eSlot = ITEM_SLOT_HAND_RIGHT; }
101 else if(7 == i) { ePlug = PLUG_POS_LEFTHAND; eSlot = ITEM_SLOT_HAND_LEFT; }
102 }
103
104 if(PART_POS_UPPER == ePart || PART_POS_LOWER == ePart || PART_POS_HANDS == ePart || PART_POS_FEET == ePart)
105 {
106 this->PartSet(ePart, szFN, pItem, pItemExt);
107 }
108 else if(PART_POS_HAIR_HELMET == ePart) // 머리카락 혹은 헬멧이면..
109 {
110 this->PartSet(ePart, szFN, pItem, pItemExt);
111 }
112 else if(5 == i)
113 {
114 }
115 else if((6 == i || 7 == i) && PLUG_POS_UNKNOWN != ePlug)
116 {
117 this->PlugSet(ePlug, szFN, pItem, pItemExt);
118 }
119
120 if(ITEM_SLOT_UNKNOWN != eSlot)
121 this->DurabilitySet(eSlot, piItenDurabilities[i]);
122 }
123
124 // 얼굴은 따로하자..
125 this->InitFace();
126 CN3CPart* pPartHairHelmet = this->Part(PART_POS_HAIR_HELMET);
127 if(pPartHairHelmet->FileName().empty()) // 헬멧에 해당되는게 없으면.. 머리카락 붙이기..
128 this->InitHair();
129
130 return true;
131}

137 // 아이템이 있고 얼굴 이름이 있으면..

141 char szDir[_MAX_DIR] = {}, szFName[_MAX_FNAME] = {}, szExt[_MAX_EXT] = {};

166void CPlayerOther::KnightsInfoSet(int iID, const std::string& szName, int iGrade, int iRank)
167{
168 CPlayerBase::KnightsInfoSet(iID, szName, iGrade, iRank);
169
170 m_InfoExt.szKnights = szName;
171 m_InfoExt.iKnightsGrade = iGrade;
172 m_InfoExt.iKnightsRank = iRank;
173
174 if (m_InfoExt.szKnights.empty())
175 {
176 delete m_pClanFont;
177 m_pClanFont = nullptr;
178 }
179
180 float fDist = Distance(CGameBase::s_pPlayer->Position());
181 if (fDist < SOUND_RANGE_TO_SET)
182 {
183 if (m_pClanFont == nullptr)
184 {
185 std::string szFontID = fmt::format_text_resource(IDS_FONT_ID);
186
187 m_pClanFont = new CDFont(szFontID, 12);
188 m_pClanFont->InitDeviceObjects(s_lpD3DDev);
189 m_pClanFont->RestoreDeviceObjects();
190 }
191
192 m_pClanFont->SetText(m_InfoExt.szKnights.c_str(), D3DFONT_BOLD); // 폰트에 텍스트 지정.
193 m_pClanFont->SetFontColor(KNIGHTS_FONT_COLOR);
194 }
