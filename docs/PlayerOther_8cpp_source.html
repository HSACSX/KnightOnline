39{
40 if (nullptr == pdwItemIDs || nullptr == piItenDurabilities)

43 m_InfoBase.eRace = eRace;
44 m_InfoExt.iFace = iFace;
45 m_InfoExt.iHair = iHair;
46
47 // 이제 패킷에 따라 캐릭터를 치장..(?) 시켜준다.. 아이템장착, 무기 장착등...

52 return false;
53 }
54
55 InitChr(pLooks); // 관절 세팅..
56
57 for (int i = 0; i < MAX_ITEM_SLOT_OPC; i++)
58 {
59 std::string szFN;
60 e_PartPosition ePart = PART_POS_UNKNOWN;
61 e_PlugPosition ePlug = PLUG_POS_UNKNOWN;
62 e_ItemSlot eSlot = ITEM_SLOT_UNKNOWN;
63 __TABLE_ITEM_BASIC* pItem = nullptr;
64 __TABLE_ITEM_EXT* pItemExt = nullptr;

66 if (0 == pdwItemIDs[i])
67 {
68 if (0 == i)
69 {
70 ePart = PART_POS_UPPER;
71 szFN = pLooks->szPartFNs[0];
72 }
73 else if (1 == i)
74 {
75 ePart = PART_POS_LOWER;
76 szFN = pLooks->szPartFNs[1];

78 // else if(2 == i) { ePart = PART_POS_HAIR_HELMET; szFN = pLooks->szPartFNs[5]; }

204void CPlayerOther::KnightsInfoSet(int iID, const std::string& szName, int iGrade, int iRank)
205{
206 CPlayerBase::KnightsInfoSet(iID, szName, iGrade, iRank);
207
208 m_InfoExt.szKnights = szName;
209 m_InfoExt.iKnightsGrade = iGrade;
210 m_InfoExt.iKnightsRank = iRank;
211
212 if (m_InfoExt.szKnights.empty())
213 {
214 delete m_pClanFont;
215 m_pClanFont = nullptr;
216 }
217
218 float fDist = Distance(CGameBase::s_pPlayer->Position());
219 if (fDist < SOUND_RANGE_TO_SET)
220 {
221 if (m_pClanFont == nullptr)
222 {
223 std::string szFontID = fmt::format_text_resource(IDS_FONT_ID);
224
225 m_pClanFont = new CDFont(szFontID, 12);
226 m_pClanFont->InitDeviceObjects(s_lpD3DDev);
227 m_pClanFont->RestoreDeviceObjects();
228 }
229
230 m_pClanFont->SetText(m_InfoExt.szKnights.c_str(), D3DFONT_BOLD); // 폰트에 텍스트 지정.
231 m_pClanFont->SetFontColor(KNIGHTS_FONT_COLOR);
232 }
