94void CTransDummy::Tick()
95{
96 if (m_SelObjArray.GetSize()==0) return;
97 // Scale 조정
98 __Vector3 vL = s_CameraData.vEye - m_vPos;
99 float fL = vL.Magnitude()*0.01f;
100 m_vScale.Set(fL, fL, fL);
101
102 CN3Transform::Tick(-1000.0f);
103 ReCalcMatrix();
104
105 // 거리에 따라 정렬
106 int i;
107 for (i=0; i<NUM_DUMMY; ++i)
108 {
109 __Vector3 vPos = m_DummyCubes[i].vCenterPos*m_Matrix;
110 m_DummyCubes[i].fDistance = (vPos - s_CameraData.vEye).Magnitude();
111 }
112 for (i=0; i<NUM_DUMMY; ++i) m_pSortedCubes[i] = &(m_DummyCubes[i]);
113 qsort(m_pSortedCubes, NUM_DUMMY, sizeof(__DUMMYCUBE*), SortCube);
114}

184__DUMMYCUBE* CTransDummy::Pick(int x, int y)
185{
186 CPick pick;
187 pick.SetPickXY(x, y);
188
189 int i, j;
190 for (i=NUM_DUMMY-1; i>=0; --i)
191 {
192 for (j=0; j<12; ++j)
193 {
194 __Vector3 v1, v2, v3;
195 __VertexXyzNormalColor* pVertex;
196
197 ASSERT(m_pSortedCubes[i]);
198 pVertex = m_pSortedCubes[i]->Vertices+j*3+0;
199 v1.Set(pVertex->x, pVertex->y, pVertex->z);
200 pVertex = m_pSortedCubes[i]->Vertices+j*3+1;
201 v2.Set(pVertex->x, pVertex->y, pVertex->z);
202 pVertex = m_pSortedCubes[i]->Vertices+j*3+2;
203 v3.Set(pVertex->x, pVertex->y, pVertex->z);
204
205 v1 *= m_Matrix; v2 *= m_Matrix; v3 *= m_Matrix;
206
207 __Vector3 vPos; float t, u, v;
208 if (pick.IntersectTriangle(v1, v2, v3, t, u, v, &vPos)) return m_pSortedCubes[i];
209 }
210 }
211
212 return nullptr;
213}

215BOOL CTransDummy::MouseMsgFilter(LPMSG pMsg)
216{
217 int iSize = static_cast<int>(m_SelObjArray.GetSize());
218 if (iSize == 0)
219 return FALSE;
220
221 switch (pMsg->message)
222 {
223 case WM_LBUTTONDOWN:
224 {
225 POINT point = {short(LOWORD(pMsg->lParam)), short(HIWORD(pMsg->lParam))};
226 m_pSelectedCube = Pick(point.x, point.y);
227 if (m_pSelectedCube)
228 {
229 CN3Transform* pSelObj0 = m_SelObjArray.GetAt(0);
230 _ASSERT(pSelObj0);
231 m_vPrevPos = pSelObj0->Pos();
232 m_qPrevRot = pSelObj0->Rot();
233 if (m_vPrevScaleArray) {delete [] m_vPrevScaleArray; m_vPrevScaleArray = nullptr;}
234 m_vPrevScaleArray = new __Vector3[iSize];
235 for (int i=0; i<iSize; ++i) // 모든 선택된 객체의 스케일 저장
236 {
237 CN3Transform* pSelObj = m_SelObjArray.GetAt(i);
238 _ASSERT(pSelObj);
239 m_vPrevScaleArray[i] = pSelObj->Scale();
240 }
241
242 SetCapture(pMsg->hwnd);
243 return TRUE;
244 }
245 }
246 break;
247 case WM_LBUTTONUP:
248 {
249 if (m_pSelectedCube)
250 {
251 ReleaseCapture();
252 m_pSelectedCube = nullptr;
253 return TRUE;
254 }
255 }
256 break;
257 case WM_RBUTTONUP: // 큐브 선택 취소 및 이번 드래그로 움직인것 되돌려 놓기
258 {
259 if (m_pSelectedCube)
260 {
261 __Vector3 vDiffPos = m_vPrevPos - m_vPos;
262// __Quaternion qDiffRot = m_qPrevRot - m_qRot;
263 __Vector3 vDiffScale; vDiffScale.Set(1.0f, 1.0f, 1.0f);
264// TransDiff(&vDiffPos, &vDiffRot, &vDiffScale);
265 TransDiff(&vDiffPos, nullptr, &vDiffScale);
266
267 m_vPos = m_vPrevPos;
268 m_qRot = m_qPrevRot;
269 _ASSERT(m_vPrevScaleArray);
270 m_vScale = m_vPrevScaleArray[0];
271

275 return TRUE;
276 }
277 }

283void CTransDummy::GetPickRay(POINT point, __Vector3& vDir, __Vector3& vOrig)
284{
285 LPDIRECT3DDEVICE9 lpD3DDev = s_lpD3DDev;
286
287 // Get the pick ray from the mouse position
288 __Matrix44 matProj;
289 lpD3DDev->GetTransform(D3DTS_PROJECTION, matProj.toD3D());
290
291 // Compute the vector of the pick ray in screen space
292 __Vector3 v;
293 v.x = ( ( ( 2.0f * point.x ) / (s_CameraData.vp.Width) ) - 1 ) / matProj.m[0][0];
294 v.y = -( ( ( 2.0f * point.y ) / (s_CameraData.vp.Height) ) - 1 ) / matProj.m[1][1];
295 v.z = 1.0f;
296
297 // Get the inverse view matrix
298 __Matrix44 matView, m;
299 lpD3DDev->GetTransform(D3DTS_VIEW, matView.toD3D());
300 m = matView.Inverse();
301
302 // Transform the screen space pick ray into 3D space
303 vDir.x = v.x*m.m[0][0] + v.y*m.m[1][0] + v.z*m.m[2][0];
304 vDir.y = v.x*m.m[0][1] + v.y*m.m[1][1] + v.z*m.m[2][1];
305 vDir.z = v.x*m.m[0][2] + v.y*m.m[1][2] + v.z*m.m[2][2];
306 vOrig.x = m.m[3][0];
307 vOrig.y = m.m[3][1];
308 vOrig.z = m.m[3][2];
309}

311void CTransDummy::TransDiff(__Vector3* pvDiffPos, __Quaternion* pqDiffRot, __Vector3* pvDiffScale)
312{
313 int iSize = static_cast<int>(m_SelObjArray.GetSize());
314 if (iSize <= 0)
315 return;
316
317 if (pvDiffPos != nullptr)
318 {
319 for (int i = 0; i < iSize; i++)
320 {
321 CN3Transform* pSelObj = m_SelObjArray.GetAt(i);
322 _ASSERT(pSelObj);
323 pSelObj->PosSet(pSelObj->Pos() + (*pvDiffPos));
324 }
325 }
326
327 if (pqDiffRot != nullptr)
328 {
329 CN3Transform* pSelObj = m_SelObjArray.GetAt(0);
330 __Vector3 vCenter = pSelObj->Pos();
331 __Vector3 vPos;
332 __Matrix44 mtx44Rotate = *pqDiffRot;
333
334 for (int i = 0; i < iSize; i++)
335 {
336 pSelObj = m_SelObjArray.GetAt(i);
337 _ASSERT(pSelObj);
338
339 __Quaternion qtRot = pSelObj->Rot();
340 qtRot *= (*pqDiffRot);
341 pSelObj->RotSet(qtRot);
342
343 vPos = pSelObj->Pos(); // 맵상에서의 위치
344 vPos -= vCenter;
345 vPos *= mtx44Rotate;
346 vPos += vCenter;
347 pSelObj->PosSet(vPos);
348 }
349 }
350
351 if (pvDiffScale != nullptr)
352 {
353 for (int i = 0; i < iSize; i++)
354 {
355 CN3Transform* pSelObj = m_SelObjArray.GetAt(i);
356 _ASSERT(pSelObj && m_vPrevScaleArray);
357 __Vector3 vScale;
358 vScale.x = m_vPrevScaleArray[i].x * pvDiffScale->x;
359 vScale.y = m_vPrevScaleArray[i].y * pvDiffScale->y;
360 vScale.z = m_vPrevScaleArray[i].z * pvDiffScale->z;
361 pSelObj->ScaleSet(vScale);
362 }
363 }
