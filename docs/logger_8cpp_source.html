19void logger::Logger::Setup(CIni& ini, const std::filesystem::path& baseDir)
20{
21 // setup file logger
22 std::string fileName = ini.GetString(ini::LOGGER, ini::FILE, _defaultLogPath);
23
24 std::filesystem::path configuredLogPath(fileName);
25 if (configuredLogPath.is_relative())
26 configuredLogPath = baseDir / fileName;
27
28 std::u8string utf8String = configuredLogPath.u8string();
29 fileName.assign(utf8String.begin(), utf8String.end());
30
31 _fxtuiSink = std::make_shared<ftxui::sink_mt>();
32
33 auto fileLogger = std::make_shared<spdlog::sinks::daily_file_format_sink_mt>(fileName, 0, 0);
34
35 std::string logPattern = ini.GetString(ini::LOGGER, ini::PATTERN, ini::DEFAULT_LOG_PATTERN);
36 fileLogger->set_pattern(logPattern);
37
38 // setup console logger
39 auto consoleLogger = std::make_shared<spdlog::sinks::stdout_color_sink_mt>();
40
41 std::string consoleLogPattern = ini.GetString(
42 ini::LOGGER, ini::CONSOLE_PATTERN, ini::DEFAULT_CONSOLE_LOG_PATTERN);
43 consoleLogger->set_pattern(consoleLogPattern);
44
45 _fxtuiSink->set_pattern(consoleLogPattern);
46 _fxtuiSink->set_console_sink(consoleLogger);
47
48 spdlog::init_thread_pool(MessageQueueSize, ThreadPoolSize);
49
50 // setup multi-sink async logger as default (combines file + console logger [under ftxui])
51 spdlog::sinks_init_list sinks = { fileLogger, _fxtuiSink };
52
53 auto threadPool = spdlog::thread_pool();
54 auto appLogger = std::make_shared<spdlog::async_logger>(
55 _appName, sinks.begin(), sinks.end(), threadPool, spdlog::async_overflow_policy::block);
56 spdlog::set_default_logger(appLogger);
57
58 // set default logger level and pattern
59 // we default to debug level logging
60 int logLevel = ini.GetInt(ini::LOGGER, ini::LEVEL, spdlog::level::debug);
61 spdlog::set_level(static_cast<spdlog::level::level_enum>(logLevel));
62
63 // add any extra loggers
64 SetupExtraLoggers(ini, threadPool, baseDir);
65
66 // periodically flush all *registered* loggers every 3 seconds:
67 // warning: only use if all your loggers are thread-safe ("_mt" loggers)
68 spdlog::flush_every(std::chrono::seconds(3));
69
70 spdlog::info("{} logger configured", _appName);
71}
