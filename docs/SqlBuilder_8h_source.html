14 /// \brief Used to create safe T-SQL statements for database ModelType objects
15 template <typename ModelType>
16 class SqlBuilder

18 public:
19 /// \brief Limits the size of the result set. Defaults to 0 for no limit
20 int64_t Limit = 0;
21
22 /// \brief adds additional SQL clause after any WHERE statements,
23 /// such as an "ORDER BY"
24 std::string PostWhereClause;
25
26 /// \brief adds a WHERE statement manually specified by the caller. Takes
27 /// priority over IsWherePK. Caller should include "WHERE" keyword when
28 /// setting this string
29 /// \see IsWherePK
30 std::string Where;
31
32 /// \brief will cause SelectString() to use a WHERE statement using PK columns
33 bool IsWherePK = false;

35 /// \brief returns a select query string based on any configured modifiers
36 std::string SelectString()
37 {
38 std::string query = "SELECT ";
39
40 if (Limit > 0)
41 query += "TOP " + std::to_string(Limit) + " ";
42
43 if (selectCols.empty())
44 {
45 // fill with all supported ModelType bindings as a default
46 fillSelectColumns();
47 }
48
49 const std::unordered_set<std::string>& modelBlobColumns
50 = ModelType::BlobColumns();
51
52 std::vector<std::string> deferredBlobColumns;
53 deferredBlobColumns.reserve(modelBlobColumns.size());
54
55 short i = 0;
56 for (const std::string& col : selectCols)
57 {
58 // long-data blob columns aren't always stored with the result set
59 // they need to be fetched after all regular columns, so we should
60 // defer them until the end of the list
61 if (modelBlobColumns.contains(col))
62 {
63 deferredBlobColumns.push_back(col);
64 continue;
65 }
66
67 if (i > 0)
68 query += ", ";
69
70 query += '[' + col + ']';
71 i++;
72 }
73
74 // all deferred long-data blob columns can now be added
75 // to the end of our list to ensure they will be fetched
76 for (const std::string& col : deferredBlobColumns)
77 {
78 if (i > 0)
79 query += ", ";
80
81 query += '[' + col + ']';
82 i++;
83 }
84
85 query += " FROM [" + ModelType::TableName() + ']';
86
87 if (!Where.empty())
88 {
89 query += " " + Where;
90 }
91 else if (IsWherePK && !ModelType::PrimaryKey().empty())
92 {
93 query += " WHERE ";
94 i = 0;
95 for (const std::string& col : ModelType::PrimaryKey())
96 {
97 if (i > 0)
98 query += " AND ";
99 query += '[' + col + "] = ?";
100 i++;
101 }
102 }
103
104 if (!PostWhereClause.empty())
105 {
106 query += " " + PostWhereClause;
107 }
108
109 spdlog::trace("using query: {}", query);
110 return query;
111 }

186 /// \brief returns a select query built using the table's PK. Delete statements
187 /// not using the PK should be carefully hand coded.
188 /// \note does not support manual use of Where
189 std::string DeleteByIdString()
190 {
191 std::string query;
192 // do not create queries for tables with defined PKs
193 if (ModelType::PrimaryKey().empty())
194 {
195 return query;
196 }
197
198 query = "DELETE FROM [" + ModelType::TableName() + "] WHERE ";
199 short i = 0;
200 for (const std::string& col : ModelType::PrimaryKey())
201 {
202 if (i > 0)
203 query += " AND ";
204 query += '[' + col + "] = ?";
205 i++;
206 }
207
208 spdlog::trace("using query: {}", query);
209 return query;
210 }

212 /// \brief sets the columns for a select statement to a subset of bindable values
213 ///
214 /// \param columns list of columns to select
215 /// \warning if an input column is not valid for binding it will be ignored and
216 /// a warn message will be logged
217 void SetSelectColumns(const std::vector<std::string>& columns)
218 {
219 selectCols.clear();
220 for (const std::string& col : columns)
221 {
222 if (!isValidColumnName(col))
223 {
224 spdlog::warn("Invalid column name: {}.{}", ModelType::TableName(), col);
225 continue;
226 }
227
228 selectCols.insert(col);
229 }
230 }
231
232 /// \brief excludes a list of columns from the current binding set
233 ///
234 /// \param columns list of columns to exclude from the query
235 void ExcludeColumns(const std::vector<std::string>& columns)
236 {
237 // if selectColumns hasn't been populated, assume the default of SELECT * and filter from that
238 if (selectCols.empty())
239 fillSelectColumns();
240
241 for (const std::string& columnName : columns)
242 selectCols.erase(columnName);
243 }
244
245 private:
246 // selectCols is a set of columns that will be used as part of a SELECT query.
247 // we use an std::set to prevent duplicate column entries
248 std::unordered_set<std::string> selectCols;
249
250 inline bool isValidColumnName(const std::string& colName) const
251 {
252 return ModelType::ColumnNames().contains(colName);
253 }
254
255 /// \brief sets selectedCols to the ModelType's supported column set
256 void fillSelectColumns()
257 {
258 selectCols.clear();
259 selectCols.insert(ModelType::ColumnNames().begin(), ModelType::ColumnNames().end());
260 }
261 };
