68void CUIRepairTooltipDlg::DisplayTooltipsEnable(int xpos, int ypos, __IconItemSkill* spItem, int iRequiredGold, bool bHaveEnough)
69{
70 int iIndex = 0;
71 if (!spItem)
72 return;
73
74 if (!IsVisible())
75 SetVisible(TRUE);
76
77 std::string szMsg;
78
79 if ((m_iPosXBack != xpos) || (m_iPosYBack != ypos))
80 {
81 int iWidth = m_rcRegion.right - m_rcRegion.left;
82 int iHeight = m_rcRegion.bottom - m_rcRegion.top;
83
84 int iRight, iTop;
85 iRight = 1024;
86 iTop = 0;
87
88 if ((xpos + iWidth < iRight) && ((ypos - iHeight) > iTop))
89 SetPos(xpos + 26, ypos - iHeight);
90 else if ((xpos + iWidth > iRight) && ((ypos - iHeight) < iTop))
91 SetPos(xpos - iWidth, ypos);
92 else if (xpos + iWidth > iRight)
93 SetPos(xpos - iWidth, ypos - iHeight);
94 else
95 SetPos(xpos + 26, ypos);
96
97 m_iPosXBack = xpos;
98 m_iPosYBack = ypos;
99 }
100
101 if ((!m_spItemBack) || (m_spItemBack->pItemBasic->dwID != spItem->pItemBasic->dwID) || (m_iBRequiredGoldBackup != m_iBRequiredGold))
102 {
103 m_spItemBack = spItem;
104
105 std::string szString;
106
107 if ((spItem->pItemBasic->byContable != UIITEM_TYPE_COUNTABLE) && (spItem->pItemBasic->byContable != UIITEM_TYPE_COUNTABLE_SMALL)
108 && (spItem->pItemBasic->siMaxDurability + spItem->pItemExt->siMaxDurability != 1))
109 {
110 szMsg = fmt::format_text_resource(IDS_TOOLTIP_NOAH, iRequiredGold);
111
112 SetTooltipTextColor(bHaveEnough);
113 if (!m_pStr[iIndex]->IsVisible())
114 m_pStr[iIndex]->SetVisible(true);
115 m_pStr[iIndex]->SetString(szMsg);
116 iIndex++;
117
118 szMsg = fmt::format_text_resource(
119 IDS_TOOLTIP_MAX_DURABILITY, spItem->pItemBasic->siMaxDurability + spItem->pItemExt->siMaxDurability);
120
121 if (!m_pStr[iIndex]->IsVisible())
122 m_pStr[iIndex]->SetVisible(true);
123 m_pStr[iIndex]->SetString(szMsg);
124 iIndex++;
125
126 szMsg = fmt::format_text_resource(IDS_TOOLTIP_CUR_DURABILITY, spItem->iDurability);
127
128 if (!m_pStr[iIndex]->IsVisible())
129 m_pStr[iIndex]->SetVisible(true);
130 m_pStr[iIndex]->SetString(szMsg);
131 iIndex++;
132
133 szMsg = fmt::format_text_resource(IDS_TOOLTIP_REPAIR_PRICE);
134
135 if (!m_pStr[iIndex]->IsVisible())
136 m_pStr[iIndex]->SetVisible(true);
137 m_pStr[iIndex]->SetString(szMsg);
138 iIndex++;
139
140 for (int i = iIndex; i < MAX_REPAIR_TOOLTIP_COUNT; i++)
141 m_pStr[i]->SetString("");
142 }
143 else
144 {
145 szMsg = fmt::format_text_resource(IDS_TOOLTIP_CANNOT);
146
147 m_pStr[0]->SetVisible(false);
148 m_pStr[1]->SetVisible(false);
149 m_pStr[2]->SetVisible(false);
150 m_pStr[3]->SetString(szMsg);
151 m_pStr[3]->SetVisible(true);
152 }
153 }
154
155 m_iBRequiredGoldBackup = m_iBRequiredGold;
156 CN3UIBase::Render();
157}
