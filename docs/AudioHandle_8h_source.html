13/// \enum e_AudioHandleType
14/// \brief Identifies the playback audio handle type.
15///
16/// Used to distinguish between buffered and streamed audio playback
17/// mechanisms.
18enum e_AudioHandleType : uint8_t

32/// \class AudioHandle
33/// \brief Base class representing an active audio playback instance.
34///
35/// AudioHandle encapsulates playback state, timing information, and
36/// configuration shared between buffered and streamed audio playback.
37/// Concrete subclasses manage backend-specific resources.
38class AudioHandle
39{
40public:
41 /// Type of audio handle (buffered or streamed).
42 e_AudioHandleType HandleType;
43
44 /// Indicates whether this handle is currently managed by the audio thread.
45 ///
46 /// The flag is set when the handle is queued for decoding/playback in
47 /// the AudioThread (i.e. via AudioThread::Add) and cleared when the
48 /// handle is removed from the AudioThread.
49 ///
50 /// \ref CN3SndObj::Tick() checks this flag to detect when the handle is
51 /// no longer active; if false, the object releases its reference to the
52 /// handle, restoring it to its associated pool for reuse.
53 std::atomic<bool> IsManaged;
54
55 /// Indicates whether playback has started.
56 bool StartedPlaying;
57
58 /// Indicates whether playback has finished.
59 ///
60 /// This flag can only be true if \ref StartedPlaying was previously true.
61 /// It is set when the audio playback completes or reaches the end of the
62 /// sound/stream.
63 bool FinishedPlaying;
64
65 /// OpenAL source identifier associated with this handle.
66 uint32_t SourceId;
67
68 /// Audio asset supplying data for playback.
69 std::shared_ptr<AudioAsset> Asset;
70
71 /// Current sound playback state.
72 ///
73 /// Indicates the lifecycle state of this audio handle, as defined by
74 /// \ref e_SndState. Typical states include initial, playing, paused, or
75 /// stopped. This is used by the audio engine to manage playback
76 /// progression, transitions, and cleanup.
77 e_SndState State;
78
79 /// Shared sound settings for this audio handle.
80 ///
81 /// Contains volume settings. While the handle
82 /// can access and modify these settings, they are not exclusively owned
83 /// by the handle. For example, if the volume is changed, the new value
84 /// persists in \ref CN3SndObj for subsequent playback of the same sound,
85 /// ensuring consistency across uses. This allows both the audio engine
86 /// and the handle instance to reference the same configuration.
87 std::shared_ptr<SoundSettings> Settings;
88
89 /// Duration of fade-in effect, in seconds.
90 float FadeInTime;
91
92 /// Duration of fade-out effect, in seconds.
93 float FadeOutTime;
94
95 /// Delay before playback begins, in seconds.
96 float StartDelayTime;
97
98 /// Internal playback timer used for state tracking, in seconds.
99 float Timer;
100
101 /// Constructs an audio handle with default state.
102 AudioHandle();
103
104 /// Virtual destructor to allow safe deletion via base-class pointers.
105 virtual ~AudioHandle();
106};
