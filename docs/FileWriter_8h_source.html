10/// \brief FileWriter implements a write-capable file interface.
11///
12/// Supports writing to a file, seeking, and flushing data to disk.
13/// Reading operations are not supported. Files may be extended by seeking past the current size.
14class FileWriter : public File
15{
16public:
17 /// \brief Gets the size of the file on disk.
18 /// \return Current file size on disk.
19 uint64_t SizeOnDisk() const
20 {
21 return _sizeOnDisk;
22 }
23
24 /// \brief Constructs a new FileWriter object.
25 FileWriter();
26
27 /// \brief Opens an existing file for writing.
28 /// \param path Path to the file to open.
29 /// \returns true if the file was successfully opened, false otherwise.
30 /// \note The initial write offset starts at 0, not at the end of the file.
31 /// The caller must manually seek to the end if they intend to append.
32 /// This matches Win32 behaviour, as well as general C file I/O behaviour
33 /// as we otherwise have no explicit append flag to rely upon.
34 bool OpenExisting(const std::filesystem::path& path) override;
35
36 /// \brief Creates a new file for writing, truncating if it exists.
37 /// \param path Path to the file to create.
38 /// \returns true if the file was created successfully, false otherwise.
39 bool Create(const std::filesystem::path& path) override;
40
41 /// \brief Reading is not supported for FileWriter.
42 /// \param buffer Destination buffer for data.
43 /// \param bytesToRead Number of bytes to read.
44 /// \param bytesRead Optional output for number of bytes actually read.
45 /// \returns false always.
46 bool Read(void* buffer, size_t bytesToRead, size_t* bytesRead = nullptr) override;
47
48 /// \brief Writes data to the file at the current offset.
49 /// \param buffer Data to write.
50 /// \param bytesToWrite Number of bytes to write.
51 /// \param bytesWritten Optional output for number of bytes actually written.
52 /// \returns true if all requested bytes were written, false otherwise.
53 /// \note Writing beyond the current file size automatically expands the internal size.
54 bool Write(const void* buffer, size_t bytesToWrite, size_t* bytesWritten = nullptr) override;
55
56 /// \brief Changes the current write offset.
57 /// \param offset Offset value.
58 /// \param origin One of SEEK_SET, SEEK_CUR, SEEK_END.
59 /// \returns true if the new offset is valid, false otherwise.
60 /// \note Seeking past the current file size will expand the internal file size but does not
61 /// immediately update the file on-disk until Flush() or Close().
62 bool Seek(int64_t offset, int origin) override;
63
64 /// \brief Flushes any buffered writes to disk.
65 /// \note This will expand the on-disk file if the offset moved beyond the previous size.
66 void Flush() override;
67
68 /// \brief Closes the file and flush any pending writes.
69 /// \returns true if the file was open and closed successfully, false if already closed.
70 bool Close() override;
71
72 /// \brief Destroys the FileWriter object.
73 ~FileWriter() override;
74
75protected:
76 LLFIO_V2_NAMESPACE::file_handle _fileHandle; ///< File handle used for writing.
77 uint64_t _sizeOnDisk; ///< Current size of the file on disk.
78};
