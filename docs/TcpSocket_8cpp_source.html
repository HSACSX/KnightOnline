22int TcpSocket::QueueAndSend(char* buffer, int length)
23{
24 std::lock_guard<std::recursive_mutex> lock(_sendMutex);
25
26 if (_pendingDisconnect)
27 return -1;
28
29 // Add this packet to the circular buffer.
30 // Ensure we do not allow resizing; we do not want these pointers invalidated.
31 auto span = _sendCircularBuffer.PutData(buffer, length, false);
32 if (span.Buffer1 != nullptr && span.Length1 > 0)
33 {
34 auto queuedSend = std::make_unique<QueuedSend>();
35 queuedSend->IsOwned = false;
36 queuedSend->BufferSpan = span;
37 _sendQueue.push(std::move(queuedSend));
38 }
39 // Failed to add to the buffer, it has no room.
40 // Allocate and queue.
41 else
42 {
43 auto queuedSend = std::make_unique<QueuedSend>();
44 queuedSend->IsOwned = true;
45 queuedSend->BufferSpan.Buffer1 = new char[length];
46 queuedSend->BufferSpan.Length1 = length;
47 memcpy(queuedSend->BufferSpan.Buffer1, buffer, length);
48 _sendQueue.push(std::move(queuedSend));
49 }
50
51 if (!AsyncSend(false))
52 return -1;
53
54 return length;
55}

57bool TcpSocket::AsyncSend(bool fromAsyncChain)
58{
59 std::unique_lock<std::recursive_mutex> lock(_sendMutex);
60
61 // When we finish a send, we should pop the last entry before queueing up another send.
62 if (fromAsyncChain)
63 {
64 assert(_sendInProgress);
65 _sendInProgress = false;
66
67 if (!_sendQueue.empty())
68 _sendQueue.pop();
69
70 if (_sendQueue.empty())
71 {
72 // Re-queue a disconnect now that all sends are complete
73 if (_pendingDisconnect)
74 {
75 lock.unlock();
76 Close();
77 }
78
79 // Send queue is empty, nothing more to queue up.
80 // Consider this successful.
81 return true;
82 }
83 }
84 else
85 {
86 // Send queue is empty, nothing more to queue up.
87 // Consider this successful.
88 if (_sendQueue.empty())
89 return true;
90
91 // Send currently in progress.
92 // Don't attempt to write; it's in the queue, it'll be processed once the send is completed.
93 // Consider this successful; it is in the queue.
94 if (_sendInProgress)
95 return true;
96 }
97
98 // Fetch the next entry to send.
99 // Note that we keep this in the queue until the send completes.
100 const auto& queuedSend = _sendQueue.front();
101 const auto& span = queuedSend->BufferSpan;
102
103 try
104 {
105 if (span.Buffer2 != nullptr && span.Length2 > 0)
106 {
107 std::array<asio::const_buffer, 2> buffers;
108 buffers[0] = asio::buffer(span.Buffer1, span.Length1);
109 buffers[1] = asio::buffer(span.Buffer2, span.Length2);
110
111 if (_socket != nullptr)
112 {
113 _socket->async_write_some(
114 buffers, std::bind(&SocketManager::OnPostSend, _socketManager,
115 std::placeholders::_1, std::placeholders::_2, this));
116 }
117 }
118 else
119 {
120 if (_socket != nullptr)
121 {
122 _socket->async_write_some(asio::buffer(span.Buffer1, span.Length1),
123 std::bind(&SocketManager::OnPostSend, _socketManager, std::placeholders::_1,
124 std::placeholders::_2, this));
125 }
126 }
127
128 _sendInProgress = true;
129 }
130 catch (const asio::system_error& ex)
131 {
132 lock.unlock();
133
134 spdlog::error(
135 "TcpSocket::AsyncSend: failed to post send for socketId={}: {}", _socketId, ex.what());
136 Close();
137 return false;
138 }
139
140 return true;
141}
