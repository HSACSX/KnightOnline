46bool ItemManagerApp::OnStart()
47{
48 // Attempt to open shared memory queue first.
49 // If it fails (memory not yet available), we'll run the _smqOpenThread to periodically check
50 // until it can finally be opened.
51 if (!AttemptOpenSharedMemory())
52 {
53 spdlog::info("ItemManagerApp::OnStart: shared memory unavailable, waiting for memory to become available");
54 _smqOpenThread->start();
55 }
56 else
57 {
58 OnSharedMemoryOpened();
59 }
60
61 return true;
62}
