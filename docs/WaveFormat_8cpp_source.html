12template <typename T>
13static const uint8_t* find_chunk(
14 const uint8_t* start, const uint8_t* end, const char* chunkID, const T** outStruct)
15{
16 constexpr size_t RequiredStructSize = std::max(sizeof(T), sizeof(RIFF_SubChunk));
17 const uint8_t* ptr = start;
18 while (ptr < (end - RequiredStructSize))
19 {
20 const RIFF_SubChunk* chunk = reinterpret_cast<const RIFF_SubChunk*>(ptr);
21
22 if (chunk->SubChunkID[0] == chunkID[0] && chunk->SubChunkID[1] == chunkID[1]
23 && chunk->SubChunkID[2] == chunkID[2] && chunk->SubChunkID[3] == chunkID[3])
24 {
25 *outStruct = reinterpret_cast<const T*>(ptr);
26 return ptr;
27 }
28
29 ptr += sizeof(RIFF_SubChunk);
30 ptr += chunk->SubChunkSize;
31 }
32
33 return nullptr;
34}

36bool ParseWAV(FileReader& file, ALenum* format, ALsizei* sampleRate, size_t* pcmChunkSize,
37 const uint8_t** pcmDataBuffer, ALsizei* pcmDataSize)
38{
39 const size_t fileSize = static_cast<size_t>(file.Size());
40 if (fileSize < sizeof(RIFF_Header))
41 return false;
42
43 const uint8_t* start = static_cast<const uint8_t*>(file.Memory());
44 const uint8_t* end = start + fileSize;
45 const uint8_t* ptr = start;
46
47 const RIFF_Header* riffHeader = reinterpret_cast<const RIFF_Header*>(ptr);
48 const WAVE_Format* waveFormat = nullptr;
49 const WAVE_Data* waveData = nullptr;
50
51 if (riffHeader->FileTypeID[0] != 'R' || riffHeader->FileTypeID[1] != 'I'
52 || riffHeader->FileTypeID[2] != 'F' || riffHeader->FileTypeID[3] != 'F'
53 || riffHeader->FileFormatID[0] != 'W' || riffHeader->FileFormatID[1] != 'A'
54 || riffHeader->FileFormatID[2] != 'V' || riffHeader->FileFormatID[3] != 'E')
55 return false;
56
57 ptr += sizeof(RIFF_Header);
58
59 ptr = find_chunk<WAVE_Format>(ptr, end, "fmt ", &waveFormat);
60 if (ptr == nullptr)
61 return false;
62
63 // We expect PCM
64 if (waveFormat->AudioFormat != 1)
65 return false;
66
67 ptr += sizeof(RIFF_SubChunk);
68 ptr += waveFormat->Size;
69
70 ptr = find_chunk<WAVE_Data>(ptr, end, "data", &waveData);
71 if (ptr == nullptr)
72 return false;
73
74 ptr += sizeof(RIFF_SubChunk);
75
76 const uint8_t* pcmDataBuffer_ = ptr;
77 const uint32_t pcmDataSize_ = waveData->Size;
78
79 if (pcmDataSize_ > static_cast<uintptr_t>(end - pcmDataBuffer_))
80 return false;
81
82 ALenum format_ = AL_FORMAT_STEREO16;
83
84 if (waveFormat->NumChannels == 2)
85 {
86 if (waveFormat->BitsPerSample == 8)
87 format_ = AL_FORMAT_STEREO8;
88 else if (waveFormat->BitsPerSample == 16)
89 format_ = AL_FORMAT_STEREO16;
90 else
91 return false;
92 }
93 else if (waveFormat->NumChannels == 1)
94 {
95 if (waveFormat->BitsPerSample == 8)
96 format_ = AL_FORMAT_MONO8;
97 else if (waveFormat->BitsPerSample == 16)
98 format_ = AL_FORMAT_MONO16;
99 else
100 return false;
101 }
102 else
103 {
104 return false;
105 }
106
107 constexpr double DurationSec = 0.418; // roughly the same duration as the MP3 chunks
108
109 ALsizei bytesPerFrame = static_cast<ALsizei>(
110 waveFormat->NumChannels * (waveFormat->BitsPerSample / 8));
111 ALsizei chunkFrames = static_cast<ALsizei>(waveFormat->SampleRate * DurationSec);
112 ALsizei pcmChunkSize_ = chunkFrames * bytesPerFrame;
113
114 if (format != nullptr)
115 *format = format_;
116
117 if (sampleRate != nullptr)
118 *sampleRate = static_cast<ALsizei>(waveFormat->SampleRate);
119
120 if (pcmDataBuffer != nullptr)
121 *pcmDataBuffer = pcmDataBuffer_;
122
123 if (pcmDataSize != nullptr)
124 *pcmDataSize = pcmDataSize_;
125
126 if (pcmChunkSize != nullptr)
127 *pcmChunkSize = pcmChunkSize_;
128
129 return true;
130}
