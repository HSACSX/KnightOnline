10bool FileWriter::OpenExisting(const std::filesystem::path& path)
11{
12 // Close any existing file handle and reset write states.
13 Close();
14
15 FILE* fileHandle = nullptr;
16
17 // Open the given file for writing.
18 // This must already exist so that we can simply load it for writing.
19 // NOTE: As we have no 'append' flag, we will leave the offset set to 0.
20 // If the user wishes to append, they must seek to the end of the file themselves.
21 // This matches WinAPI & general C file I/O API behaviour (where the "a" mode isn't used).
22#ifdef _MSC_VER
23 _wfopen_s(&fileHandle, path.native().c_str(), L"rb+");
24#else
25 fileHandle = fopen(path.native().c_str(), "rb+");
26#endif
27 if (fileHandle == nullptr)
28 return false;
29
30 _fileHandle = fileHandle;
31 _path = path;
32 _open = true;
33 _size = 0;
34
35 if (fseek(fileHandle, 0, SEEK_END) == 0)
36 {
37 long fileSize = ftell(fileHandle);
38 if (fileSize >= 0)
39 _size = static_cast<uint64_t>(fileSize);
40
41 // Reset offset to start
42 (void) fseek(fileHandle, 0, SEEK_SET);
43 }
44
45 _sizeOnDisk = _size;
46
47 return true;
48}

83bool FileWriter::Write(const void* buffer, size_t bytesToWrite, size_t* bytesWritten /*= nullptr*/)
84{
85 // Ensure bytesWritten is always initialised, regardless of whether this succeeds or fails.
86 if (bytesWritten != nullptr)
87 *bytesWritten = 0;
88
89 // Fail on invalid buffers.
90 if (buffer == nullptr)
91 return false;
92
93 // Nothing to write, so just skip ahead and consider this successful.
94 if (bytesToWrite == 0)
95 return true;
96
97 assert(_fileHandle != nullptr);
98
99 // Write the given bytes out to file.
100 if (fwrite(buffer, bytesToWrite, 1, static_cast<FILE*>(_fileHandle)) != 1)
101 return false;
102
103 // Advance the current file's offset by the number of bytes written.
104 _offset += bytesToWrite;
105
106 // Update the caller with the number of bytes actually written.
107 if (bytesWritten != nullptr)
108 *bytesWritten = bytesToWrite;
109
110 // Expand the size of the file if we've written past the end of it.
111 if (_offset > _size)
112 {
113 _size = _offset;
114 _sizeOnDisk = _size;
115 }
116
117 return true;
118}

120bool FileWriter::Seek(int64_t offset, int origin)
121{
122 // File not opened, we cannot seek.
123 if (!IsOpen())
124 return false;
125
126 int64_t newOffset = offset;
127
128 switch (origin)
129 {
130 // Explicitly set to the given offset
131 case SEEK_SET:
132 break;
133
134 // Set is relative to the current offset
135 case SEEK_CUR:
136 newOffset += static_cast<int64_t>(_offset);
137 break;
138
139 // Set is relative to the end offset (i.e. the size)
140 case SEEK_END:
141 newOffset += static_cast<int64_t>(_size);
142 break;
143
144 default:
145 return false;
146 }
147
148 if (newOffset < 0)
149 return false;
150
151 _offset = static_cast<uint64_t>(newOffset);
152
153 // Expand the size of the file if we're seeking past it.
154 // NOTE: As this doesn't make a physical change to the file,
155 // we don't yet update _sizeOnDisk.
156 // This will be deferred until either a manually triggered Flush(),
157 // or the file is closed.
158 // At that time, the file on disk will be expanded to match our expected
159 // size.
160 if (_offset > _size)
161 _size = _offset;
162
163 fseek(static_cast<FILE*>(_fileHandle), static_cast<long>(offset), origin);
164
165 return true;
166}

173void FileWriter::FlushImpl()
174{
175 if (_fileHandle == nullptr)
176 return;
177
178 if (fflush(static_cast<FILE*>(_fileHandle)) != 0)
179 return;
180
181 // Our size in memory is larger than the physical size on disk.
182 // This can happen when we seek beyond the file; this doesn't
183 // expand the file right way, it waits until a flush (now).
184 // Write a byte to the last expected offset (_size - 1) and
185 // the operating system will fill in the blanks.
186 if (_size > _sizeOnDisk)
187 {
188 uint8_t dummy = 0;
189 fseek(static_cast<FILE*>(_fileHandle), static_cast<long>(_size - 1), SEEK_SET);
190
191 if (fwrite(&dummy, sizeof(uint8_t), 1, static_cast<FILE*>(_fileHandle)) == 1)
192 _sizeOnDisk = _size;
193
194 fseek(static_cast<FILE*>(_fileHandle), static_cast<long>(_offset), SEEK_SET);
195 }
196}
