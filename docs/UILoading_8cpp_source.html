46bool CUILoading::Load(File& file)
47{
48 if (!CN3UIBase::Load(file))
49 return false;
50
51 N3_VERIFY_UI_COMPONENT(m_pText_Version, GetChildByID<CN3UIString>("Text_Version"));
52 if (m_pText_Version != nullptr)
53 {
54 std::string version = fmt::format("Ver. {:.3f}", CURRENT_VERSION / 1000.0f);
55 m_pText_Version->SetString(version);
56 }
57
58 N3_VERIFY_UI_COMPONENT(m_pText_Info, GetChildByID<CN3UIString>("Text_Info"));
59 N3_VERIFY_UI_COMPONENT(m_pProgress_Loading, GetChildByID<CN3UIProgress>("Progress_Loading"));
60
61 SetPosCenter(); // 가운데로 맞추기..
62 m_pText_Version->SetPos(10, 10); // Version 은 맨위에 표시..
63
64 if (m_pProgress_Loading != nullptr)
65 m_pProgress_Loading->SetRange(0, 100);
66
67 return true;
68}
