54void TelnetClientThread::thread_loop()
55{
56 try
57 {
58 WriteLine("User:");
59 std::string accountId;
60 do
61 {
62 accountId = ReadLine();
63 }
64 while (accountId.empty());
65
66 WriteLine("Password:");
67 std::string password;
68 do
69 {
70 password = ReadLine();
71 }
72 while (password.empty());
73
74 if (!Authenticate(accountId, password))
75 {
76 Disconnect();
77 return;
78 }
79
80 _accountId = std::move(accountId);
81
82 spdlog::info("TelnetClientThread::thread_loop: {} authenticated [remoteIp={}]", _accountId,
83 _remoteIp);
84 WriteLine("Authenticated. Accepting commands. \"quit\" to close connection.");
85
86 while (CanTick())
87 {
88 if (!_clientSocket.is_open())
89 {
90 spdlog::info("TelnetClientThread::thread_loop: socket longer open [accountId={} "
91 "remoteIp={}]",
92 _accountId, _remoteIp);
93
94 shutdown(false);
95 continue;
96 }
97
98 std::string input = ReadLine();
99 if (input.empty())
100 continue;
101
102 spdlog::info("TelnetClientThread::thread_loop: input: {} [accountId={} remoteIp={}]",
103 input, _accountId, _remoteIp);
104
105 if (input == "quit")
106 {
107 spdlog::info("TelnetClientThread::thread_loop: socket closed by request "
108 "[accountId={} remoteIp={}]",
109 accountId, _remoteIp);
110
111 Disconnect();
112 break;
113 }
114
115 if (input == "healthcheck")
116 {
117 HealthCheck();
118 continue;
119 }
120
121 AppThread* appThread = AppThread::instance();
122 if (appThread == nullptr)
123 {
124 Disconnect();
125 continue;
126 }
127
128 if (appThread->HandleCommand(input))
129 WriteLine(fmt::format("Input command accepted: {}", input));
130 else
131 WriteLine(fmt::format("Input command failed: {}", input));
132 }
133 }
134 catch (const std::exception& e)
135 {
136 spdlog::error("TelnetClientThread::thread_loop: {} [accountId={} remoteIp={}]", e.what(),
137 _accountId, _remoteIp);
138 }
139}

146std::string TelnetClientThread::ReadLine()
147{
148 constexpr char Delimiter = '\n';
149
150 std::string line;
151 asio::streambuf buffer;
152 std::promise<size_t> promise;
153
154 asio::async_read_until(_clientSocket, buffer, Delimiter,
155 [&](const asio::error_code& ec, std::size_t bytesRead)
156 {
157 if (ec)
158 promise.set_exception(std::make_exception_ptr(asio::system_error(ec)));
159 else
160 promise.set_value(bytesRead);
161 });
162
163 auto future = promise.get_future();
164 size_t bytesRead = future.get();
165 if (bytesRead == 0)
166 throw std::runtime_error("socket disconnected");
167
168 std::istream is(&buffer);
169 std::getline(is, line, Delimiter);
170
171 // strip any trailing \r
172 if (!line.empty() && line.back() == '\r')
173 line.pop_back();
174
175 return line;
176}

178bool TelnetClientThread::Authenticate(const std::string& accountId, const std::string& password)
179{
180 full_model::TbUser user {};
181
182 try
183 {
184 db::SqlBuilder<full_model::TbUser> sql;
185 sql.IsWherePK = true;
186
187 db::ModelRecordSet<full_model::TbUser> recordSet;
188
189 auto stmt = recordSet.prepare(sql);
190 if (stmt == nullptr)
191 {
192 throw db::ApplicationError(
193 "TelnetClientThread::Authenticate: statement could not be allocated");
194 }
195
196 stmt->bind(0, accountId.c_str());
197 recordSet.execute();
198
199 if (!recordSet.next())
200 {
201 spdlog::warn(
202 "TelnetClientThread::Authenticate: failed authentication attempt for user {} - "
203 "missing account [remoteIp={}]",
204 accountId, _remoteIp);
205 return false;
206 }
207
208 recordSet.get_ref(user);
209 }
210 catch (const nanodbc::database_error& dbErr)
211 {
212 db::utils::LogDatabaseError(dbErr, "TelnetClientThread::Authenticate()");
213 return false;
214 }
215
216 if (user.Password != password)
217 {
218 spdlog::warn(
219 "TelnetClientThread::Authenticate: failed authentication attempt for user {} - "
220 "incorrect password [remoteIp={}]",
221 accountId, _remoteIp);
222 return false;
223 }
224
225 if (user.Authority != AUTHORITY_MANAGER)
226 {
227 spdlog::warn(
228 "TelnetClientThread::Authenticate: failed authentication attempt for user {} - "
229 "not AUTHORITY_MANAGER [remoteIp={}]",
230 accountId, _remoteIp);
231 return false;
232 }
233
234 return true;
235}
