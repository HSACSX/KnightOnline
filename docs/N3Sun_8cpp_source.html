54void CN3Sun::Render(__Matrix44& matView, __Matrix44& matProj)
55{
56 // 해가 그려지는 화면 좌표 계산하기 (2D로 그리는 이유는 왜곡되어서 보이는것을 막기 위해)
57 __Matrix44 matWorld;
58 matWorld.RotationZ(m_fCurRadian);
59 __Matrix44 matFinal;
60 matFinal = matWorld * matView;
61 matFinal *= matProj;
62
63 __Vector3 vSun;
64 vSun.Set(5, 0, 0);
65 __Vector4 vOut;
66 vOut.Transform(vSun, matFinal);
67
68 int Width = s_CameraData.vp.Width;
69 int Height = s_CameraData.vp.Height;
70 int X = s_CameraData.vp.X;
71 int Y = s_CameraData.vp.Y;
72
73 float rhw = 1.0f / vOut.w;
74 vSun.z = vOut.z * rhw;
75 if (vSun.z < 0.0f || vSun.z > 1.0f)
76 return;
77 // Mapping Screen Coordinate.
78 vSun.x = (float) X + int((vOut.x * rhw + 1) * Width * 0.5f);
79 vSun.y = (float) Y + int((-vOut.y * rhw + 1) * Height * 0.5f);
80
81 // back up render state
82 DWORD dwSrcBlend, dwDestBlend;
83 s_lpD3DDev->GetRenderState(D3DRS_SRCBLEND, &dwSrcBlend);
84 s_lpD3DDev->GetRenderState(D3DRS_DESTBLEND, &dwDestBlend);
85
86 // set render state
87 if (D3DBLEND_ONE != dwSrcBlend)
88 s_lpD3DDev->SetRenderState(D3DRS_SRCBLEND, D3DBLEND_ONE);
89 if (D3DBLEND_ONE != dwDestBlend)
90 s_lpD3DDev->SetRenderState(D3DRS_DESTBLEND, D3DBLEND_ONE);
91 s_lpD3DDev->SetTextureStageState(0, D3DTSS_COLOROP, D3DTOP_MODULATE);
92 s_lpD3DDev->SetTextureStageState(0, D3DTSS_COLORARG1, D3DTA_TEXTURE);
93 s_lpD3DDev->SetTextureStageState(0, D3DTSS_COLORARG2, D3DTA_DIFFUSE);
94 s_lpD3DDev->SetFVF(FVF_TRANSFORMED);
95
96 RECT rcSun[NUM_SUNPART];
97 RECT rcScreen;
98 SetRect(&rcScreen, X, Y, X + Width, Y + Width);
99 for (int i = 0; i < NUM_SUNPART; ++i)
100 {
101 __SunPart* pSP = &(m_Parts[i]);
102
103 float iWTmp = iWTmp = (Width * pSP->Delta.GetCurDelta()) / 2;
104 SetRect(&(rcSun[i]), (int) (vSun.x - iWTmp), (int) (vSun.y - iWTmp), (int) (vSun.x + iWTmp),
105 (int) (vSun.y + iWTmp));
106 // clipping with screen.
107 if (rcSun[i].right < rcScreen.left || rcSun[i].bottom < rcScreen.top
108 || rcSun[i].left > rcScreen.right || rcSun[i].top > rcScreen.bottom)
109 continue; // 화면 밖에 그려진다.
110
111 // 2D로 그리기
112 pSP->pVertices[0].x = (float) rcSun[i].left;
113 pSP->pVertices[0].y = (float) rcSun[i].top;
114 pSP->pVertices[1].x = (float) rcSun[i].right;
115 pSP->pVertices[1].y = (float) rcSun[i].top;
116 pSP->pVertices[2].x = (float) rcSun[i].right;
117 pSP->pVertices[2].y = (float) rcSun[i].bottom;
118 pSP->pVertices[3].x = (float) rcSun[i].left;
119 pSP->pVertices[3].y = (float) rcSun[i].bottom;
120
121 if (pSP->pTex)
122 s_lpD3DDev->SetTexture(0, pSP->pTex->Get());
123 else
124 s_lpD3DDev->SetTexture(0, nullptr);
125 s_lpD3DDev->DrawPrimitiveUP(
126 D3DPT_TRIANGLEFAN, 2, pSP->pVertices, sizeof(__VertexTransformed));
127 }
128
129 // restore render state
130 if (D3DBLEND_ONE != dwSrcBlend)
131 s_lpD3DDev->SetRenderState(D3DRS_SRCBLEND, dwSrcBlend);
132 if (D3DBLEND_ONE != dwDestBlend)
133 s_lpD3DDev->SetRenderState(D3DRS_DESTBLEND, dwDestBlend);
134}

149void CN3Sun::Init(const std::string* pszFNs)
150{
151 Release();
152
153 const float fZ = 0.9f;
154 const float rhw = 1.0f;
155 const D3DCOLOR color = 0xffffffff;
156 for (int i = 0; i < NUM_SUNPART; ++i)
157 {
158 m_Parts[i].pTex = s_MngTex.Get(pszFNs[i]); // load texture
159
160 m_Parts[i].pVertices[0].Set(0, 0, fZ, rhw, color, 0.0f, 0.0f);
161 m_Parts[i].pVertices[1].Set(0, 0, fZ, rhw, color, 1.0f, 0.0f);
162 m_Parts[i].pVertices[2].Set(0, 0, fZ, rhw, color, 1.0f, 1.0f);
163 m_Parts[i].pVertices[3].Set(0, 0, fZ, rhw, color, 0.0f, 1.0f);
164 }
165
166 m_Parts[SUNPART_SUN].Delta.ChangeDelta(0.1f); // ViewPort 에서 상대적인 크기
167 m_Parts[SUNPART_GLOW].Delta.ChangeDelta(0.25f);
168 m_Parts[SUNPART_FLARE].Delta.ChangeDelta(0.13f);
169}
