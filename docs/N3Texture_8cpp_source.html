14static inline uint32_t GetTextureSize(const D3DSURFACE_DESC& sd)
15{
16 uint32_t nTexSize = sd.Width * sd.Height;
17 if (sd.Format == D3DFMT_DXT1)
18 return nTexSize / 2;
19
20 return nTexSize & ~0xF;
21}

42void CN3Texture::Release()
43{
44 if (32 == m_Header.nWidth && 32 == m_Header.nHeight)
45 s_ResrcInfo.nTexture_Loaded_32X32--;
46 else if (64 == m_Header.nWidth && 64 == m_Header.nHeight)
47 s_ResrcInfo.nTexture_Loaded_64X64--;
48 else if (128 == m_Header.nWidth && 128 == m_Header.nHeight)
49 s_ResrcInfo.nTexture_Loaded_128X128--;
50 else if (256 == m_Header.nWidth && 256 == m_Header.nHeight)
51 s_ResrcInfo.nTexture_Loaded_256X256--;
52 else if (512 == m_Header.nWidth && 512 == m_Header.nHeight)
53 s_ResrcInfo.nTexture_Loaded_512X512--;
54 else if (512 < m_Header.nWidth && 512 < m_Header.nHeight)
55 s_ResrcInfo.nTexture_Loaded_Huge--;
56 else
57 s_ResrcInfo.nTexture_Loaded_OtherSize--;
58
59 memset(&m_Header, 0, sizeof(m_Header));
60 if (m_lpTexture && m_lpTexture->Release() == 0)
61 m_lpTexture = nullptr;
62 m_iLOD = 0;
63
64 CN3BaseFileAccess::Release();
65}

67bool CN3Texture::Create(int nWidth, int nHeight, D3DFORMAT Format, BOOL bGenerateMipMap)
68{
69 if (nWidth != nHeight)
70
71 if (nWidth <= 1 || nHeight <= 1 || D3DFMT_UNKNOWN == Format)
72 return false;
73 if (m_lpTexture != nullptr)
74 this->Release();
75
76 if (s_dwTextureCaps & TEX_CAPS_POW2) // 2 의 승수만 된다면..
77 {
78 int nW = 0, nH = 0;
79 for (nW = 1; nW <= nWidth; nW *= 2)
80 ;
81 nW /= 2;
82 for (nH = 1; nH <= nHeight; nH *= 2)
83 ;
84 nH /= 2;
85
86 nWidth = nW;
87 nHeight = nH;
88 }
89
90 if ((s_dwTextureCaps & TEX_CAPS_SQUAREONLY) && nWidth != nHeight) // 정사각형 텍스처만 되면..
91 {
92 if (nWidth > nHeight)
93 nWidth = nHeight;
94 else
95 nHeight = nWidth;
96 }
97
98 // 비디오 카드가 256 이상의 텍스처를 지원 하지 못하면..
99 if (nWidth > 256 && CN3Base::s_DevCaps.MaxTextureWidth <= 256)
100 nWidth = CN3Base::s_DevCaps.MaxTextureWidth;
101 if (nHeight > 256 && CN3Base::s_DevCaps.MaxTextureHeight <= 256)
102 nHeight = CN3Base::s_DevCaps.MaxTextureHeight;
103
104 // 헤더 세팅..
105 memset(&m_Header, 0, sizeof(m_Header));
106
107 // MipMap 단계 결정..
108 // 4 X 4 픽셀까지만 MipMap 을 만든다..
109 int nMMC = 1;
110 if (bGenerateMipMap)
111 {
112 nMMC = 0;
113 for (int nW = nWidth, nH = nHeight; nW >= 4 && nH >= 4; nW /= 2, nH /= 2)
114 nMMC++;
115 }
116
117 HRESULT rval = s_lpD3DDev->CreateTexture(
118 nWidth, nHeight, nMMC, 0, Format, D3DPOOL_MANAGED, &m_lpTexture, nullptr);
119
120#ifdef _N3GAME
121 if (rval == D3DERR_INVALIDCALL)
122 {
123 CLogWriter::Write("N3Texture: createtexture err D3DERR_INVALIDCALL({})", m_szFileName);
124 return false;
125 }
126 if (rval == D3DERR_OUTOFVIDEOMEMORY)
127 {
128 CLogWriter::Write("N3Texture: createtexture err D3DERR_OUTOFVIDEOMEMORY({})", m_szFileName);
129 return false;
130 }
131 if (rval == E_OUTOFMEMORY)
132 {
133 CLogWriter::Write("N3Texture: createtexture err E_OUTOFMEMORY({})", m_szFileName);
134 return false;
135 }
136#endif
137 if (nullptr == m_lpTexture)
138 {
139 __ASSERT(m_lpTexture, "Texture pointer is NULL!");
140 return false;
141 }
142
143 m_Header.nWidth = nWidth;
144 m_Header.nHeight = nHeight;
145 m_Header.Format = Format;
146 m_Header.bMipMap = bGenerateMipMap;
147
148 if (32 == m_Header.nWidth && 32 == m_Header.nHeight)
149 s_ResrcInfo.nTexture_Loaded_32X32++;
150 else if (64 == m_Header.nWidth && 64 == m_Header.nHeight)
151 s_ResrcInfo.nTexture_Loaded_64X64++;
152 else if (128 == m_Header.nWidth && 128 == m_Header.nHeight)
153 s_ResrcInfo.nTexture_Loaded_128X128++;
154 else if (256 == m_Header.nWidth && 256 == m_Header.nHeight)
155 s_ResrcInfo.nTexture_Loaded_256X256++;
156 else if (512 == m_Header.nWidth && 512 == m_Header.nHeight)
157 s_ResrcInfo.nTexture_Loaded_512X512++;
158 else if (512 < m_Header.nWidth && 512 < m_Header.nHeight)
159 s_ResrcInfo.nTexture_Loaded_Huge++;
160 else
161 s_ResrcInfo.nTexture_Loaded_OtherSize++;
162
163 return true;
164}

187bool CN3Texture::LoadFromFile(const std::string& szFileName)
188{
189 if (m_lpTexture != nullptr)
190 this->Release();
191
192 this->FileNameSet(szFileName); // 파일 이름을 복사하고..
193 std::string szFullPath;
194 if (-1 != m_szFileName.find(':') || -1 != m_szFileName.find("\\\\")
195 || -1 != m_szFileName.find("//")) // 문자열에 ':', '\\', '//' 이 들어 있으면 전체 경로이다..
196 {
197 szFullPath = m_szFileName;
198 }
199 else
200 {
201 if ('\0' != s_szPath[0])
202 szFullPath = s_szPath;
203 szFullPath += m_szFileName;
204 }
205
206 size_t nFNL = szFullPath.size();
207 if (nFNL >= 3 && lstrcmpi(&szFullPath[nFNL - 3], "DXT") == 0)
208 {
209 FileReader file;
210 if (!file.OpenExisting(szFullPath))
211 {
212#ifdef _N3GAME
213 CLogWriter::Write("Can't open texture file({})", szFullPath);
214#endif
215 return false;
216 }
217
218 try
219 {
220 LoadSupportedVersions(file);
221 }
222 catch (const std::exception& ex)
223 {
224 std::string szErr = szFullPath + " - Failed to read file (" + std::string(ex.what())
225 + ")";
226#ifdef _N3GAME
227 CLogWriter::Write(szErr);
228#endif
229#ifdef _N3TOOL
230 MessageBox(s_hWndBase, szErr.c_str(), "Failed to read file", MB_OK);
231#endif
232 }
233 }
234 else
235 {
236 D3DXIMAGE_INFO ImgInfo;
237 HRESULT rval = D3DXCreateTextureFromFileEx(s_lpD3DDev, szFullPath.c_str(), D3DX_DEFAULT,
238 D3DX_DEFAULT, D3DX_DEFAULT, 0, D3DFMT_UNKNOWN, D3DPOOL_MANAGED,
239 D3DX_FILTER_TRIANGLE | D3DX_FILTER_MIRROR, D3DX_FILTER_TRIANGLE | D3DX_FILTER_MIRROR, 0,
240 &ImgInfo, nullptr, &m_lpTexture);
241 if (rval == D3D_OK)
242 {
243 D3DSURFACE_DESC sd;
244 m_lpTexture->GetLevelDesc(0, &sd);
245
246 m_Header.nWidth = sd.Width;
247 m_Header.nHeight = sd.Height;
248 m_Header.Format = sd.Format;
249 }
250 else
251 {
252#ifdef _N3GAME
253 CLogWriter::Write("N3Texture - Failed to load texture({})", szFullPath);
254#endif
255 }
256
257 if (32 == m_Header.nWidth && 32 == m_Header.nHeight)
258 s_ResrcInfo.nTexture_Loaded_32X32++;
259 else if (64 == m_Header.nWidth && 64 == m_Header.nHeight)
260 s_ResrcInfo.nTexture_Loaded_64X64++;
261 else if (128 == m_Header.nWidth && 128 == m_Header.nHeight)
262 s_ResrcInfo.nTexture_Loaded_128X128++;
263 else if (256 == m_Header.nWidth && 256 == m_Header.nHeight)
264 s_ResrcInfo.nTexture_Loaded_256X256++;
265 else if (512 == m_Header.nWidth && 512 == m_Header.nHeight)
266 s_ResrcInfo.nTexture_Loaded_512X512++;
267 else if (512 < m_Header.nWidth && 512 < m_Header.nHeight)
268 s_ResrcInfo.nTexture_Loaded_Huge++;
269 else
270 s_ResrcInfo.nTexture_Loaded_OtherSize++;
271 }
272
273 if (m_lpTexture == nullptr)
274 {
275 Release();
276 return false;
277 }
278
279 return true;
280}

282bool CN3Texture::Load(File& file)
283{
284 if (!CN3BaseFileAccess::Load(file))
285 return false;
286
287 CWinCrypt crypt;
288
289 __DXT_HEADER HeaderOrg {}; // 헤더를 저장해 놓고..
290 file.Read(&HeaderOrg, sizeof(HeaderOrg)); // 헤더를 읽는다..
291 if ('N' != HeaderOrg.szID[0] || 'T' != HeaderOrg.szID[1] || 'F' != HeaderOrg.szID[2])
292 return false;
293
294 // "NTF"3 - Noah Texture File Ver. 3.0
295 if (HeaderOrg.szID[3] < 3)
296 {
297#ifdef _N3GAME
298 CLogWriter::Write("N3Texture Warning - Old format DXT file ({})", m_szFileName);
299#endif
300 }
301
302 if (HeaderOrg.szID[3] == 7 && !crypt.Load())
303 return false;
304
305 // DXT Format 을 읽어야 하는데 지원이 되는지 안되는지 보고 지원안되면 대체 포맷을 정한다.
306 bool bDXTSupport = FALSE;
307 D3DFORMAT fmtNew = HeaderOrg.Format;
308 if (D3DFMT_DXT1 == HeaderOrg.Format)
309 {
310 if (s_dwTextureCaps & TEX_CAPS_DXT1)
311 bDXTSupport = true;
312 else
313 fmtNew = D3DFMT_A1R5G5B5;
314 }
315 else if (D3DFMT_DXT2 == HeaderOrg.Format)
316 {
317 if (s_dwTextureCaps & TEX_CAPS_DXT2)
318 bDXTSupport = true;
319 else
320 fmtNew = D3DFMT_A4R4G4B4;
321 }
322 else if (D3DFMT_DXT3 == HeaderOrg.Format)
323 {
324 if (s_dwTextureCaps & TEX_CAPS_DXT3)
325 bDXTSupport = true;
326 else
327 fmtNew = D3DFMT_A4R4G4B4;
328 }
329 else if (D3DFMT_DXT4 == HeaderOrg.Format)
330 {
331 if (s_dwTextureCaps & TEX_CAPS_DXT4)
332 bDXTSupport = true;
333 else
334 fmtNew = D3DFMT_A4R4G4B4;
335 }
336 else if (D3DFMT_DXT5 == HeaderOrg.Format)
337 {
338 if (s_dwTextureCaps & TEX_CAPS_DXT5)
339 bDXTSupport = true;
340 else
341 fmtNew = D3DFMT_A4R4G4B4;
342 }
343
344 int iWCreate = HeaderOrg.nWidth, iHCreate = HeaderOrg.nHeight;
345 if (fmtNew != HeaderOrg.Format)
346 {
347 iWCreate /= 2;
348 iHCreate /= 2;
349 } // DXT 지원이 안되면 너비 높이를 줄인다.
350 if (m_iLOD > 0 && m_iLOD <= 2 && HeaderOrg.nWidth >= 16
351 && HeaderOrg.nHeight >= 16) // LOD 만큼 작게 생성..
352 {
353 for (int i = 0; i < m_iLOD; i++)
354 {
355 iWCreate /= 2;
356 iHCreate /= 2;
357 }
358 }
359 else
360 m_iLOD = 0; // LOD 적용이 아니면..
361
362 int iLODPrev = m_iLOD;
363 this->Create(iWCreate, iHCreate, fmtNew, HeaderOrg.bMipMap); // 서피스 만들고..
364 m_iLOD = iLODPrev;
365
366 if (m_lpTexture == nullptr)
367 {
368#ifdef _N3GAME
369 CLogWriter::Write("N3Texture error - Can't create texture ({})", m_szFileName);
370#endif
371 return false;
372 }
373
374 D3DSURFACE_DESC sd;
375 D3DLOCKED_RECT LR;
376 int iMMC = m_lpTexture->GetLevelCount(); // 생성한 MipMap 수
377
378 // 압축 포맷이면..
379 if (D3DFMT_DXT1 == HeaderOrg.Format || D3DFMT_DXT2 == HeaderOrg.Format
380 || D3DFMT_DXT3 == HeaderOrg.Format || D3DFMT_DXT4 == HeaderOrg.Format
381 || D3DFMT_DXT5 == HeaderOrg.Format)
382 {
383 if (TRUE == bDXTSupport) // 압축 텍스처 지원이면..
384 {
385 if (iMMC > 1)
386 {
387 if (m_iLOD > 0) // LOD 만큼 건너뛰기...
388 {
389 int iWTmp = HeaderOrg.nWidth, iHTmp = HeaderOrg.nHeight, iSkipSize = 0;
390 for (int i = 0; i < m_iLOD; i++, iWTmp /= 2, iHTmp /= 2)
391 {
392 // DXT1 형식은 16비트 포맷에 비해 1/4 로 압축..
393 if (D3DFMT_DXT1 == HeaderOrg.Format)
394 iSkipSize += iWTmp * iHTmp / 2;
395 // DXT2 ~ DXT5 형식은 16비트 포맷에 비해 1/2 로 압축..
396 else
397 iSkipSize += iWTmp * iHTmp;
398 }
399
400 file.Seek(iSkipSize, SEEK_CUR); // 건너뛰고.
401 }
402
403 for (int i = 0; i < iMMC; i++)
404 {
405 m_lpTexture->GetLevelDesc(i, &sd);
406
407 uint32_t nTexSize = GetTextureSize(sd);
408
409 m_lpTexture->LockRect(i, &LR, nullptr, 0);
410
411 // 일렬로 된 데이터를 쓰고..
412 crypt.ReadFile(file, LR.pBits, nTexSize);
413
414 m_lpTexture->UnlockRect(i);
415 }
416
417 // 텍스처 압축안되는 비디오 카드를 위한 여분의 데이터 건너뛰기..
418 int iWTmp = HeaderOrg.nWidth / 2, iHTmp = HeaderOrg.nHeight / 2;
419 for (; iWTmp >= 4 && iHTmp >= 4; iWTmp /= 2,
420 iHTmp /=
421 2) // 한픽셀에 두바이트가 들어가는 A1R5G5B5 혹은 A4R4G4B4 포맷으로 되어 있다..
422 file.Seek(iWTmp * iHTmp * 2, SEEK_CUR); // 건너뛰고.
423 }
424 else // pair of if(iMMC > 1)
425 {
426 m_lpTexture->GetLevelDesc(0, &sd);
427
428 uint32_t nTexSize = GetTextureSize(sd);
429
430 m_lpTexture->LockRect(0, &LR, nullptr, 0);
431
432 // 일렬로 된 데이터를 쓰고..
433 crypt.ReadFile(file, LR.pBits, nTexSize);
434
435 m_lpTexture->UnlockRect(0);
436
437 // 텍스처 압축안되는 비디오 카드를 위한 여분의 데이터 건너뛰기..
438 file.Seek(HeaderOrg.nWidth * HeaderOrg.nHeight / 4, SEEK_CUR); // 건너뛰고.
439 if (HeaderOrg.nWidth >= 1024)
440 file.Seek(256 * 256 * 2,
441 SEEK_CUR); // 사이즈가 512 보다 클경우 부두용 데이터 건너뛰기..
442 }
443 }
444 else // DXT 지원이 안되면..
445 {
446 if (iMMC > 1) // LOD 만큼 건너뛰기...
447 {
448 // 압축 데이터 건너뛰기..
449 int iWTmp = HeaderOrg.nWidth, iHTmp = HeaderOrg.nHeight, iSkipSize = 0;
450 for (; iWTmp >= 4 && iHTmp >= 4; iWTmp /= 2, iHTmp /= 2)
451 {
452 // DXT1 형식은 16비트 포맷에 비해 1/4 로 압축..
453 if (D3DFMT_DXT1 == HeaderOrg.Format)
454 iSkipSize += iWTmp * iHTmp / 2;
455 // DXT2 ~ DXT5 형식은 16비트 포맷에 비해 1/2 로 압축..
456 else
457 iSkipSize += iWTmp * iHTmp;
458 }
459
460 file.Seek(iSkipSize, SEEK_CUR); // 건너뛰고.
461
462 // LOD 만큼 건너뛰기..
463 iWTmp = HeaderOrg.nWidth / 2;
464 iHTmp = HeaderOrg.nHeight / 2;
465 iSkipSize = 0;
466 if (m_iLOD > 0)
467 {
468 // 피치에 너비를 나눈게 픽셀의 크기라 생각한다...
469 for (int i = 0; i < m_iLOD; i++, iWTmp /= 2, iHTmp /= 2)
470 iSkipSize += iWTmp * iHTmp * 2;
471 }
472
473 // 비디오 카드 지원 텍스처 크기가 작을경우 건너뛰기..
474 for (; iWTmp > static_cast<int>(s_DevCaps.MaxTextureWidth)
475 || iHTmp > static_cast<int>(s_DevCaps.MaxTextureHeight);
476 iWTmp /= 2, iHTmp /= 2)
477 iSkipSize += iWTmp * iHTmp * 2;
478 if (iSkipSize != 0)
479 file.Seek(iSkipSize, SEEK_CUR); // 건너뛰고.
480
481 for (int i = 0; i < iMMC; i++)
482 {
483 m_lpTexture->GetLevelDesc(i, &sd);
484 m_lpTexture->LockRect(i, &LR, nullptr, 0);
485 int nH = sd.Height;
486 for (int y = 0; y < nH; y++)
487 {
488 uint8_t* pBits = (uint8_t*) LR.pBits + y * LR.Pitch;
489 if (!crypt.ReadFile(file, pBits, 2 * sd.Width))
490 break;
491 }
492 m_lpTexture->UnlockRect(i);
493 }
494 }
495 else // pair of if(iMMC > 1)
496 {
497 // 압축 데이터 건너뛰기..
498 int iWTmp = HeaderOrg.nWidth, iHTmp = HeaderOrg.nHeight, iSkipSize = 0;
499 // DXT1 형식은 16비트 포맷에 비해 1/4 로 압축..
500 if (D3DFMT_DXT1 == HeaderOrg.Format)
501 iSkipSize = iWTmp * iHTmp / 2;
502 // DXT2 ~ DXT5 형식은 16비트 포맷에 비해 1/2 로 압축..
503 else
504 iSkipSize = iWTmp * iHTmp;
505
506 if (iSkipSize != 0)
507 file.Seek(iSkipSize, SEEK_CUR); // 건너뛰고.
508 }
509 }
510 }
511 else
512 {
513 int iPixelSize = 0;
514 if (fmtNew == D3DFMT_A1R5G5B5 || fmtNew == D3DFMT_A4R4G4B4)
515 iPixelSize = 2;
516 else if (fmtNew == D3DFMT_R8G8B8)
517 iPixelSize = 3;
518 else if (fmtNew == D3DFMT_A8R8G8B8 || fmtNew == D3DFMT_X8R8G8B8)
519 iPixelSize = 4;
520 else
521 {
522 __ASSERT(0, "Not supported texture format");
523 }
524
525 if (iMMC > 1)
526 {
527 if (m_iLOD > 0) // LOD 만큼 건너뛰기...
528 {
529 int iWTmp = HeaderOrg.nWidth, iHTmp = HeaderOrg.nHeight, iSkipSize = 0;
530 for (int i = 0; i < m_iLOD; i++, iWTmp /= 2, iHTmp /= 2)
531 iSkipSize += iWTmp * iHTmp
532 * iPixelSize; // 피치에 너비를 나눈게 픽셀의 크기라 생각한다...
533 file.Seek(iSkipSize, SEEK_CUR); // 건너뛰고.
534 }
535
536 // 비디오 카드 지원 텍스처 크기가 작을경우 건너뛰기..
537 int iWTmp = HeaderOrg.nWidth, iHTmp = HeaderOrg.nHeight, iSkipSize = 0;
538 for (; iWTmp > static_cast<int>(s_DevCaps.MaxTextureWidth)
539 || iHTmp > static_cast<int>(s_DevCaps.MaxTextureHeight);
540 iWTmp /= 2, iHTmp /= 2)
541 iSkipSize += iWTmp * iHTmp * iPixelSize;
542 if (iSkipSize != 0)
543 file.Seek(iSkipSize, SEEK_CUR); // 건너뛰고.
544
545 // 데이터 읽기..
546 for (int i = 0; i < iMMC; i++)
547 {
548 m_lpTexture->GetLevelDesc(i, &sd);
549 m_lpTexture->LockRect(i, &LR, nullptr, 0);
550 for (int y = 0; y < (int) sd.Height; y++)
551 {
552 uint8_t* pBits = (uint8_t*) LR.pBits + y * LR.Pitch;
553 if (!crypt.ReadFile(file, pBits, iPixelSize * sd.Width))
554 break;
555 }
556 m_lpTexture->UnlockRect(i);
557 }
558 }
559 else // pair of if(iMMC > 1)
560 {
561 // 비디오 카드 지원 텍스처 크기가 작을경우 건너뛰기..
562 if (HeaderOrg.nWidth >= 512 && m_Header.nWidth <= 256)
563 file.Seek(HeaderOrg.nWidth * HeaderOrg.nHeight * iPixelSize, SEEK_CUR); // 건너뛰고.
564
565 m_lpTexture->LockRect(0, &LR, nullptr, 0);
566 D3DSURFACE_DESC sd;
567 m_lpTexture->GetLevelDesc(0, &sd);
568 for (int y = 0; y < (int) sd.Height; y++)
569 {
570 uint8_t* pBits = (uint8_t*) LR.pBits + y * LR.Pitch;
571 if (!crypt.ReadFile(file, pBits, iPixelSize * sd.Width))
572 break;
573 }
574 m_lpTexture->UnlockRect(0);
575
576 if (m_Header.nWidth >= 512 && m_Header.nHeight >= 512)
577 file.Seek(

624 iHTmp /=

626 iSkipSize += iWTmp * iHTmp * 2; // 건너뛰고.
627 }

637 iSkipSize += HeaderOrg.nWidth * HeaderOrg.nHeight * 2;
638 if (HeaderOrg.nWidth >= 1024)

640 }
641
642 file.Seek(iSkipSize, SEEK_CUR); // 건너뛰고.
643 }
644 else
645 {

647 if (HeaderOrg.Format == D3DFMT_A1R5G5B5 || HeaderOrg.Format == D3DFMT_A4R4G4B4)
648 iPixelSize = 2;

650 iPixelSize = 3;
651 else if (HeaderOrg.Format == D3DFMT_A8R8G8B8 || HeaderOrg.Format == D3DFMT_X8R8G8B8)
652 iPixelSize = 4;
653 else
654 {
655 __ASSERT(0, "Not supported texture format");
656 }
657
658 int iWTmp = HeaderOrg.nWidth, iHTmp = HeaderOrg.nHeight, iSkipSize = 0;
659 if (HeaderOrg.bMipMap)
660 {
661 for (; iWTmp >= 4 && iHTmp >= 4; iWTmp /= 2, iHTmp /= 2)
662 iSkipSize += iWTmp * iHTmp * iPixelSize;
663 }
664 else

666 iSkipSize += iWTmp * iHTmp * iPixelSize;
667 if (HeaderOrg.nWidth >= 512)
668 iSkipSize += 256 * 256 * 2; // 사이즈가 512 보다 클경우 부두용 데이터 건너뛰기..
669 }
670
671 file.Seek(iSkipSize, SEEK_CUR); // 건너뛰고.
672 }
