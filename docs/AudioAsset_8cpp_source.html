28bool BufferedAudioAsset::LoadFromFile(const std::string& filename)
29{
30 // Expect ".wav", 4 characters long.
31 constexpr size_t ExtensionLength = 4;
32
33 if (filename.length() < ExtensionLength)
34 return false;
35
36 const char* extension = filename.data() + filename.length() - ExtensionLength;
37 if (strnicmp(extension, ".wav", ExtensionLength) != 0)
38 return false;
39
40 FileReader file;
41 if (!file.OpenExisting(filename))
42 return false;
43
44 ALenum alFormat;
45 ALsizei sampleRate, pcmDataSize;
46 size_t pcmChunkSize;
47 const uint8_t* pcmDataBuffer = nullptr;
48 if (!ParseWAV(file, &alFormat, &sampleRate, &pcmChunkSize, &pcmDataBuffer, &pcmDataSize))
49 return false;
50
51 alGenBuffers(1, &BufferId);
52 if (AL_CHECK_ERROR())
53 return false;
54
55 alBufferData(BufferId, alFormat, pcmDataBuffer, pcmDataSize, sampleRate);
56 if (AL_CHECK_ERROR())
57 return false;
58
59 Filename = filename;
60 AlFormat = alFormat;
61 SampleRate = static_cast<int32_t>(sampleRate);
62
63 return true;
64}
