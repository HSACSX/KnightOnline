116std::shared_ptr<Connection> ConnectionManager::CreateConnectionImpl(
117 modelUtil::DbType dbType, long timeout) noexcept(false)
118{
119 auto config = GetDatasourceConfig(dbType);
120 if (config == nullptr)
121 throw DatasourceConfigNotFoundException(
122 "Invalid database type: " + modelUtil::DbTypeString(dbType));
123
124 // If not supplied (we default the arg to -1), use the configured default connection timeout instead.
125 if (timeout == -1)
126 timeout = DefaultConnectionTimeout;
127
128 auto nanoconn = std::make_shared<nanodbc::connection>(
129 GetOdbcConnectionString(config.get()), timeout);
130
131 return std::make_shared<Connection>(nanoconn, config, timeout);
132}

134std::shared_ptr<PoolConnection> ConnectionManager::CreatePoolConnectionImpl(
135 modelUtil::DbType dbType, long timeout) noexcept(false)
136{
137 auto conn = PullPoolConnection(dbType);
138
139 // We pulled an existing connection; return a wrapper to manage it.
140 if (conn == nullptr)
141 {
142 // No available connection - spin up a new connection.
143 conn = CreateConnectionImpl(dbType, timeout);
144
145 // Failed to allocate.
146 if (conn == nullptr)
147 return nullptr;
148 }
149
150 return std::make_shared<PoolConnection>(dbType, conn);

153void ConnectionManager::ExpireUnusedPoolConnectionsImpl()
154{
155 time_t nowTime = time(nullptr);
156
157 std::lock_guard<std::mutex> lock(_connectionPoolLock);
158 for (auto itr = _connectionPoolMap.begin(); itr != _connectionPoolMap.end();)
159 {
160 modelUtil::DbType dbType = itr->first;
161
162 // We should always keep 1 (the first) available.
163 auto nextItr = std::next(itr);
164 while (nextItr != _connectionPoolMap.end() && nextItr->first == dbType)
165 {
166 ConnectionInfo& connInfo = nextItr->second;
167
168 // Expire any others that haven't been used in the specified time.
169 if ((nowTime - connInfo.LastUsedTime) >= PoolConnectionTimeout)
170 nextItr = _connectionPoolMap.erase(nextItr);
171 else
172 ++nextItr;
173 }
174
175 itr = nextItr;
176 }
177}

179std::shared_ptr<Connection> ConnectionManager::PullPoolConnection(
180 modelUtil::DbType dbType) noexcept(false)
181{
182 std::lock_guard<std::mutex> lock(_connectionPoolLock);
183
184 auto itr = _connectionPoolMap.find(dbType);
185 if (itr == _connectionPoolMap.end())
186 return nullptr;
187
188 ConnectionInfo& connInfo = itr->second;
189 auto conn = connInfo.Conn;
190 _connectionPoolMap.erase(itr);
191 return conn;
192}
